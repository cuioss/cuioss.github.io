001package io.cui.core.uimodel.field;
002
003import java.io.Serializable;
004
005/**
006 * Represents a field that is computed dynamically by the backend.
007 * This is useful for dynamic forms and provide possibility to reset to initialized value.
008 *
009 * @author Eugen Fischer
010 * @param <T> bounded type must be {@link Serializable}
011 */
012public interface TracedDynamicField<T extends Serializable> extends Serializable {
013
014    /**
015     * Depending on state indicates if field change is allowed
016     *
017     * @return true if field is changeable, false otherwise
018     */
019    boolean isEditable();
020
021    /**
022     * @return true if value is available
023     */
024    boolean isAvailable();
025
026    /**
027     * @return null if not available
028     */
029    T getValue();
030
031    /**
032     * Overwrite current value.
033     *
034     * @param newValue
035     * @throws IllegalStateException if value is not allowed to be change in state
036     */
037    void setValue(T newValue);
038
039    /**
040     * If the value has changed since being loaded. Used for optimized saving.
041     *
042     * @return If changed.
043     */
044    boolean isChanged();
045
046    /**
047     * Resets to the old value.
048     *
049     * @return the newly set old value
050     */
051    T resetValue();
052}
