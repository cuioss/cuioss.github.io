001package io.cui.core.uimodel.service;
002
003/**
004 * An optional service is a service that may be accessible or not. This usually defined by the
005 * concrete configuration or at runtime, e.g. if a remote service is not available
006 *
007 * @author Oliver Wolff
008 *
009 */
010public interface OptionalService {
011
012    /**
013     * @return boolean indicating whether the service is currently enabled or not.
014     */
015    default ServiceState getServiceState() {
016        return ServiceState.ACTIVE;
017    }
018
019    /**
020     * Shortcut for checking whether the service is available for the current call context.
021     *
022     * @return boolean indicating that the service is up and running and accessible for the current
023     *         logged in user
024     */
025    default boolean isServiceAvailable() {
026        return ServiceState.ACTIVE.equals(getServiceState());
027    }
028}
