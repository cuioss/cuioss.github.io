001package io.cui.core.uimodel.nameprovider;
002
003import java.io.Serializable;
004
005/**
006 * <p>
007 * Interface IDisplayNameProvider indicates entity which provide content that need to be presented
008 * on ui.
009 * </p>
010 * The resolving of the display name is not part of the entity itself.
011 *
012 * It's provide just the information that it should be resolved somehow and deliver the content
013 * which need to be resolved.
014 *
015 * If for example the source data is distributed over several attributes but needed as compound
016 * object it could be solved by creating a additional compound object which implements the interface
017 * IDisplayNameProvider. How this content should be displayed is a decision of converter which is
018 * registered for the compound type. In this way a lose coupling is reached by using standard
019 * mechanism of JSF.
020 *
021 * @author Eugen Fischer
022 * @param <T> bounded content parameter
023 */
024@FunctionalInterface
025public interface IDisplayNameProvider<T> extends Serializable {
026
027    /**
028     * @return actual content
029     */
030    T getContent();
031
032}
