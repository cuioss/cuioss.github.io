001package de.cuioss.uimodel.service;
002
003/**
004 * Indicates the state / availability of a service
005 *
006 * @author Oliver Wolff
007 *
008 */
009public enum ServiceState {
010
011    /** The service is active and can be accessed. */
012    ACTIVE,
013
014    /** Indicates the service is configured but currently not available. */
015    TEMPORARILY_UNAVAILABLE,
016
017    /**
018     * Indicates that the service is not configured at all or the configuration is incomplete or
019     * invalid.
020     */
021    NOT_CONFIGURED,
022
023    /**
024     * Indicates that the service is {@link #ACTIVE} but can can not be used by the currently logged
025     * in user
026     */
027    NOT_AVAILABLE_FOR_USER
028}
