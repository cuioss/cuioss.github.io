001package de.cuioss.uimodel.field;
002
003import java.io.Serializable;
004
005/**
006 * A {@linkplain TracedDynamicField} that is read-only per default, but can be unlocked upon demand.
007 *
008 * @author Eugen Fischer
009 * @param <T> bounded type must be {@link Serializable}
010 */
011public interface UnlockableTracedDynamicField<T extends Serializable> extends TracedDynamicField<T> {
012
013    /**
014     * execute action unlock edit mode
015     */
016    void unlockEditMode();
017
018    /**
019     * Retrieve information if edit mode was enforced.
020     *
021     * @return {@code true} if edit mode was overwritten, {@code false} otherwise
022     */
023    boolean isEditModeEnforced();
024
025    /**
026     * execute action to reset edit mode
027     */
028    void resetEditMode();
029}
