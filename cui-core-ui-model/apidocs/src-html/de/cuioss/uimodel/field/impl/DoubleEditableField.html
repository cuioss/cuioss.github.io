001package de.cuioss.uimodel.field.impl;
002
003import de.cuioss.uimodel.field.DynamicFieldType;
004
005/**
006 * An input field where the user can type in a {@link Double}.
007 *
008 * @author Oliver Wolff
009 */
010public class DoubleEditableField extends BaseDynamicField<Double> {
011
012    /** */
013    private static final long serialVersionUID = 4466539654104251395L;
014
015    /**
016     * C'tor.
017     *
018     * @param editable
019     *            If the input field is editable.
020     */
021    public DoubleEditableField(final boolean editable) {
022        super(editable, DynamicFieldType.DOUBLE);
023    }
024
025    /**
026     * C'tor.
027     *
028     * @param value
029     *            The value of the input field
030     * @param editable
031     *            If the value is editable
032     */
033    public DoubleEditableField(final Double value, final boolean editable) {
034        super(value, editable, DynamicFieldType.DOUBLE);
035    }
036
037}
