001package de.cuioss.uimodel.result;
002
003import de.cuioss.tools.logging.CuiLogger;
004import de.cuioss.uimodel.service.OptionalService;
005
006/**
007 * Default error codes for {@linkplain ResultObject} to be used like HTTP error codes.
008 */
009public enum ResultErrorCodes {
010
011    /**
012     * The element to be retrieved could not be found with the given identifier
013     */
014    NOT_FOUND,
015
016    /**
017     * The user does not have the necessary privileges
018     */
019    NOT_AUTHORIZED,
020
021    /**
022     * The is not authenticated
023     */
024    NOT_AUTHENTICATED,
025
026    /** The HttpRequest was not properly constructed. */
027    BAD_REQUEST,
028
029    /**
030     * The {@linkplain OptionalService} is not available
031     */
032    SERVICE_NOT_AVAILABLE,
033
034    /**
035     * A general runtime error occurred. No further information available
036     */
037    RUNTIME_ERROR;
038
039    private static final CuiLogger log = new CuiLogger(ResultErrorCodes.class);
040
041    /**
042     * Computes a concrete {@link ResultErrorCodes} from a given HttpCode
043     *
044     * @param httpCode
045     * @return the matching {@link ResultErrorCodes} if it can be determined,
046     *         {@link ResultErrorCodes#RUNTIME_ERROR} otherwise.
047     */
048    public static ResultErrorCodes parseHttpCode(int httpCode) {
049        log.trace("Parsing ResultErrorCode from httpCode '{}'", httpCode);
050
051        switch (httpCode) {
052            case 400:
053                return BAD_REQUEST;
054            case 401:
055                return NOT_AUTHENTICATED;
056            case 403:
057                return NOT_AUTHORIZED;
058            case 404:
059                return NOT_FOUND;
060            case 503:
061                return SERVICE_NOT_AVAILABLE;
062            default:
063                return RUNTIME_ERROR;
064        }
065    }
066}
