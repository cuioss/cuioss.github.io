001package de.cuioss.uimodel.model.conceptkey;
002
003import java.io.Serializable;
004import java.util.Map;
005import java.util.Set;
006
007import de.cuioss.uimodel.model.code.CodeType;
008
009/**
010 * An extension to {@link CodeType} providing additional information like {@link ConceptCategory}
011 * and augmented data. The augmented data is to be kept in a {@link Map}-like structure with
012 * {@link Serializable} keys and values. All implementations must be immutable because the approach
013 * is about implementing a type-safe-enum pattern, therefore the concrete instances are to be
014 * treated like constants. The concept is not about content data but defining data-structures.
015 *
016 * @author Oliver Wolff
017 */
018public interface ConceptKeyType extends CodeType, Comparable<ConceptKeyType> {
019
020    /**
021     * @return the {@link ConceptCategory} this {@link ConceptKeyType} is related to. Must never be
022     *         <code>null</code>
023     */
024    ConceptCategory getCategory();
025
026    /**
027     * @return a set of aliases for this code. Defines whether this code maps to another identifier
028     *         of the same category
029     */
030    Set<String> getAliases();
031
032    /**
033     * @param key identifying the content must not be null nor empty.
034     * @param defaultValue optional element that will be returned if there is no entry for the given
035     *            key.
036     * @return the object stored under the given key or the defaultValue, if nothing is stored under
037     *         the key.
038     */
039    String get(String key, String defaultValue);
040
041    /**
042     * @param key identifying the content must not be null nor empty.
043     * @return the object stored under the given key or null if nothing is stored under the key.
044     */
045    String get(String key);
046
047    /**
048     * @param key identifying the content must not be null nor empty.
049     * @return Returns <code>true</code> if the map contains a mapping for the specified key
050     *         <code>false</code> if not.
051     */
052    boolean containsKey(String key);
053
054    /**
055     * Returns a {@link Set} view of the mappings contained in the map. The set is never
056     * <code>null</code> but may be empty.
057     *
058     * @return a set view of the mappings contained in this map
059     */
060    Set<Map.Entry<String, String>> entrySet();
061}
