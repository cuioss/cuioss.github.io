001package de.cuioss.uimodel.model;
002
003import java.io.Serializable;
004
005/**
006 * Some elements display semantics like x elements of y. Therefore this interface is utilized.
007 *
008 * @author Oliver Wolff
009 */
010public interface RangeCounter extends Serializable {
011
012    /**
013     * @return count of elements
014     */
015    Integer getCount();
016
017    /**
018     * @return total count of elements
019     */
020    Integer getTotalCount();
021
022    /**
023     * @return boolean indicating whether there is a value available for {@link #getCount()}.
024     */
025    boolean isCountAvailable();
026
027    /**
028     * @return boolean indicating whether there is a value available for {@link #getTotalCount()}.
029     */
030    boolean isTotalCountAvailable();
031
032    /**
033     * @return boolean indicating whether there is only one value set.
034     */
035    boolean isSingleValueOnly();
036
037    /**
038     * @return boolean indicating whether there is no set.
039     */
040    boolean isEmpty();
041
042    /**
043     * @return boolean indicating whether all values are set.
044     */
045    boolean isComplete();
046}
