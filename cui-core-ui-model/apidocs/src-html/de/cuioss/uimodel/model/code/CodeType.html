001package de.cuioss.uimodel.model.code;
002
003import java.io.Serializable;
004import java.util.Locale;
005
006/**
007 * Abstracts any implementation of Code, CodeDto, CodedValueDto, AnnotatedCode etc.
008 * <p>
009 * The idea is use (resolve) any of this object without care about how the magic resolving
010 * is working.
011 * This approach provide a lose coupling from UI to BE. Additional advantage is the simple creation
012 * of mocks.
013 * </p>
014 * <p>
015 * <em>PRECONDITION</em> : appropriate use is only possible if you have knowledge about DAO pattern
016 * (data access object)
017 * </p>
018 * This interface doesn't make assumptions about the resolving strategies or caching feel free to
019 * use the power of this.
020 *
021 * @author Eugen Fischer
022 */
023public interface CodeType extends Serializable {
024
025    /**
026     * Provide translated value by using implementation defined resolving / fall-back logic.
027     *
028     * @param locale {@linkplain Locale} must not be null.
029     * @return resolved value if possible, {@code null} otherwise.
030     */
031    String getResolved(Locale locale);
032
033    /**
034     * @return unique id must be always defined
035     */
036    String getIdentifier();
037}
