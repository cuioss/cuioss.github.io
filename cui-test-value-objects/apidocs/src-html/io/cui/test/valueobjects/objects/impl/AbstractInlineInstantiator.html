001package io.cui.test.valueobjects.objects.impl;
002
003import java.util.Collections;
004import java.util.List;
005
006import io.cui.test.valueobjects.api.object.ObjectTestContract;
007import io.cui.test.valueobjects.objects.ParameterizedInstantiator;
008import io.cui.test.valueobjects.objects.RuntimeProperties;
009import io.cui.test.valueobjects.property.PropertyMetadata;
010import io.cui.test.valueobjects.property.PropertySupport;
011
012/**
013 * {@link ParameterizedInstantiator} for cases you actually are not able to use the testing
014 * Infrastructure but want to benefit of the testing of the {@link ObjectTestContract}s
015 *
016 * @author Oliver Wolff
017 * @param <T> identifying the type of object to be instantiated
018 */
019public abstract class AbstractInlineInstantiator<T> implements ParameterizedInstantiator<T> {
020
021    /** "Properties must not be null, but may be empty". */
022    public static final String PROPERTIES_MUST_NOT_BE_NULL = "Properties must not be null, but may be empty";
023
024    @Override
025    public T newInstance(final List<PropertySupport> properties,
026            final boolean generatePropertyValues) {
027        return any();
028    }
029
030    @Override
031    public T newInstance(final List<PropertyMetadata> properties) {
032        return any();
033    }
034
035    @Override
036    public RuntimeProperties getRuntimeProperties() {
037        return new RuntimeProperties(Collections.emptyList());
038    }
039
040    @Override
041    public T newInstanceMinimal() {
042        return any();
043    }
044
045    @Override
046    public T newInstanceFull() {
047        return any();
048    }
049
050    protected abstract T any();
051}
