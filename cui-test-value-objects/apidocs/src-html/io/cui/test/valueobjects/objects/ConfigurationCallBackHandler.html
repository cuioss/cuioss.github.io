001package io.cui.test.valueobjects.objects;
002
003/**
004 * Defines a callback-method in case of further configuration of concrete
005 * test-objects. After initialization and prior to testing the method {@link #configure(Object)}
006 * will be called allowing the concrete test-class to do some specific configuration.
007 *
008 * @author Oliver Wolff
009 * @param <T> identifying the type to be configured
010 */
011public interface ConfigurationCallBackHandler<T> {
012
013    /**
014     * Callback method for configuring a concrete created test-object. The default implementation is
015     * NOOP
016     *
017     * @param toBeConfigured must never be null
018     */
019    default void configure(final T toBeConfigured) {
020    }
021
022}
