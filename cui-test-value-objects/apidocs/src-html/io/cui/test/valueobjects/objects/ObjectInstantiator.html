001package io.cui.test.valueobjects.objects;
002
003import io.cui.test.valueobjects.objects.impl.DefaultInstantiator;
004
005/**
006 * An {@link ObjectInstantiator} creates objects without further parameter, e.g. the
007 * {@link DefaultInstantiator} uses the Default-Constructor
008 *
009 * @author Oliver Wolff
010 * @param <T>
011 */
012public interface ObjectInstantiator<T> {
013
014    /**
015     * @return a newly created instance
016     */
017    T newInstance();
018
019    /**
020     * @return the type of object to be generated
021     */
022    Class<T> getTargetClass();
023
024}
