001package io.cui.test.valueobjects.objects.impl;
002
003import static io.cui.test.valueobjects.objects.impl.ExceptionHelper.extractCauseMessageFromThrowable;
004import static java.util.Objects.requireNonNull;
005
006import java.lang.reflect.Constructor;
007import java.lang.reflect.InvocationTargetException;
008import java.util.ArrayList;
009import java.util.List;
010
011import io.cui.test.valueobjects.objects.ParameterizedInstantiator;
012import io.cui.test.valueobjects.objects.RuntimeProperties;
013import io.cui.tools.logging.CuiLogger;
014
015/**
016 * This {@link ParameterizedInstantiator} uses a constructor derived by the given
017 * {@link RuntimeProperties#getAllProperties()} in order to instantiate {@link Object}s
018 *
019 * @param <T> identifying the type of objects to be created
020 *
021 * @author Oliver Wolff
022 */
023public class ConstructorBasedInstantiator<T> extends AbstractOrderedArgsInstantiator<T> {
024
025    private final Constructor<T> constructor;
026
027    private static final CuiLogger log = new CuiLogger(ConstructorBasedInstantiator.class);
028
029    /**
030     * Constructor.
031     *
032     * @param type identifying the actual type to be instantiated, must not be null
033     * @param runtimeProperties must not be null. defines the attributes in the exact order
034     *            to be used for the constructor:
035     *            {@link RuntimeProperties#getAllProperties()}
036     */
037    public ConstructorBasedInstantiator(final Class<T> type,
038            final RuntimeProperties runtimeProperties) {
039
040        super(runtimeProperties);
041        requireNonNull(type);
042
043        final List<Class<?>> parameter = new ArrayList<>();
044        runtimeProperties.getAllProperties()
045                .forEach(meta -> parameter.add(meta.resolveActualClass()));
046        try {
047            if (parameter.isEmpty()) {
048                this.constructor = type.getConstructor();
049            } else {
050                this.constructor = type.getConstructor(toClassArray(parameter));
051            }
052            requireNonNull(this.constructor,
053                    "Unable to find a constructor with signature " + parameter);
054        } catch (NoSuchMethodException | SecurityException e) {
055            final var message = new StringBuilder("Unable to find a constructor with signature ")
056                    .append(parameter).append(", for type ").append(type.getName()).toString();
057            log.error(message, e);
058            for (Constructor<?> tempConstructor : type.getConstructors()) {
059                log.error("Found constructor: {}", tempConstructor.toGenericString());
060            }
061            if (0 == type.getConstructors().length) {
062                log.error("No public constructor found!");
063            }
064            throw new AssertionError(message);
065        }
066    }
067
068    @Override
069    protected T doInstantiate(final Object... args) {
070        try {
071            return this.constructor.newInstance(args);
072        } catch (InstantiationException | IllegalAccessException | IllegalArgumentException
073                | InvocationTargetException e) {
074            final var message = new StringBuilder("Unable to invoke constructor ")
075                    .append(", due to ").append(extractCauseMessageFromThrowable(e)).toString();
076            throw new AssertionError(message, e);
077        }
078    }
079
080    @Override
081    public String toString() {
082        final var builder = new StringBuilder(getClass().getName());
083        builder.append("\nConstructor: ").append(this.constructor);
084        builder.append("\nProperty Configuration: ")
085                .append(getRuntimeProperties().toString());
086        return builder.toString();
087    }
088}
