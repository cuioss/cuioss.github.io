173     * Creates a copy of this instance. In addition it tries to create a generatedValue that is not
174     * equal to the contained one. It will try this 50 times and will then throw an
175     * {@link AssertionError}
176     *
177     * @return the created copy
178     */
179    public PropertySupport createCopyWithNonEqualValue() {
180        final var support = new PropertySupport(propertyMetadata);
181        if (null == getGeneratedValue()) {
182            support.generateTestValue();
183            return support;
184        }
185        var times = 0;
186        final var initialTestValue = getGeneratedValue();
187        while (true) {
188            final var otherValue = support.generateTestValue();
189            if (!initialTestValue.equals(otherValue)) {
190                break;
191            }
192            times++;
193            if (ENTROPY_GUARD == times) {
194                throw new AssertionError("Unable to create non equal test-value for " + propertyMetadata);
195            }
196        }
197        return support;
198    }
