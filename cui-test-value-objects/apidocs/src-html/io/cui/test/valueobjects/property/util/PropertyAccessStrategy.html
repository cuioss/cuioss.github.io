001package io.cui.test.valueobjects.property.util;
002
003import static io.cui.tools.collect.CollectionLiterals.mutableList;
004import static org.junit.jupiter.api.Assertions.assertNotNull;
005
006import java.lang.reflect.InvocationTargetException;
007import java.lang.reflect.Method;
008import java.util.Collection;
009import java.util.List;
010
011import io.cui.test.valueobjects.objects.impl.ExceptionHelper;
012import io.cui.test.valueobjects.property.PropertyMetadata;
013import io.cui.test.valueobjects.property.impl.BuilderMetadata;
014import io.cui.test.valueobjects.property.impl.BuilderMetadata.BuilderMetadataBuilder;
015import io.cui.tools.logging.CuiLogger;
016import io.cui.tools.property.PropertyUtil;
017import io.cui.tools.reflect.MoreReflection;
018
019/**
020 * Defines different ways for reading / writing properties.
021 *
022 * @author Oliver Wolff
023 */
024
025public enum PropertyAccessStrategy {
026
027    /**
028     * Reads and writes property according to the JavaBean-Spec. It uses {@link PropertyUtil} to do
029     * so. It acts as part of JUnit testing, therefore it will translate many of the more technical
030     * Exceptions to corresponding {@link AssertionError}
031     *
032     * @author Oliver Wolff
033     */
034    BEAN_PROPERTY {
035
036        @Override
037        public Object writeProperty(final Object target,
038                final PropertyMetadata propertyMetadata,
039                final Object propertyValue) {
040            assertNotNull(target, TARGET_MUST_NOT_BE_NULL);
041            assertNotNull(target, PROPERTY_METADATA_MUST_NOT_BE_NULL);
042            try {
043                PropertyUtil.writeProperty(target, propertyMetadata.getName(), propertyValue);
044                return target;
045            } catch (IllegalArgumentException | IllegalStateException e) {
046                throw new AssertionError(String.format(UNABLE_TO_SET_PROPERTY, propertyMetadata.getName(),
047                        ExceptionHelper.extractCauseMessageFromThrowable(e)), e);
048            }
049
050        }
051
052        @Override
053        public Object readProperty(final Object target,
054                final PropertyMetadata propertyMetadata) {
055            assertNotNull(target, TARGET_MUST_NOT_BE_NULL);
056            assertNotNull(target, PROPERTY_METADATA_MUST_NOT_BE_NULL);
057            try {
058                return PropertyUtil.readProperty(target, propertyMetadata.getName());
059            } catch (IllegalArgumentException | IllegalStateException e) {
060                throw new AssertionError(String.format(UNABLE_TO_READ_PROPERTY, propertyMetadata.getName(),
061                        ExceptionHelper.extractCauseMessageFromThrowable(e)), e);
062            }
063        }
064    },
065    /**
066     * In some cases the builder supports multiple ways to fill {@link Collection} based
067     * elements, e.g. there is a field with the structure
068     *
069     * <pre>
070     * <code>
071     *     private List&lt;String&gt; name;
072     * </code>
073     * </pre>
074     *
075     * There can be two methods to fill this elements in the builder:
076     *
077     * <pre>
078     * <code>
079     *  public Builder names(List&lt;String&gt; names);
080     *  public Builder names(String name);
081     * </code>
082     * </pre>
083     *
084     * This strategy writes the property using <em>both</em> methods. Therefore it uses
085     * {@link BuilderMetadata#getBuilderSingleAddMethodName()} in order to find the
086     * single addMethod. The plural add method is supposed to be the name of the property itself
087     * therefore derived by {@link BuilderMetadata#getBuilderAddMethodName()}.
088     * In case there is different methodName for adding, e.g.
089     *
090     * <pre>
091     * <code>
092     *  public Builder names(List&lt;String&gt; names);
093     *  public Builder name(String name);
094     * </code>
095     * </pre>
096     *
097     * you can specify the method-name explicitly by using
098     * {@link BuilderMetadataBuilder#builderSingleAddMethodName(String)}
099     * <p>
100     * The read method delegates to {@link PropertyAccessStrategy#BEAN_PROPERTY} because it
101     * can not be read from an actual builder but from the later created bean.
102     * </p>
103     *
104     * @author Oliver Wolff
105     */
106    BUILDER_COLLECTION_AND_SINGLE_ELEMENT {
107
108        @Override
109        public Object writeProperty(final Object target,
110                final PropertyMetadata propertyMetadata,
111                final Object propertyValue) {
112            if (!(propertyValue instanceof Iterable)) {
113                throw new AssertionError(
114                        "Invalid valueType given, must be at least Iterable, but was " + propertyValue);
115            }
116            final Iterable<?> iterable = (Iterable<?>) propertyValue;
117            final List<?> elements = mutableList(iterable);
118            BuilderMetadata builderMetadata;
119            if (!(propertyMetadata instanceof BuilderMetadata)) {
120                builderMetadata = BuilderMetadata.wrapFromMetadata(propertyMetadata);
121            } else {
122                builderMetadata = (BuilderMetadata) propertyMetadata;
123            }
124            try {
125                if (!elements.isEmpty()) {
126                    final Object singleElement = elements.iterator().next();
127
128                    final var writeAddMethod = target.getClass().getMethod(
129                            builderMetadata.getBuilderSingleAddMethodName(), propertyMetadata.getPropertyClass());
130                    writeAddMethod.invoke(target, singleElement);
131
132                    // Remove the element from the elements list
133                    elements.remove(singleElement);
134                }
135                final var writeCollectionMethod =
136                    determineCollectionWriteMethod(target, propertyMetadata, builderMetadata);
137
138                // Now write the remaining elements
139                return writeCollectionMethod.invoke(target,
140                        propertyMetadata.getCollectionType().wrapToIterable(elements));
141            } catch (NoSuchMethodException | SecurityException | IllegalAccessException | IllegalArgumentException
142                    | InvocationTargetException e) {
143                throw new AssertionError(String.format(UNABLE_TO_SET_PROPERTY, propertyMetadata.getName(),
144                        ExceptionHelper.extractCauseMessageFromThrowable(e)), e);
145            }
146
147        }
148
149        private Method determineCollectionWriteMethod(final Object target, final PropertyMetadata propertyMetadata,
150                BuilderMetadata builderMetadata)
151            throws NoSuchMethodException {
152            try {
153                return target.getClass().getMethod(builderMetadata.getBuilderAddMethodName(),
154                        propertyMetadata.getCollectionType().getIterableType());
155            } catch (NoSuchMethodException e) {
156                // Noop -> Assuming collection-parameter
157            }
158            return target.getClass().getMethod(builderMetadata.getBuilderAddMethodName(),
159                    CollectionType.COLLECTION.getIterableType());
160        }
161
162        @Override
163        public Object readProperty(final Object target,
164                final PropertyMetadata propertyMetadata) {
165            return PropertyAccessStrategy.BEAN_PROPERTY.readProperty(target, propertyMetadata);
166        }
167    },
168    /**
169     * Writes a property in a builder using
170     * {@link BuilderMetadata#getBuilderAddMethodName()} to
171     * determine the correct write method. The parameter type is exactly the same as defined at
172     * {@link PropertyMetadata#getPropertyClass()}. The read method delegates to
173     * {@link PropertyAccessStrategy#BEAN_PROPERTY} because it
174     * can not be read from an actual builder but from the later created bean.
175     *
176     * @author Oliver Wolff
177     */
178    BUILDER_DIRECT {
179
180        @Override
181        public Object writeProperty(final Object target,
182                final PropertyMetadata propertyMetadata,
183                final Object propertyValue) {
184            BuilderMetadata builderMetadata;
185            if (!(propertyMetadata instanceof BuilderMetadata)) {
186                builderMetadata = BuilderMetadata.wrapFromMetadata(propertyMetadata);
187            } else {
188                builderMetadata = (BuilderMetadata) propertyMetadata;
189            }
190            try {
191                final var writeMethod = target.getClass().getMethod(builderMetadata.getBuilderAddMethodName(),
192                        propertyMetadata.resolveActualClass());
193                return writeMethod.invoke(target, propertyValue);
194            } catch (NoSuchMethodException | SecurityException | IllegalAccessException | IllegalArgumentException
195                    | InvocationTargetException e) {
196                var message = String.format(UNABLE_TO_SET_PROPERTY, propertyMetadata.getName(),
197                        ExceptionHelper.extractCauseMessageFromThrowable(e));
198                new CuiLogger(getClass()).error(message);
199                throw new AssertionError(message, e);
200            }
201        }
202
203        @Override
204        public Object readProperty(final Object target,
205                final PropertyMetadata propertyMetadata) {
206            return PropertyAccessStrategy.BEAN_PROPERTY.readProperty(target, propertyMetadata);
207        }
208    },
209    /**
210     * This strategy is for cases where {@link PropertyUtil} is not capable of writing an
211     * attribute. This is usually the case for elements that defined a fluent-api (not void as
212     * return value). The read method delegates to {@link PropertyAccessStrategy#BEAN_PROPERTY}
213     */
214    FLUENT_WRITER {
215
216        @Override
217        public Object writeProperty(Object target, PropertyMetadata propertyMetadata, Object propertyValue) {
218            var writeMethod = MoreReflection.retrieveWriteMethod(target.getClass(),
219                    propertyMetadata.getName(), propertyMetadata.resolveActualClass());
220            if (!writeMethod.isPresent()) {
221                throw new AssertionError(String.format(UNABLE_TO_SET_PROPERTY, propertyMetadata.getName(),
222                        "No write-method could be found"));
223            }
224            try {
225                return writeMethod.get().invoke(target, propertyValue);
226            } catch (SecurityException | IllegalAccessException | IllegalArgumentException
227                    | InvocationTargetException e) {
228                throw new AssertionError(String.format(UNABLE_TO_SET_PROPERTY, propertyMetadata.getName(),
229                        ExceptionHelper.extractCauseMessageFromThrowable(e)), e);
230            }
231        }
232
233        @Override
234        public Object readProperty(final Object target,
235                final PropertyMetadata propertyMetadata) {
236            return PropertyAccessStrategy.BEAN_PROPERTY.readProperty(target, propertyMetadata);
237        }
238
239    };
240
241    private static final String UNABLE_TO_READ_PROPERTY =
242        "Unable to read property '%s' because of '%s'";
243    private static final String TARGET_MUST_NOT_BE_NULL = "target must not be null";
244    private static final String PROPERTY_METADATA_MUST_NOT_BE_NULL =
245        "propertyMetadata must not be null";
246    private static final String UNABLE_TO_SET_PROPERTY =
247        "Unable to set property '%s' because of '%s'";
248
249    /**
250     * Writes the property into the given target;
251     *
252     * @param target to be written to.
253     * @param propertyMetadata identifying the concrete property, must not be null
254     * @param propertyValue to be set, may be null
255     * @return the modified object
256     * @throws AssertionError in case the property can not be written.
257     */
258    public abstract Object writeProperty(final Object target,
259            final PropertyMetadata propertyMetadata,
260            final Object propertyValue);
261
262    /**
263     * Reads the property from the given target;
264     *
265     * @param target to be written to.
266     * @param propertyMetadata identifying the concrete property, must not be null
267     * @return the read property, may be null
268     * @throws AssertionError in case the property can not be read.
269     */
270    public abstract Object readProperty(final Object target,
271            final PropertyMetadata propertyMetadata);
272
273}
