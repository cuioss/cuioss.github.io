001package io.cui.test.valueobjects.api.contracts;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * If used on ValueObjectTest this test verifies a copy-constructor.
010 * Depending on its configuration it tests the individual properties to be copied correctly and of
011 * course the {@link Object#equals(Object)} method, see {@link #useObjectEquals()}. It can only be
012 * used if at least one other verification contract is there.
013 *
014 * @author Oliver Wolff
015 */
016@Retention(RetentionPolicy.RUNTIME)
017@Target({ ElementType.TYPE })
018public @interface VerifyCopyConstructor {
019
020    /**
021     * @return an array of properties, identified by their names that are not to be considered for
022     *         this test: black-list
023     */
024    String[] exclude() default {};
025
026    /**
027     * @return an array of properties, identified by their names that are to be considered for
028     *         this test: white-list
029     */
030    String[] of() default {};
031
032    /**
033     * @return the actual type needed for the copy-constructor. It is only needed if it differs from
034     *         the type of the bean under test. {@link VerifyCopyConstructor} acts as {@code null}
035     *         value
036     */
037    Class<?> argumentType() default VerifyCopyConstructor.class;
038
039    /**
040     * @return boolean indicating whether to {@link Object#equals(Object)} as well, defaults to
041     *         {@code true}
042     */
043    boolean useObjectEquals() default true;
044
045    /**
046     * @return boolean indicating whether to implicitly check whether the copy is a deep instead of
047     *         a shallow copy, defaults to {@code true}. See {@link #verifyDeepCopyIgnore()} as well
048     */
049    boolean verifyDeepCopy() default false;
050
051    /**
052     * @return an array of properties, identified by their names that are to be ignored by checking
053     *         the deep copy, see {@link #verifyDeepCopy()}
054     */
055    String[] verifyDeepCopyIgnore() default {};
056}
