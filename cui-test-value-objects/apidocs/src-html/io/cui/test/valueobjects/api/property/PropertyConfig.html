001package io.cui.test.valueobjects.api.property;
002
003import static java.lang.annotation.ElementType.TYPE;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Repeatable;
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009
010import io.cui.test.generator.TypedGenerator;
011import io.cui.test.valueobjects.generator.dynamic.DynamicTypedGenerator;
012import io.cui.test.valueobjects.property.PropertyMetadata;
013import io.cui.test.valueobjects.property.util.AssertionStrategy;
014import io.cui.test.valueobjects.property.util.CollectionType;
015import io.cui.test.valueobjects.property.util.PropertyAccessStrategy;
016import io.cui.tools.property.PropertyMemberInfo;
017import io.cui.tools.property.PropertyReadWrite;
018
019/**
020 * While the test classes are capable of auto-detecting JavaProperties you need to adjust them from
021 * time to time. With this annotation you can do this.
022 *
023 * @author Oliver Wolff
024 */
025@Retention(RUNTIME)
026@Target(TYPE)
027@Repeatable(PropertyConfigs.class)
028public @interface PropertyConfig {
029
030    /**
031     * Identifies the name of the property
032     *
033     * @return the actual name of the property, must never be null nor empty
034     * @see {@link PropertyMetadata#getName()}.
035     */
036    String name();
037
038    /**
039     * @return the type of the property. This can either be the actual type, in case
040     *         {@link PropertyMetadata#getCollectionType()} is {@link CollectionType#NO_ITERABLE},
041     *         the component-type in case of {@link CollectionType#ARRAY_MARKER} or the type
042     *         argument for a collection for the other {@link CollectionType}s, see
043     *         {@link PropertyMetadata#next()} and
044     *         {@link PropertyMetadata#resolveActualClass()}
045     * @see {@link PropertyMetadata#getPropertyClass()}.
046     */
047    Class<?> propertyClass();
048
049    /**
050     * @return the wrapped {@link TypedGenerator} to dynamically create properties. If it is not set
051     *         {@link DynamicTypedGenerator} will be chosen
052     * @see {@link PropertyMetadata#getGenerator()}.
053     */
054    @SuppressWarnings("rawtypes")
055    Class<? extends TypedGenerator> generator() default DynamicTypedGenerator.class;
056
057    /**
058     * @return boolean indicating whether the property defines a default value, defaults to false
059     * @see {@link PropertyMetadata#isDefaultValue()}.
060     */
061    boolean defaultValue() default false;
062
063    /**
064     * @return boolean indicating whether the given property is required, defaults to false
065     * @see {@link PropertyMetadata#isRequired()}.
066     */
067    boolean required() default false;
068
069    /**
070     * @return The {@link PropertyMemberInfo}, defaults to {@link PropertyMemberInfo#DEFAULT}
071     * @see {@link PropertyMetadata#getPropertyMemberInfo()}.
072     */
073    PropertyMemberInfo propertyMemberInfo() default PropertyMemberInfo.DEFAULT;
074
075    /**
076     * In case there is a collectionType defined the generated values will implicitly wrapped in
077     * the corresponding collection class defined by that wrapper, defaults to
078     * {@link CollectionType#NO_ITERABLE}.
079     *
080     * @return the {@link CollectionType}
081     * @see {@link PropertyMetadata#getCollectionType()}.
082     */
083    CollectionType collectionType() default CollectionType.NO_ITERABLE;
084
085    /**
086     * @return whether the property can be read or written, default to
087     *         {@link PropertyReadWrite#READ_WRITE}
088     * @see {@link PropertyMetadata#getPropertyReadWrite()}.
089     */
090    PropertyReadWrite propertyReadWrite() default PropertyReadWrite.READ_WRITE;
091
092    /**
093     * Defines different ways for reading / writing properties.
094     *
095     * @return the {@link PropertyAccessStrategy}, defaults to
096     *         {@link PropertyAccessStrategy#BEAN_PROPERTY}
097     * @see {@link PropertyMetadata#getPropertyAccessStrategy()}.
098     */
099    PropertyAccessStrategy propertyAccessStrategy() default PropertyAccessStrategy.BEAN_PROPERTY;
100
101    /**
102     * Defines the the way how to deal with equality regarding
103     * PropertySupport.assertValueSet(Object)
104     *
105     * @return the {@link AssertionStrategy}, defaults to {@link AssertionStrategy#DEFAULT}
106     */
107    AssertionStrategy assertionStrategy() default AssertionStrategy.DEFAULT;
108}
