001package io.cui.test.valueobjects.api.generator;
002
003import static java.lang.annotation.ElementType.TYPE;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.io.Serializable;
007import java.lang.annotation.Repeatable;
008import java.lang.annotation.Retention;
009import java.lang.annotation.Target;
010
011import io.cui.test.generator.TypedGenerator;
012
013/**
014 * For some cases it is simpler to give a hint for a certain type instead of declaring
015 * {@link PropertyGenerator}.
016 * e.g. Your class to be tested needs instances of certain Interface that need to be
017 * {@link Serializable} but this is not declared at interface level. Therefore this hint may suffice
018 * in case you provide a concrete implementation that can be instantiated. The test-framework will
019 * than always use a {@link TypedGenerator} for all the tests.
020 *
021 * @author Oliver Wolff
022 */
023@Retention(RUNTIME)
024@Target(TYPE)
025@Repeatable(PropertyGeneratorHints.class)
026public @interface PropertyGeneratorHint {
027
028    /**
029     * @return The type of object that are declared at field and or method / constructor level
030     */
031    Class<?> declaredType();
032
033    /**
034     * @return the actual implementation type to be used for the {@link #declaredType()}. It should
035     *         provide a publicly accessible constructor
036     */
037    Class<?> implementationType();
038
039}
