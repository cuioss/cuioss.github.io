001package io.cui.test.valueobjects.api.generator;
002
003import static java.lang.annotation.ElementType.TYPE;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Repeatable;
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009
010import io.cui.test.generator.TypedGenerator;
011
012/**
013 * Used for adding an additional {@link TypedGenerator} for the actual
014 * test-class. This can be done defining one or more {@link TypedGenerator} as
015 * class, see {@link #value()}.
016 *
017 * @author Oliver Wolff
018 */
019@Retention(RUNTIME)
020@Target(TYPE)
021@Repeatable(PropertyGenerators.class)
022public @interface PropertyGenerator {
023
024    /**
025     * @return one or an array of {@link TypedGenerator}. This is the standard usage for this
026     *         annotation.
027     */
028    @SuppressWarnings("java:S1452")
029    Class<? extends TypedGenerator<?>>[] value() default {};
030
031}
