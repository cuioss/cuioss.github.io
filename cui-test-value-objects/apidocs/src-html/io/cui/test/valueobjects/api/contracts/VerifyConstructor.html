001package io.cui.test.valueobjects.api.contracts;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Repeatable;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008import java.util.Collection;
009
010import io.cui.test.valueobjects.api.property.PropertyConfig;
011import io.cui.test.valueobjects.property.PropertyMetadata;
012import io.cui.test.valueobjects.property.util.AssertionStrategy;
013import io.cui.tools.property.PropertyMemberInfo;
014import io.cui.tools.property.PropertyReadWrite;
015
016/**
017 * If used on ValueObjectTest this test checks / tests constructors.
018 * <p>
019 * In order to define the constructor-args in the correct order you need to pass them using
020 * {@link #of()}
021 * </p>
022 * <p>
023 * As default it assumes the individual arguments to be optional, saying null is allowed. This can
024 * be controlled by {@link #required()}
025 * </p>
026 *
027 * @author Oliver Wolff
028 */
029@Retention(RetentionPolicy.RUNTIME)
030@Target({ ElementType.TYPE })
031@Repeatable(VerifyConstructors.class)
032public @interface VerifyConstructor {
033
034    /**
035     * @return an array of properties, identified by their names that are to be considered for
036     *         this test: Caution: The order of the names must exactly match the order of the
037     *         attributes of corresponding constructor.
038     */
039    String[] of();
040
041    /**
042     * @return an array of properties, identified by their names that are to be treated as required
043     *         properties, see {@link PropertyMetadata#isRequired()}. it is used
044     */
045    String[] required() default {};
046
047    /**
048     * @return boolean indicating whether all attribute defined at {@link #of()} are to be treated
049     *         as required attributes. Defaults to {@code false}
050     */
051    boolean allRequired() default false;
052
053    /**
054     * @return an array of properties, identified by their names that are to be treated as transient
055     *         properties, see {@link PropertyMemberInfo#TRANSIENT}
056     */
057    String[] transientProperties() default {};
058
059    /**
060     * @return an array of properties, identified by their names that are to be treated as having a
061     *         default values, see {@link PropertyMetadata#isDefaultValue()}
062     */
063    String[] defaultValued() default {};
064
065    /**
066     * @return an array of properties, identified by their names that are to be treated as being
067     *         read-only, see {@link PropertyReadWrite#READ_ONLY}, usually used in conjunction with
068     *         {@link #defaultValued()}
069     */
070    String[] readOnly() default {};
071
072    /**
073     * @return an array of properties, identified by their names that are to be treated as being
074     *         write-only, see {@link PropertyReadWrite#WRITE_ONLY}, usually used in cases where a
075     *         property to be written will result in other properties but itself can not be accessed
076     *         directly
077     */
078    String[] writeOnly() default {};
079
080    /**
081     * @return an array of properties, identified by their names representing at least a
082     *         {@link Collection} that are to be asserted ignoring the concrete order, see
083     *         {@link PropertyConfig#assertionStrategy()} and
084     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
085     *         always respect / assert the same order of elements.
086     */
087    String[] assertUnorderedCollection() default {};
088}
