001package io.cui.test.valueobjects.api.contracts;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007import java.util.Collection;
008
009import io.cui.test.valueobjects.api.property.PropertyBuilderConfig;
010import io.cui.test.valueobjects.api.property.PropertyConfig;
011import io.cui.test.valueobjects.contract.BuilderContractImpl;
012import io.cui.test.valueobjects.property.PropertyMetadata;
013import io.cui.test.valueobjects.property.impl.PropertyMetadataImpl;
014import io.cui.test.valueobjects.property.impl.PropertyMetadataImpl.PropertyMetadataBuilder;
015import io.cui.test.valueobjects.property.util.AssertionStrategy;
016import io.cui.tools.property.PropertyMemberInfo;
017import io.cui.tools.property.PropertyReadWrite;

019/**
020 * If used on ValueObjectTest this test checks / tests builder of value objects.
021 * <p>
022 * As default the test assumes a static factory method with the name "builder", e.g.
023 * {@link PropertyMetadataImpl#builder()} being present on the
024 * type to be tested. This can be controlled using {@link #builderFactoryMethodName()},
025 * {@link #builderFactoryProvidingClass()} and {@link #builderMethodName()}. In case you have a
026 * builder that can be instantiated directly by using a no-args public constructor directly you can
027 * use {@link #builderClass()} instead.
028 * </p>
029 * <p>
030 * The builder methods for the individual properties are assumed to use the property-name only as
031 * method-name, e.g. {@link PropertyMetadataBuilder#propertyClass(Class)}. In case you prefer a
032 * prefix you can either configure it using {@link #methodPrefix()} or can do some more fine grained
033 * configuration with {@link PropertyBuilderConfig}
034 * </p>
035 *
036 * @author Oliver Wolff
037 */
038@Retention(RetentionPolicy.RUNTIME)
039@Target({ ElementType.TYPE })
040public @interface VerifyBuilder {
041
042    /**
043     * @return the name of the factory method for creating the builder instances. It is assumed that
044     *         it is parameter-free static method. If not set it defaults to
045     *         {@value BuilderContractImpl#DEFAULT_BUILDER_FACTORY_METHOD_NAME}
046     */
047    String builderFactoryMethodName() default BuilderContractImpl.DEFAULT_BUILDER_FACTORY_METHOD_NAME;
048
049    /**
050     * @return the name of the build method on the {@link #builderClass()}. It is assumed that
051     *         it is parameter-free method on the builderType and returns instances
052     *         of the type on unit class level of ValueObjectTest. Defaults to
053     *         {@value BuilderContractImpl#DEFAULT_BUILD_METHOD_NAME}
054     */
055    String builderMethodName() default BuilderContractImpl.DEFAULT_BUILD_METHOD_NAME;
056
057    /**
058     * This attribute defines the actual type of the concrete builder. The default value
059     * {@link VerifyBuilder} acts as {@code null} value. If it is set it will be used instead of any
060     * factory. It is assumed to have a no args public constructor.
061     *
062     * @return the builder-class
063     */
064    Class<?> builderClass() default VerifyBuilder.class;
065
066    /**
067     * Optional attribute. The default assumes that the factory method for creating the concrete
068     * builder is located at the object under test. This method is only necessary if it is located
069     * on another type, e.g the actual unit-test-class. The default value
070     * {@link VerifyBuilder} acts as null value
071     *
072     * @return the type where the builderFactoryMethod is defined, the name of this method is
073     *         defined within {@link #builderFactoryMethodName()}
074     */
075    Class<?> builderFactoryProvidingClass() default VerifyBuilder.class;
076
077    /**
078     * @return an array of properties, identified by their names that are not to be considered for
079     *         this test: black-list
080     */
081    String[] exclude() default {};
082
083    /**
084     * @return an array of properties, identified by their names that are to be considered for
085     *         this test: white-list
086     */
087    String[] of() default {};
088
089    /**
090     * @return an array of properties, identified by their names that are to be treated as required
091     *         properties, see {@link PropertyMetadata#isRequired()}
092     */
093    String[] required() default {};
094
095    /**
096     * @return an array of properties, identified by their names that are to be treated as transient
097     *         properties, see {@link PropertyMemberInfo#TRANSIENT}
098     */
099    String[] transientProperties() default {};
100
101    /**
102     * @return an array of properties, identified by their names that are to be treated as having a
103     *         default values, see {@link PropertyMetadata#isDefaultValue()}
104     */
105    String[] defaultValued() default {};
106
107    /**
108     * @return an array of properties, identified by their names that are to be treated as being
109     *         read-only, see {@link PropertyReadWrite#READ_ONLY}, usually used in conjunction with
110     *         {@link #defaultValued()}
111     */
112    String[] readOnly() default {};
113
114    /**
115     * @return an array of properties, identified by their names that are to be treated as being
116     *         write-only, see {@link PropertyReadWrite#WRITE_ONLY}, usually used in cases where a
117     *         property to be written will result in other properties but itself can not be accessed
118     *         directly
119     */
120    String[] writeOnly() default {};
121
122    /**
123     * @return an array of properties, identified by their names representing at least a
124     *         {@link Collection} that are to be asserted ignoring the concrete order, see
125     *         {@link PropertyConfig#assertionStrategy()} and
126     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
127     *         always respect / assert the same order of elements.
128     */
129    String[] assertUnorderedCollection() default {};
130
131    /**
132     * In case methodPrefix is not set the corresponding build method to be accessed for setting the
133     * value is the name of the attribute: propertyName(), in case it is a concrete value, e.g.
134     * 'with' it will taken into account: withPropertName().
135     *
136     * @return the method prefix, defaults to empty string
137     */
138    String methodPrefix() default "";
139
140}
