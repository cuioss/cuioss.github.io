029/**
030 * Helper class providing base functionality for test for the
031 * {@link Object#equals(Object)} and {@link Object#hashCode()} variants of classes.
032 *
033 * @author Oliver Wolff (i001428)
034 */
035public class EqualsAndHashcodeContractImpl implements ObjectTestContract {
036
037    private static final Integer DEFAULT_INT_VALUE = 0;
038
039    private static final CuiLogger log = new CuiLogger(EqualsAndHashcodeContractImpl.class);
040
041    @Override
042    public void assertContract(final ParameterizedInstantiator<?> instantiator,
043            final ObjectTestConfig objectTestConfig) {
044
045        requireNonNull(instantiator, "parameterizedInstantiator must not be null");
046
047        final var builder = new StringBuilder("Verifying ");
048        builder.append(getClass().getName()).append("\nWith configuration: ")
049                .append(instantiator.toString());
050        log.info(builder.toString());
051
052        final Object target = instantiator.newInstanceMinimal();
053        assertBasicContractOnEquals(target);
054        ReflectionUtil.assertHashCodeMethodIsOverriden(target.getClass());
055        assertBasicContractOnHashCode(target);
056
057        if (shouldTestPropertyContract(objectTestConfig)) {
058            executePropertyTests(instantiator, objectTestConfig);
059        } else {
060            log.info("Only checking basic contract of equals() and hasCode()");
061        }
062
063    }
064
065    private static void executePropertyTests(final ParameterizedInstantiator<?> instantiator,
066            final ObjectTestConfig objectTestConfig) {
067        final SortedSet<String> consideredAttributes = new TreeSet<>();
068        instantiator.getRuntimeProperties().getWritableProperties().stream()
069                .filter(p -> PropertyMemberInfo.DEFAULT.equals(p.getPropertyMemberInfo()))
070                .forEach(p -> consideredAttributes.add(p.getName()));
071
072        if (null != objectTestConfig) {
073            // Whitelist takes precedence
074            if (objectTestConfig.equalsAndHashCodeOf().length > 0) {
075                consideredAttributes.clear();
076                consideredAttributes.addAll(Arrays.asList(objectTestConfig.equalsAndHashCodeOf()));
077            } else {
078                consideredAttributes
079                        .removeAll(Arrays.asList(objectTestConfig.equalsAndHashCodeExclude()));
080            }
081        }
082        if (consideredAttributes.isEmpty()) {
083            log.debug("No configured properties to be tested. Is this intentional?");
084        } else {
085            log.info("Configured attributes found for equalsAndHashCode-testing: "
086                    + consideredAttributes);
087            assertEqualsAndHashCodeWithVariants(instantiator, consideredAttributes);
088        }
089    }
090
091    private static boolean shouldTestPropertyContract(final ObjectTestConfig objectTestConfig) {
092        return null == objectTestConfig || !objectTestConfig.equalsAndHashCodeBasicOnly();
093    }

095    /**
096     * Asserts the {@link Object#equals(Object)} and {@link Object#hashCode()} contract with
097     * variants of data.
098     *
099     * @param instantiator
100     */
101    private static void assertEqualsAndHashCodeWithVariants(
102            final ParameterizedInstantiator<?> instantiator,
103            final SortedSet<String> consideredAttributes) {
104
105        assertEqualsAndHasCodeWithAllPropertiesSet(instantiator, consideredAttributes);
106
107        assertEqualsAndHashCodeWithSkippingProperties(instantiator, consideredAttributes);
108
109        assertEqualsAndHashCodeWithChangingProperties(instantiator, consideredAttributes);
110
111    }
112
113    /**
114     * Assert the methods {@link Object#equals(Object)} and {@link Object#hashCode()} with all
115     * properties set
116     *
117     * @param instantiator
118     * @param consideredAttributes
119     */
120    private static void assertEqualsAndHasCodeWithAllPropertiesSet(
121            final ParameterizedInstantiator<?> instantiator,
122            final SortedSet<String> consideredAttributes) {
123        final Collection<String> actualAttributes = new ArrayList<>(consideredAttributes);
124
125        final Collection<String> requiredNames = RuntimeProperties
126                .extractNames(instantiator.getRuntimeProperties().getRequiredProperties());
127
128        actualAttributes.addAll(requiredNames);
129
130        final var properties =
131            instantiator.getRuntimeProperties().getWritableAsPropertySupport(true, actualAttributes);
132
133        final Object fullObject1 = instantiator.newInstance(properties, false);
134        final Object fullObject2 = instantiator.newInstance(properties, false);
135
136        assertEquals(
137                fullObject1, fullObject2, "Objects should be equal with all properties set");
138
139        assertEquals(fullObject2, fullObject1,
140                "Objects should be equal with all properties set, but are not symmetric");
141
142        assertBasicContractOnHashCode(fullObject1);
143    }

145    /**
146     * Assert the methods {@link Object#equals(Object)} and {@link Object#hashCode()} with the given
147     * set of properties and iterates accordingly the available variants
148     *
149     * @param instantiator
150     * @param consideredAttributes
151     */
152    private static void assertEqualsAndHashCodeWithSkippingProperties(
153            final ParameterizedInstantiator<?> instantiator,
154            final Set<String> consideredAttributes) {
155
156        final var information = instantiator.getRuntimeProperties();
157
158        final List<PropertySupport> allWritableProperties =
159            information.getWritableAsPropertySupport(true).stream()
160                    .filter(prop -> consideredAttributes.contains(prop.getName()))
161                    .collect(Collectors.toList());
162
163        final List<PropertySupport> nonDefaultProperties = allWritableProperties.stream()
164                .filter(prop -> !prop.isDefaultValue())
165                .collect(Collectors.toList());
166
167        final List<PropertySupport> requiredProperties = nonDefaultProperties.stream()
168                .filter(PropertySupport::isRequired)
169                .collect(Collectors.toList());
170
171        final List<PropertySupport> additionalProperties = nonDefaultProperties.stream()
172                .filter(property -> !property.isRequired())
173                .collect(Collectors.toList());
174
175        final var upperBound =
176            Math.min(nonDefaultProperties.size(), consideredAttributes.size()) - 2;
177        if (additionalProperties.isEmpty()) {
178            log.info("Only required or default properties found, therefore no further testing");
179        } else {
180            final Object minimalObject =
181                instantiator.newInstance(requiredProperties, false);
182            final Object fullObject =
183                instantiator.newInstance(allWritableProperties, false);
184            List<PropertySupport> iteratingProperties = new ArrayList<>(requiredProperties);
185            // Common Order of properties
186            for (final PropertySupport support : additionalProperties) {
187                if (iteratingProperties.size() < upperBound) {
188                    iteratingProperties.add(support);
189                } else {
190                    // Special case for the last property to be set but the objects
191                    // still need to be unequal. For this last property to be iterated the value
192                    // will be set to an explicit unequal value:
193                    iteratingProperties.add(support.createCopyWithNonEqualValue());
194                }
195                final Object iterating =
196                    instantiator.newInstance(iteratingProperties, false);
197                final var current = support.getName();
198                assertEqualObjectAreNotEqual(minimalObject, iterating,
199                        current);
200                assertEqualObjectAreNotEqual(fullObject, iterating,
201                        current);
202                assertBasicContractOnHashCode(iterating);
203            }
204            // reverse Order of additional properties
205            iteratingProperties = new ArrayList<>(requiredProperties);
206            final List<PropertySupport> reverseAddtionalProperties =
207                new ArrayList<>(additionalProperties);
208            Collections.reverse(reverseAddtionalProperties);
209            for (final PropertySupport support : reverseAddtionalProperties) {
210                if (iteratingProperties.size() < upperBound) {
211                    iteratingProperties.add(support);
212                } else {
213                    iteratingProperties.add(support.createCopyWithNonEqualValue());
214                }
215                final Object iterating =
216                    instantiator.newInstance(iteratingProperties, false);
217                final var current = support.getName();
218                assertEqualObjectAreNotEqual(minimalObject, iterating,
219                        current);
220                assertEqualObjectAreNotEqual(fullObject, iterating,
221                        current);
222                assertBasicContractOnHashCode(iterating);
223            }
224        }
225
226    }
227
228    private static void assertEqualsAndHashCodeWithChangingProperties(
229            final ParameterizedInstantiator<?> instantiator,
230            final SortedSet<String> consideredAttributes) {
231        final Map<String, PropertySupport> allWritableProperties = new HashMap<>();
232
233        instantiator.getRuntimeProperties()
234                .getWritableAsPropertySupport(true).forEach(p -> allWritableProperties.put(p.getName(), p));
235
236        final Object expected =
237            instantiator.newInstance(new ArrayList<>(allWritableProperties.values()), false);
238        for (final String name : consideredAttributes) {
239            final Map<String, PropertySupport> current = new HashMap<>(allWritableProperties);
240            assertTrue(
241                    current.containsKey(name), "Invalid configuration found: " + name + " not defined as property.");
242            current.put(name, current.get(name).createCopyWithNonEqualValue());
243            final Object actual =
244                instantiator.newInstance(new ArrayList<>(current.values()), false);
245            assertEqualObjectAreNotEqual(expected, actual, name);
246        }
247        // Now reverse order
248        final List<String> reverse = new ArrayList<>(consideredAttributes);
249        Collections.reverse(reverse);
250        for (final String name : reverse) {
251            final Map<String, PropertySupport> current = new HashMap<>(allWritableProperties);
252            assertTrue(current.containsKey(name), "Invalid configuration found: " + name + " not defined as property.");
253            current.put(name, current.get(name).createCopyWithNonEqualValue());
254            final Object actual =
255                instantiator.newInstance(new ArrayList<>(current.values()), false);
256            assertEqualObjectAreNotEqual(expected, actual, name);
257        }
258    }
259
260    private static void assertEqualObjectAreNotEqual(final Object expected, final Object actual,
261            final String deltaPropertyName) {
262        final var message = new StringBuilder("The Objects of type ")
263                .append(expected.getClass().getName())
264                .append(" should not be equal, current property=")
265                .append(deltaPropertyName).toString();
266        assertNotEquals(expected, actual, message);
267        assertNotEquals(actual, expected, message);
268    }

270    /**
271     * Verify object has implemented {@link Object#equals(Object)} method. In addition it checks
272     * whether the basic functionality like
273     * <ul>
274     * <li>equals(null) will be 'false'</li>
275     * <li>equals(new Object()) will be 'false'</li>
276     * <li>underTest.equals(underTest) will be 'true'</li>
277     * </ul>
278     * is implemented correctly
279     *
280     * @param underTest
281     *            object under test
282     */
283    @SuppressWarnings({ "squid:S2159", "squid:S1764" }) // Sonar complains that the x.equals(x) is
284                                                        // always true. This is
285    // only the case if implemented correctly, what is checked
286    // within this test
287    public static void assertBasicContractOnEquals(final Object underTest) {
288
289        ReflectionUtil.assertEqualsMethodIsOverriden(underTest.getClass());
290
291        // basic checks to equals implementation
292        final var msgNotEqualsNull =
293            "Expected result for equals(null) will be 'false'. Class was : " + underTest.getClass();
294
295        assertFalse(underTest.equals(null), msgNotEqualsNull);
296
297        final var msgNotEqualsObject =
298            "Expected result for equals(new Object()) will be 'false'. Class was : "
299                    + underTest.getClass();
300
301        assertFalse(underTest.equals(new Object()), msgNotEqualsObject);
302
303        final var msgEqualsToSelf =
304            "Expected result for equals(underTest) will be 'true'. Class was : "
305                    + underTest.getClass();
306
307        assertTrue(underTest.equals(underTest), msgEqualsToSelf);
308
309    }

311    /**
312     * Verify object has implemented {@link Object#hashCode()} method.
313     *
314     * @param underTest
315     *            object under test
316     */
317    public static void assertBasicContractOnHashCode(final Object underTest) {
318
319        // basic checks to hashCode implementation
320        assertNotEquals(DEFAULT_INT_VALUE, underTest.hashCode(),
321                "Expected result of hashCode method is not '0'. Class was : "
322                        + underTest.getClass());
323    }
