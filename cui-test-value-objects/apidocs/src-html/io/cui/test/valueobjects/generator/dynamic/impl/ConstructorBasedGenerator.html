057    private void logExtendedInformationAboutUsedConstructor(final ArrayList<Object> parameterValues) {
058        final var constructorModifierValue = constructor.getModifiers();
059        if (!Modifier.isPublic(constructorModifierValue)) {
060            log.warn("!!! Attention : " + "A non public constructor will be used to create an instance for {}. " + "This is illegal and can cause unexpected behaviour. " + "Solution: provide a fitting generator instead!", constructor.getName());
061            final List<String> parameterInfo = new ArrayList<>(constructor.getParameters().length);
062            for (final Parameter parameter : constructor.getParameters()) {
063                parameterInfo.add(parameter.getType().getSimpleName() + " " + parameter.getName());
064            }
065            final var modifier = Modifier.toString(constructorModifierValue);
066            final var constructorInfo = modifier + " " + constructor.getName() + "(" + Joiner.on(", ").skipNulls().join(parameterInfo) + ")";
067            log.info("Used constructor : {}", constructorInfo);
068            log.info("Used constructor parameter : {}", logUsedValuesForConstructor(parameterValues));
069        }
070    }

087     * to find a public, than protected, package privates, private constructor. It always uses the
088     * constructor with the fewest arguments
089     *
090     * @param type to be checked for constructors, must not be null, nor an interface, nor an

095    public static final <T> Optional<TypedGenerator<T>> getGeneratorForType(final Class<T> type) {
096        if (!isReponsibleForType(type)) {
097            return Optional.empty();
098        }
099        final List<Constructor<?>> constructors = Arrays.asList(type.getDeclaredConstructors());
100        if (constructors.isEmpty()) {
101            log.warn("Unable to determine constructor for class {} ", type);
102            return Optional.empty();
103        }
104        // Order according to parameter-count
105        constructors.sort(Comparator.comparingInt(Constructor::getParameterCount));
106        // Find fitting public constructor
107        List<Constructor<?>> filteredConstructors = constructors.stream().filter(c -> Modifier.isPublic(c.getModifiers())).collect(Collectors.toList());
108        if (!filteredConstructors.isEmpty()) {
109            return findFittingConstructor(type, filteredConstructors);
110        }
111        // Find fitting protected constructor
112        filteredConstructors = constructors.stream().filter(c -> Modifier.isProtected(c.getModifiers())).collect(Collectors.toList());
113        if (!filteredConstructors.isEmpty()) {
114            return findFittingConstructor(type, filteredConstructors);
115        }
116        // Find fitting package private constructor
117        filteredConstructors = constructors.stream().filter(c -> 0 == c.getModifiers()).collect(Collectors.toList());
118        if (!filteredConstructors.isEmpty()) {
119            return findFittingConstructor(type, filteredConstructors);
120        }
121        // Find fitting private constructor
122        filteredConstructors = constructors.stream().filter(c -> Modifier.isPrivate(c.getModifiers())).collect(Collectors.toList());
123        if (!filteredConstructors.isEmpty()) {
124            return findFittingConstructor(type, filteredConstructors);
125        }
126        log.warn("Unable to determine constructor for class {} ", type);
127        return Optional.empty();
128    }

130    private static boolean isReponsibleForType(final Class<?> type) {
131        if (null == type || type.isAnnotation()) {
132            return false;
133        }
134        return !type.isEnum() && !type.isInterface() && !Modifier.isAbstract(type.getModifiers());
135    }

137    private static <T> Optional<TypedGenerator<T>> findFittingConstructor(final Class<T> type, final List<Constructor<?>> constructorList) {
138        log.debug("Searching constructor for class {}", type);
139        if (1 == constructorList.size()) {
140            log.debug("Only one constructor present, so choosing this one");
141            return createForConstructor(type, constructorList.get(0));
142        }
143        for (final Constructor<?> con : constructorList) {
144            // Ok, try to find a constructor where the parameter are already registered Generator
145            final List<Class<?>> parameter = Arrays.asList(con.getParameterTypes());
146            var allGeneratorAvailable = true;
147            for (final Class<?> param : parameter) {
148                if (!TypedGeneratorRegistry.containsGenerator(param)) {
149                    allGeneratorAvailable = false;
150                    log.debug("Missing generator for {}", param);
151                    break;
152                }
153            }
154            if (allGeneratorAvailable) {
155                return createForConstructor(type, con);
156            }
157        }
158        log.warn("No valid constructor found for class {}", type);
159        for (final Constructor<?> con : constructorList) {
160            if (1 == con.getParameterCount() && con.getParameterTypes()[0].equals(type)) {
161                log.debug("Skipping copy constructor...");
162                continue;
163            }
164            return createForConstructor(type, con);
165        }
166        throw new IllegalStateException("No matching constructor found for class " + type);
167    }

169    private static <T> Optional<TypedGenerator<T>> createForConstructor(final Class<T> type, final Constructor<?> con) {
170        @SuppressWarnings("unchecked")
171        final var constructor = (Constructor<T>) con;
172        SecuritySupport.setAccessible(constructor, true);
173        final List<TypedGenerator<?>> generators = new ArrayList<>();
174        for (final Class<?> parameterType : constructor.getParameterTypes()) {
175            if (parameterType.equals(type)) {
176                // Special case her: eventually copy-constructor -> Play safe prevent infinite
177                // loop, type in this case is not an interface
178                log.warn("Unable to create a generator for copy-constuctor of same type for class {}, constructor parameter type = {}", type, parameterType);
179                return Optional.empty();
180            }
181            generators.add(GeneratorResolver.resolveGenerator(parameterType));
182        }
183        return Optional.of(new ConstructorBasedGenerator<>(type, generators, constructor));
184    }
