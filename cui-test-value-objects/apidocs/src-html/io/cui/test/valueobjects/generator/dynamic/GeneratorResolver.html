032     * It works through all existing find methods. as last resort is uses
033     * {@link InterfaceProxyGenerator} or {@link DynamicProxyGenerator} that will always return a
034     * valid one
035     * <em>Caution:</em> The resolving system relies on {@link TypedGeneratorRegistry} being
036     * configured properly, saying {@link TypedGeneratorRegistry#registerBasicTypes()} has been
037     * called prior to this method

040     * @return a concrete {@link TypedGenerator} for the given type
041     */
042    public static <T> TypedGenerator<T> resolveGenerator(final Class<T> type) {

045        Optional<TypedGenerator<T>> found = TypedGeneratorRegistry.getGenerator(type);
046        if (found.isPresent()) {
047            log.trace(FOUND_GENERATOR_FOR_TYPE, found.get().getClass().getName(), type.getName());

049        }

078        log.debug("resolveProxyGenerator for type {}", type.getName());
079        Optional<TypedGenerator<T>> found = InterfaceProxyGenerator.getGeneratorForType(type);
080        if (found.isPresent()) {
081            TypedGeneratorRegistry.registerGenerator(found.get());
082            log.trace(FOUND_GENERATOR_FOR_TYPE, found.get().getClass().getName(), type.getName());
083            return found.get();
084        }
085        found = DynamicProxyGenerator.getGeneratorForType(type);
086        if (found.isPresent()) {
087            TypedGeneratorRegistry.registerGenerator(found.get());
088            log.trace(FOUND_GENERATOR_FOR_TYPE, found.get().getClass().getName(), type.getName());
089            return found.get();
090        }
091        throw new IllegalArgumentException("Unable to determine generator for type=" + type);
092    }

103    @SuppressWarnings("unchecked")
104    public static <T> Optional<TypedGenerator<T>> resolveCollectionGenerator(final Class<T> type) {
105        if (null == type || !type.isInterface()) {
106            return Optional.empty();
107        }
108        final var optional = CollectionType.findResponsibleCollectionType(type);
109        if (optional.isPresent()) {
110            return Optional.of(new CollectionTypeGenerator<>(type, optional.get()));
111        }
112        if (Map.class.isAssignableFrom(type)) {
113            return Optional.of((TypedGenerator<T>) new EmptyMapGenerator());
114        }
115        return Optional.empty();
