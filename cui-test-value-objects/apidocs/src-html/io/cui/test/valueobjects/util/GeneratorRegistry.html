001package io.cui.test.valueobjects.util;

003import java.util.ArrayList;
004import java.util.List;

006import io.cui.test.generator.TypedGenerator;
007import io.cui.test.valueobjects.api.generator.PropertyGenerator;
008import io.cui.test.valueobjects.api.property.PropertyConfig;

010import io.cui.test.valueobjects.property.PropertyMetadata;

012/**
013 * Interface providing the capability for handling the {@link TypedGeneratorRegistry}
014 * programmatically. Implementors of this interface, usually the base test-classes,
015 * are supposed to take care about the runtime handling of the {@link TypedGeneratorRegistry}
016 * The runtime handling consists of three steps
017 * <ul>
018 * <li>Initialize {@link TypedGeneratorRegistry} with the basic Java-types, by calling
019 * {@link TypedGeneratorRegistry#registerBasicTypes()} usually called before all tests.
020 * </li>
021 * <li>Pick up all additional configured {@link TypedGenerator} and register them to the
022 * {@link TypedGeneratorRegistry}, see Configuration</li>
023 * <li>Tear down / clear {@link TypedGeneratorRegistry} by calling
024 * {@link TypedGeneratorRegistry#clear()} using called in the context after all tests.
025 * </li>
026 * </ul>
027 * <h2>Configuration</h2>
028 * <ul>
029 * <li>{@link PropertyMetadata}: The actual property values are generated using
030 * {@link TypedGenerator}s,
031 * see {@link PropertyConfig#generator()} .
032 * They can be tweaked in multiple ways, see {@link io.cui.test.valueobjects.api.generator} for
033 * details</li>
034 * <li>In case the actual test-class implements {@link TypedGenerator} itself it will implicitly
035 * registered as {@link TypedGenerator} at {@link TypedGeneratorRegistry}</li>
036 * <li>In cases you can not use the {@link PropertyGenerator} annotation, e.g. if you have
037 * instances of {@link TypedGenerator} but not types you can override
038 * {@link #registerAdditionalGenerators()}.</li>
039 * </ul>
040 *
041 * @author Oliver Wolff
042 */
043public interface GeneratorRegistry {
044
045    /**
046     * Callback method for registering additional generator, in cases you can not use the
047     * {@link PropertyGenerator} annotation, e.g. if you have instances of {@link TypedGenerator}
048     * but not types. The default implementation solely
049     * return a mutable empty {@link List}, must therefore not be considered using super
050     *
051     * @return A list of {@link TypedGenerator}s to be registered
052     */
053    @SuppressWarnings("squid:S1452") // owolff: No type information available at this level,
054                                     // therefore the wildcard is needed
055    default List<TypedGenerator<?>> registerAdditionalGenerators() {
056        return new ArrayList<>();
057    }
058
059}
