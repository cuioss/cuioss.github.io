043     * One stop method for the deriving of configured metadata
044     *
045     * @param annotated must not be null
046     * @param targetClass must not be null
047     * @return an immutable {@link List} of {@link PropertyMetadata} containing the result of the
048     *         actual scanning with {@link #scanBeanTypeForProperties(Class,
049     *         PropertyReflectionConfig)} and
050     *         {@link #handlePostProcess(Class, SortedSet)} and
051     *         {@link PropertyHelper#handlePropertyConfigAnnotations(Class)} and
052     *         {@link PropertyHelper#handlePrimitiveAsDefaults(Collection)}
053     */
054    public static <T> List<PropertyMetadata> handlePropertyMetadata(final Class<?> annotated, final Class<T> targetClass) {
055        requireNonNull(annotated);
056        requireNonNull(targetClass);
057        final List<PropertyMetadata> builder = new ArrayList<>();
058        if (shouldScanClass(annotated)) {
059            final SortedSet<PropertyMetadata> scanned = new TreeSet<>(scanBeanTypeForProperties(targetClass, MoreReflection.extractAnnotation(annotated, PropertyReflectionConfig.class).orElse(null)));
060            builder.addAll(handlePostProcess(annotated, scanned));
061        }
062        final var mapView = PropertyHelper.toMapView(builder);
063        PropertyHelper.handlePropertyConfigAnnotations(annotated).forEach(p -> mapView.put(p.getName(), p));
064        final var handled = PropertyHelper.handlePrimitiveAsDefaults(mapView.values());
065        PropertyHelper.logMessageForPropertyMetadata(handled);
066        return new ArrayList<>(handled);
067    }

096    /**
097     * Uses {@link MoreReflection} to scan the concrete bean and describe the properties with
098     * fitting {@link PropertyMetadata}. Each property will contain the derived data for the
099     * attributes:
100     * <ul>
101     * <li>{@link PropertyMetadata#getName()}</li>
102     * <li>{@link PropertyMetadata#getGenerator()} with the generator being dynamically resolved
103     * using {@link GeneratorResolver#resolveGenerator(Class)}</li>
104     * <li>{@link PropertyMetadata#getPropertyReadWrite()}</li>
105     * <li>{@link PropertyMetadata#getCollectionType()}</li>
106     * <li>{@link PropertyMetadata#getPropertyMemberInfo()}</li>
107     * </ul>
108     * Other attributes use there defaults:
109     * <ul>
110     * <li>{@link PropertyMetadata#isRequired()} defaults to false</li>
111     * <li>{@link PropertyMetadata#isDefaultValue()} defaults to false</li>
112     * </ul>
113     *
114     * @param beanType to be checked by reflection
115     * @param config optional instance of {@link PropertyReflectionConfig} used for filtering the
116     *            scanning
117     * @return a {@link SortedSet} containing the result of the inspection
118     */
119    public static SortedSet<PropertyMetadata> scanBeanTypeForProperties(final Class<?> beanType, final PropertyReflectionConfig config) {
120        final Set<String> filter = new HashSet<>(PROPERTY_IGNORE_SET);
121        if (null != config) {
122            filter.addAll(Arrays.asList(config.exclude()));
123        }
124        final var found = new CollectionBuilder<PropertyMetadata>();
125        var builder = new CollectionBuilder<PropertyHolder>();
126        for (Method method : MoreReflection.retrieveAccessMethods(beanType)) {
127            var attributeName = MoreReflection.computePropertyNameFromMethodName(method.getName());
128            if (filter.contains(attributeName)) {
129                log.debug("Filtering attribute \'%s\' for type \'%s\' as configured", attributeName, beanType);
130                continue;
131            }
132            var holder = PropertyHolder.from(beanType, attributeName);
133            if (!holder.isPresent()) {
134                log.info("Unable to extract metadata for type \'%s\' and method \'%s\'", beanType, method.getName());
135            } else {
136                builder.add(holder.get());
137            }
138        }
139        for (PropertyHolder holder : builder) {
140            found.add(createPropertyMetadata(beanType, holder));
141        }
142        return found.toImmutableNavigableSet();
143    }

145    /**
146     * Creates a {@link PropertyMetadata} for a given field.
147     *
148     * @param beanType providing the property, must not be null
149     * @param propertyHolder identifying the property-metadata, must not be null
150     * @return an instance of {@link PropertyMetadata} describing the field.
151     * @throws IllegalArgumentException wrapping underlying reflection exceptions.
152     */
153    public static PropertyMetadata createPropertyMetadata(final Class<?> beanType, PropertyHolder propertyHolder) {
154        requireNonNull(beanType);
155        requireNonNull(propertyHolder);
156        var collectionType = CollectionType.NO_ITERABLE;
157        Class<?> propertyType = propertyHolder.getType();
158        final var field = MoreReflection.accessField(beanType, propertyHolder.getName());
159        if (field.isPresent()) {
160            final var collectionTypeOption = CollectionType.findResponsibleCollectionType(field.get().getType());
161            if (collectionTypeOption.isPresent()) {
162                collectionType = collectionTypeOption.get();
163                if (CollectionType.ARRAY_MARKER.equals(collectionType)) {
164                    propertyType = field.get().getType().getComponentType();
165                } else {
166                    propertyType = extractParameterizedType(field.get(), (ParameterizedType) field.get().getGenericType());
167                }
168            }
169        }
170        if (null == propertyType) {
171            throw new IllegalArgumentException(String.format("Unable to extract property \'%s\' on type \'%s\'", propertyHolder.getName(), beanType.getName()));
172        }
173        var defaultValued = propertyType.isPrimitive();
174        if (defaultValued && CollectionType.ARRAY_MARKER.equals(collectionType)) {
175            defaultValued = false;
176        }
177        return PropertyMetadataImpl.builder().name(propertyHolder.getName()).defaultValue(defaultValued).collectionType(collectionType).propertyMemberInfo(propertyHolder.getMemberInfo()).propertyReadWrite(propertyHolder.getReadWrite()).generator(GeneratorResolver.resolveGenerator(propertyType)).build();
178    }

180    private static Class<?> extractParameterizedType(final Field field, final ParameterizedType parameterizedType) {
181        try {
182            return (Class<?>) parameterizedType.getActualTypeArguments()[0];
183        } catch (final ClassCastException e) {
184            throw new IllegalStateException(String.format("Unable to determine generic-type for %s, ususally this is the case with nested generics. " + "\nYou need to provide a custom @PropertyConfig for this field and exclude it from scanning" + ", by using PropertyReflectionConfig#exclude.\nSee package-javadoc of io.cui.test.valueobjects for samples.", field.toString()), e);
185        }
186    }

203     * Filters the given properties according to the annotated element given. It checks for the
204     * annotation {@link PropertyReflectionConfig} and filters the given {@link SortedSet}
205     *
206     * @param config must not be null
207     * @param metatdata must not be null
208     * @return the filtered {@link SortedSet}

210    public static SortedSet<PropertyMetadata> handlePostProcessConfig(final PropertyReflectionConfig config, final SortedSet<PropertyMetadata> metatdata) {
211        requireNonNull(metatdata);
212        if (metatdata.isEmpty() || null == config) {
213            return metatdata;
214        }
215        var map = PropertyHelper.handleWhiteAndBlacklist(config.of(), config.exclude(), metatdata);
216        map = AnnotationHelper.modifyPropertyMetadata(map, config.defaultValued(), config.readOnly(), config.required(), config.transientProperties(), config.writeOnly(), config.assertUnorderedCollection());
217        return immutableSortedSet(map.values());
218    }

255        requireNonNull(iterable, "iterable must not be null");
256        final var iterator = iterable.iterator();
257        if (iterator.hasNext()) {
