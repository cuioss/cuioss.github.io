183    public static Map<String, PropertyMetadata> handleWhiteAndBlacklist(final String[] of, final String[] exclude, final Collection<PropertyMetadata> givenMetadata) {
184        var map = PropertyHelper.toMapView(givenMetadata);
185        if (of.length != 0) {
186            // Whitelist takes precedence
187            final Map<String, PropertyMetadata> copyMap = new HashMap<>();
188            for (final String name : of) {
189                assertPropertyExists(name, map);
190                copyMap.put(name, map.get(name));
191            }
192            map = copyMap;
193        } else {
194            // Remove all excluded properties
195            for (final String name : exclude) {
196                map.remove(name);
197            }
198        }
199        return map;
200    }
