001package de.cuioss.test.valueobjects.junit5.extension;
002
003import java.util.Collections;
004
005import org.junit.jupiter.api.extension.AfterAllCallback;
006import org.junit.jupiter.api.extension.ExtensionContext;
007import org.junit.jupiter.api.extension.TestInstancePostProcessor;
008import org.junit.platform.commons.logging.Logger;
009import org.junit.platform.commons.logging.LoggerFactory;
010
011import de.cuioss.test.valueobjects.generator.TypedGeneratorRegistry;
012import de.cuioss.test.valueobjects.util.GeneratorAnnotationHelper;
013import de.cuioss.test.valueobjects.util.GeneratorRegistry;
014
015/**
016 * This extension handles the test-generator handling, see {@link GeneratorRegistry} for details
017 *
018 * @author Oliver Wolff
019 *
020 */
021public class GeneratorRegistryController implements TestInstancePostProcessor, AfterAllCallback {
022
023    private static final Logger log = LoggerFactory.getLogger(GeneratorRegistryController.class);
024
025    @Override
026    public void postProcessTestInstance(Object testInstance, ExtensionContext context) {
027        log.debug(() -> "Clearing TypedGeneratorRegistry registry");
028        TypedGeneratorRegistry.clear();
029        if (testInstance instanceof GeneratorRegistry) {
030            log.debug(() -> "Test-class '" + testInstance.getClass()
031                    + "' is of type de.cuioss.test.valueobjects.util.GeneratorRegistry, initializing Generator framework");
032            var registry = (GeneratorRegistry) testInstance;
033            GeneratorAnnotationHelper.handleGeneratorsForTestClass(registry,
034                    registry.registerAdditionalGenerators());
035        } else {
036            log.debug(() -> "Test-class '{" + testInstance.getClass()
037                    + "}' is NOT of type de.cuioss.test.valueobjects.util.GeneratorRegistry, initializing Generator framework without local Generator");
038            GeneratorAnnotationHelper.handleGeneratorsForTestClass(testInstance,
039                    Collections.emptyList());
040        }
041
042    }
043
044    @Override
045    public void afterAll(ExtensionContext context) {
046        log.debug(() -> "Tearing down TypedGeneratorRegistry registry");
047        TypedGeneratorRegistry.clear();
048    }
049
050}
