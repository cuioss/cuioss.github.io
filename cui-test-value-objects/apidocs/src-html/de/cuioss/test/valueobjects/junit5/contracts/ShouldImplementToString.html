001package de.cuioss.test.valueobjects.junit5.contracts;
002
003import static org.junit.jupiter.api.Assertions.assertNotNull;
004
005import org.junit.jupiter.api.Test;
006
007import de.cuioss.test.valueobjects.contract.ReflectionUtil;
008import de.cuioss.test.valueobjects.objects.TestObjectProvider;
009
010/**
011 * Simple check whether the {@link TestObjectProvider#getUnderTest()} implements
012 * {@link Object#equals(Object)} and {@link Object#hashCode()}
013 *
014 * @author Oliver Wolff
015 * @param <T> identifying the type to be tested is usually but not necessarily at least
016 *            Serializable.
017 *
018 */
019public interface ShouldImplementToString<T> extends TestObjectProvider<T> {
020
021    /**
022     * Simple check whether the {@link TestObjectProvider#getUnderTest()} implements
023     * {@link Object#equals(Object)} and {@link Object#hashCode()}
024     */
025    @Test
026    default void shouldImplementToString() {
027        var underTest = getUnderTest();
028        ReflectionUtil.assertToStringMethodIsOverriden(underTest.getClass());
029        assertNotNull(underTest.toString(), "toString must not return 'null'");
030    }
031}
