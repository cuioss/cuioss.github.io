001package de.cuioss.test.valueobjects.property;
002
003import java.io.Serializable;
004
005import de.cuioss.test.generator.TypedGenerator;
006import de.cuioss.test.generator.impl.CollectionGenerator;
007import de.cuioss.test.valueobjects.property.util.AssertionStrategy;
008import de.cuioss.test.valueobjects.property.util.CollectionType;
009import de.cuioss.test.valueobjects.property.util.PropertyAccessStrategy;
010import de.cuioss.tools.property.PropertyMemberInfo;
011import de.cuioss.tools.property.PropertyReadWrite;
012
013/**
014 * Represents runtime information for a certain property. This contains all metadata that may be
015 * needed.
016 *
017 * @author Oliver Wolff
018 */
019public interface PropertyMetadata extends Comparable<PropertyMetadata> {
020
021    /**
022     * Identifies the name of the property
023     *
024     * @return the actual name of the property, must never be null nor empty
025     */
026    String getName();
027
028    /**
029     * @return the type of the property. This can either be the actual type, in case
030     *         {@link #getCollectionType()} is {@link CollectionType#NO_ITERABLE}, the
031     *         component-type in case of {@link CollectionType#ARRAY_MARKER} or the type argument
032     *         for a collection for the other {@link CollectionType}s, see {@link #next()} and
033     *         {@link #resolveActualClass()}
034     */
035    Class<?> getPropertyClass();
036
037    /**
038     * Generates a next random value
039     * Similar to {@link TypedGenerator#next()} but in case there is a {@link #getCollectionType()}
040     * that is not {@link CollectionType#NO_ITERABLE} it wraps the content of the contained
041     * generator into a the corresponding collectionWrapper, see
042     * {@link CollectionType#nextIterable(CollectionGenerator)}
043     *
044     * @return the next generated value
045     */
046    Object next();
047
048    /**
049     * @return The actual class of the property. PropertyMetadata can always be used directly or the
050     *         data can be implicitly wrapped into a collection or an array. This method computes
051     *         the actual class of the property combining both informations
052     */
053    Class<?> resolveActualClass();
054
055    /**
056     * @return a new instance of {@link CollectionGenerator} wrapping the contained generator;
057     */
058    @SuppressWarnings("squid:S1452") // owolff: No type information available at this level,
059                                     // therefore the wildcard is needed
060    CollectionGenerator<?> resolveCollectionGenerator();
061
062    /**
063     * @return the wrapped {@link TypedGenerator} to dynamically create properties.
064     */
065    @SuppressWarnings("squid:S1452") // owolff: No type information available at this level,
066                                     // therefore the wildcard is needed
067    TypedGenerator<?> getGenerator();
068
069    /**
070     * @return boolean indicating whether the attribute defines a default-value. Default is false.
071     */
072    boolean isDefaultValue();
073
074    /**
075     * Indicates that the attribute is required.
076     *
077     * @return boolean indicating whether the attribute is required, defaults to false.
078     */
079    boolean isRequired();
080
081    /**
082     * Defines the way properties are to be written / read
083     *
084     * @return the {@link PropertyAccessStrategy}, must never be null, defaults to
085     *         {@link PropertyAccessStrategy#BEAN_PROPERTY}
086     */
087    PropertyAccessStrategy getPropertyAccessStrategy();
088
089    /**
090     * In case there is a collectionWrapper defined the generated values will implicitly wrapped
091     * in the corresponding collection class defined by that wrapper.
092     *
093     * @return the {@link CollectionType} represented by this {@link PropertyMetadata}. Must never
094     *         be null but may be {@link CollectionType#NO_ITERABLE}
095     */
096    CollectionType getCollectionType();
097
098    /**
099     * Indicates the way the corresponding property is subject to the contracts regarding the
100     * canonical Object methods like {@link Object#equals(Object)}, {@link Object#hashCode()} and
101     * {@link Object#toString()} and the {@link Serializable} contract. The default is
102     * {@link PropertyMemberInfo#DEFAULT}
103     *
104     * @return the {@link PropertyMemberInfo}, must never be null
105     */
106    PropertyMemberInfo getPropertyMemberInfo();
107
108    /**
109     * @return {@link PropertyReadWrite} indicating whether a property can be read or written too,
110     *         defaults to {@link PropertyReadWrite#READ_WRITE}
111     */
112    PropertyReadWrite getPropertyReadWrite();
113
114    /**
115     * Defines the the way how to deal with equality regarding
116     * PropertySupport.assertValueSet(Object)
117     *
118     * @return the {@link AssertionStrategy}, defaults to {@link AssertionStrategy#DEFAULT}
119     */
120    AssertionStrategy getAssertionStrategy();
121
122}
