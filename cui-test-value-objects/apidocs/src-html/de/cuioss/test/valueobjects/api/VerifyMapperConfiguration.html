001package de.cuioss.test.valueobjects.api;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007import java.util.Collection;
008
009import de.cuioss.test.valueobjects.MapperTest;
010import de.cuioss.test.valueobjects.api.property.PropertyConfig;
011import de.cuioss.test.valueobjects.property.PropertyMetadata;
012import de.cuioss.test.valueobjects.property.util.AssertionStrategy;
013import de.cuioss.tools.property.PropertyReadWrite;
014
015/**
016 * To be used in conjunction with {@link MapperTest}. Defines the mapper-specific definitions
017 * that will be tested.
018 *
019 * @author Oliver Wolff
020 */
021@Retention(RetentionPolicy.RUNTIME)
022@Target({ ElementType.TYPE })
023public @interface VerifyMapperConfiguration {
024
025    /**
026     * @return identifies the properties that can be asserted using {@link Object#equals(Object)}.
027     *         The format is 'source.attributeName:target.attributeName'. The target.attributeName
028     *         and source.attributeName must refer to an already defined {@link PropertyMetadata}.
029     *         Sample: {@code @VerifyMapperConfiguration(equals="name:firstName")} results in
030     *         {@code assertEquals(source.getName(),target.getFirstName()}
031     */
032    String[] equals() default {};
033
034    /**
035     * @return identifies the properties that change the target but result in a non null element. In
036     *         case of a {@link Collection} it will additionally checked on emptiness
037     *         The format is 'source.attributeName:target.attributeName'. The target.attributeName
038     *         and source.attributeName must refer to an already existing {@link PropertyMetadata}.
039     *         Sample: {@code @VerifyMapperConfiguration(notNullNorEmpty="name:lastName")} results
040     *         in the attribute {@code target.getLastName()} being not null
041     */
042    String[] notNullNorEmpty() default {};
043
044    /**
045     * @return an array of properties, identified by their names that are not to be considered for
046     *         this test: black-list
047     */
048    String[] exclude() default {};
049
050    /**
051     * @return an array of properties, identified by their names that are to be considered for
052     *         this test: white-list
053     */
054    String[] of() default {};
055
056    /**
057     * @return an array of properties, identified by their names that are to be treated as required
058     *         properties, see {@link PropertyMetadata#isRequired()}
059     */
060    String[] required() default {};
061
062    /**
063     * @return an array of properties, identified by their names that are to be treated as having a
064     *         default values, see {@link PropertyMetadata#isDefaultValue()}
065     */
066    String[] defaultValued() default {};
067
068    /**
069     * @return an array of properties, identified by their names that are to be treated as being
070     *         read-only, see {@link PropertyReadWrite#READ_ONLY}, usually used in conjunction with
071     *         {@link #defaultValued()}
072     */
073    String[] readOnly() default {};
074
075    /**
076     * @return an array of properties, identified by their names that are to be treated as being
077     *         write-only, see {@link PropertyReadWrite#WRITE_ONLY}, usually used in cases where a
078     *         property to be written will result in other properties but itself can not be accessed
079     *         directly
080     */
081    String[] writeOnly() default {};
082
083    /**
084     * @return an array of properties, identified by their names representing at least a
085     *         {@link Collection} that are to be asserted ignoring the concrete order, see
086     *         {@link PropertyConfig#assertionStrategy()} and
087     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
088     *         always respect / assert the same order of elements.
089     */
090    String[] assertUnorderedCollection() default {};
091
092}
