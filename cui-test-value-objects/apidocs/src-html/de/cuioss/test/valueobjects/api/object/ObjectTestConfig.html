001package de.cuioss.test.valueobjects.api.object;
002
003import static java.lang.annotation.ElementType.TYPE;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Retention;
007import java.lang.annotation.Target;
008
009/**
010 * Simple annotation for extended configuration of {@link ObjectTestContract}. Usually not needed.
011 * Used for corner cases.
012 *
013 * @author Oliver Wolff
014 */
015@Retention(RUNTIME)
016@Target(TYPE)
017public @interface ObjectTestConfig {
018
019    /**
020     * @return an array of properties, identified by their names that are not to be considered for
021     *         equalsAndHashcode-tests: Blacklist
022     */
023    String[] equalsAndHashCodeExclude() default {};
024
025    /**
026     * @return an array of properties, identified by their names that are to be considered for
027     *         equalsAndHashcode-tests: Whitelist: takes precedence over
028     *         {@link #equalsAndHashCodeExclude()}
029     */
030    String[] equalsAndHashCodeOf() default {};
031
032    /**
033     * @return boolean indicating whether to only use the default checks ignoring any properties,
034     *         defaults to {@code false}
035     */
036    boolean equalsAndHashCodeBasicOnly() default false;
037
038    /**
039     * @return an array of properties, identified by their names that are not to be considered for
040     *         serializable-tests: Blacklist
041     */
042    String[] serializableExclude() default {};
043
044    /**
045     * @return an array of properties, identified by their names that are to be considered for
046     *         equalsAndHashcode-tests: Whitelist: takes precedence over
047     *         {@link #serializableExclude()}
048     */
049    String[] serializableOf() default {};
050
051    /**
052     * @return boolean indicating whether to only use the default checks ignoring any properties,
053     *         defaults to {@code false}
054     */
055    boolean serializableBasicOnly() default false;
056
057    /**
058     * @return boolean indicating whether the serialized and de-serialized object should be compared
059     *         using {@link Object#equals(Object)} with the original object
060     *         defaults to {@code true}
061     */
062    boolean serializableCompareUsingEquals() default true;
063
064    /**
065     * @return boolean indicating whether during the {@link ObjectTestContracts#TO_STRING} minimal
066     *         objects should be used regarding the properties defaults to {@code false}
067     */
068    boolean toStringUseMinimalInstance() default false;
069}
