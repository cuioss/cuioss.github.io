001package de.cuioss.test.valueobjects.api;
002
003import de.cuioss.test.valueobjects.objects.ParameterizedInstantiator;
004
005/**
006 * A {@link TestContract} provides a method that again runs a number of asserts.
007 *
008 * @author Oliver Wolff
009 * @param <T> identifying the type of objects to be tested
010 */
011public interface TestContract<T> {
012
013    /**
014     * Checks the concrete contract.
015     */
016    void assertContract();
017
018    /**
019     * @return the underlying {@link ParameterizedInstantiator}
020     */
021    ParameterizedInstantiator<T> getInstantiator();
022}
