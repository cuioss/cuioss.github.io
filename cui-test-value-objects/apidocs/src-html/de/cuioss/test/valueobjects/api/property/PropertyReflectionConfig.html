001package de.cuioss.test.valueobjects.api.property;
002
003import static java.lang.annotation.ElementType.METHOD;
004import static java.lang.annotation.ElementType.TYPE;
005import static java.lang.annotation.RetentionPolicy.RUNTIME;
006
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009import java.util.Collection;
010
011import de.cuioss.test.valueobjects.property.PropertyMetadata;
012import de.cuioss.test.valueobjects.property.util.AssertionStrategy;
013import de.cuioss.tools.property.PropertyMemberInfo;
014import de.cuioss.tools.property.PropertyReadWrite;
015
016/**
017 * While the test classes are capable of auto-detecting JavaProperties you need to adjust them from
018 * time to time. With this annotation you can do this for all properties detected
019 *
020 * @author Oliver Wolff
021 */
022@Retention(RUNTIME)
023@Target({ TYPE, METHOD })
024public @interface PropertyReflectionConfig {
025
026    /**
027     * @return boolean indicating whether to scan the class at all. if it is set {@code true} there
028     *         will be no scanning at all.
029     */
030    boolean skip() default false;
031
032    /**
033     * @return an array of properties, identified by their names that are not to be considered for
034     *         the tests: black-list. At this level of configuration this will skip the actual
035     *         reflection-based scanning for that properties as well
036     */
037    String[] exclude() default {};
038
039    /**
040     * @return an array of properties, identified by their names that are to be considered for
041     *         the tests: white-list
042     */
043    String[] of() default {};
044
045    /**
046     * @return an array of properties, identified by their names that are to be treated as required
047     *         properties, see {@link PropertyMetadata#isRequired()}
048     */
049    String[] required() default {};
050
051    /**
052     * @return an array of properties, identified by their names that are to be treated as transient
053     *         properties, see {@link PropertyMemberInfo#TRANSIENT}
054     */
055    String[] transientProperties() default {};
056
057    /**
058     * @return an array of properties, identified by their names that are to be treated as having a
059     *         default values, see {@link PropertyMetadata#isDefaultValue()}
060     */
061    String[] defaultValued() default {};
062
063    /**
064     * @return an array of properties, identified by their names that are to be treated as being
065     *         read-only, see {@link PropertyReadWrite#READ_ONLY}, usually used in conjunction with
066     *         {@link #defaultValued()}
067     */
068    String[] readOnly() default {};
069
070    /**
071     * @return an array of properties, identified by their names that are to be treated as being
072     *         write-only, see {@link PropertyReadWrite#WRITE_ONLY}, usually used in cases where a
073     *         property to be written will result in other properties but itself can not be accessed
074     *         directly
075     */
076    String[] writeOnly() default {};
077
078    /**
079     * @return an array of properties, identified by their names representing at least a
080     *         {@link Collection} that are to be asserted ignoring the concrete order, see
081     *         {@link PropertyConfig#assertionStrategy()} and
082     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
083     *         always respect / assert the same order of elements.
084     */
085    String[] assertUnorderedCollection() default {};
086}
