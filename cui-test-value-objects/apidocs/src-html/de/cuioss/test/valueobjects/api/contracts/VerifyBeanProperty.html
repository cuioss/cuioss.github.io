001package de.cuioss.test.valueobjects.api.contracts;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007import java.util.Collection;
008
009import de.cuioss.test.valueobjects.api.property.PropertyConfig;
010import de.cuioss.test.valueobjects.property.PropertyMetadata;
011import de.cuioss.test.valueobjects.property.util.AssertionStrategy;
012import de.cuioss.tools.property.PropertyMemberInfo;
013import de.cuioss.tools.property.PropertyReadWrite;
014
015/**
016 * If used on ValueObjectTest this test checks / tests all Bean-properties.
017 * <p>
018 * <em>Caution:</em> The implementation for this contract assumes a valid bean regarding Java-Bean
019 * Spec, therefore a no-args public constructor is required.
020 * </p>
021 * <p>
022 * In essence it checks the getters and setters. As default it assumes the individual properties to
023 * not provide a default value. This can be controlled using {@link #defaultValued()}
024 * </p>
025 *
026 * @author Oliver Wolff
027 */
028@Retention(RetentionPolicy.RUNTIME)
029@Target({ ElementType.TYPE })
030public @interface VerifyBeanProperty {
031
032    /**
033     * @return an array of properties, identified by their names that are not to be considered for
034     *         this test: black-list
035     */
036    String[] exclude() default {};
037
038    /**
039     * @return an array of properties, identified by their names that are to be considered for
040     *         this test: white-list
041     */
042    String[] of() default {};
043
044    /**
045     * @return an array of properties, identified by their names that are to be treated as required
046     *         properties, see {@link PropertyMetadata#isRequired()}
047     */
048    String[] required() default {};
049
050    /**
051     * @return an array of properties, identified by their names that are to be treated as transient
052     *         properties, see {@link PropertyMemberInfo#TRANSIENT}
053     */
054    String[] transientProperties() default {};
055
056    /**
057     * @return an array of properties, identified by their names that are to be treated as having a
058     *         default values, see {@link PropertyMetadata#isDefaultValue()}
059     */
060    String[] defaultValued() default {};
061
062    /**
063     * @return an array of properties, identified by their names that are to be treated as being
064     *         read-only, see {@link PropertyReadWrite#READ_ONLY}, usually used in conjunction with
065     *         {@link #defaultValued()}
066     */
067    String[] readOnly() default {};
068
069    /**
070     * @return an array of properties, identified by their names that are to be treated as being
071     *         write-only, see {@link PropertyReadWrite#WRITE_ONLY}, usually used in cases where a
072     *         property to be written will result in other properties but itself can not be accessed
073     *         directly
074     */
075    String[] writeOnly() default {};
076
077    /**
078     * @return an array of properties, identified by their names representing at least a
079     *         {@link Collection} that are to be asserted ignoring the concrete order, see
080     *         {@link PropertyConfig#assertionStrategy()} and
081     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
082     *         always respect / assert the same order of elements.
083     */
084    String[] assertUnorderedCollection() default {};
085}
