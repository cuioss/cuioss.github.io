001package de.cuioss.test.valueobjects.api.object;
002
003import de.cuioss.test.valueobjects.objects.ParameterizedInstantiator;
004
005/**
006 * An {@link ObjectTestContract} provides a method that again runs a number of asserts regarding the
007 * canonical {@link Object} method.
008 *
009 * @author Oliver Wolff
010 */
011@FunctionalInterface
012public interface ObjectTestContract {
013
014    /**
015     * Checks the concrete contract.
016     *
017     * @param instantiator must not be null. The actual type of the {@link Object} is not of
018     *            interest, because we check {@link Object} contracts
019     * @param objectTestConfig optional configuration configuring the tests
020     */
021    void assertContract(ParameterizedInstantiator<?> instantiator,
022            ObjectTestConfig objectTestConfig);
023}
