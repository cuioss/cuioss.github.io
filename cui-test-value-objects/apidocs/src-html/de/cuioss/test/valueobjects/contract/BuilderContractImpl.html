001package de.cuioss.test.valueobjects.contract;
002
003import static de.cuioss.tools.collect.CollectionLiterals.mutableList;
004import static java.util.Objects.requireNonNull;
005
006import java.util.List;
007import java.util.Optional;
008import java.util.stream.Collectors;
009
010import de.cuioss.test.valueobjects.api.TestContract;
011import de.cuioss.test.valueobjects.api.contracts.VerifyBuilder;
012import de.cuioss.test.valueobjects.objects.BuilderInstantiator;
013import de.cuioss.test.valueobjects.objects.ParameterizedInstantiator;
014import de.cuioss.test.valueobjects.objects.RuntimeProperties;
015import de.cuioss.test.valueobjects.objects.impl.BuilderConstructorBasedInstantiator;
016import de.cuioss.test.valueobjects.objects.impl.BuilderFactoryBasedInstantiator;
017import de.cuioss.test.valueobjects.objects.impl.BuilderParameterizedInstantiator;
018import de.cuioss.test.valueobjects.property.PropertyMetadata;
019import de.cuioss.test.valueobjects.property.PropertySupport;
020import de.cuioss.test.valueobjects.util.AnnotationHelper;
021import de.cuioss.tools.logging.CuiLogger;
022import de.cuioss.tools.reflect.MoreReflection;
023
024/**
025 * Defines basic tests for builder. In essence it will try to create a builder with a minimal set
026 * (required only) and one with all properties set. It sets the properties, build the actual object
027 * and verifies whether the properties are set correctly
028 *
029 * @author Oliver Wolff
030 * @param <T> identifying the type of objects to be tested
031 */
032public class BuilderContractImpl<T> implements TestContract<T> {
033
034    private static final CuiLogger log = new CuiLogger(BuilderContractImpl.class);
035
036    private final BuilderInstantiator<T> builderInstantiator;
037
038    private final RuntimeProperties runtimeProperties;
039
040    /** The usually chosen name for the actual build method. */
041    public static final String DEFAULT_BUILD_METHOD_NAME = "build";
042
043    /** The usually chosen name for a factory builder method. */
044    public static final String DEFAULT_BUILDER_FACTORY_METHOD_NAME = "builder";
045
046    /**
047     * @param instantiator must not be null
048     * @param runtimeProperties must not be null
049     */
050    public BuilderContractImpl(final BuilderInstantiator<T> instantiator,
051            final RuntimeProperties runtimeProperties) {
052        super();
053        this.builderInstantiator =
054            requireNonNull(instantiator, "builderInstantiator must not be null");
055        this.runtimeProperties =
056            requireNonNull(runtimeProperties,
057                    "runtimeProperties must not be null.");
058    }
059
060    @Override
061    public void assertContract() {
062
063        final var builder = new StringBuilder("Verifying ");
064        builder.append(getClass().getName()).append("\nWith configuration: ")
065                .append(this.builderInstantiator.toString());
066        log.info(builder.toString());
067        setAndVerifyProperties(this.runtimeProperties.getRequiredProperties());
068        setAndVerifyProperties(this.runtimeProperties.getAllProperties());
069        shouldFailOnMissingRequiredAttributes();
070    }
071
072    private void setAndVerifyProperties(final List<PropertyMetadata> propertiesToBeChecked) {
073        final List<PropertySupport> properties =
074            propertiesToBeChecked.stream()
075                    .map(PropertySupport::new)
076                    .collect(Collectors.toList());
077
078        properties.forEach(PropertySupport::generateTestValue);
079
080        final var builder = this.builderInstantiator.newBuilderInstance();
081        for (final PropertySupport support : properties) {
082            support.apply(builder);
083        }
084        final var built = this.builderInstantiator.build(builder);
085
086        for (final PropertySupport support : properties) {
087            if (support.isReadable()) {
088                support.assertValueSet(built);
089            }
090        }
091    }
092
093    private void shouldFailOnMissingRequiredAttributes() {
094        if (this.runtimeProperties.getRequiredProperties().isEmpty()) {
095            return;
096        }
097        for (final PropertyMetadata property : this.runtimeProperties
098                .getRequiredProperties()) {
099            final List<PropertyMetadata> requiredMinusOne =
100                mutableList(this.runtimeProperties.getRequiredProperties());
101            requiredMinusOne.remove(property);
102            var failed = false;
103            try {
104                setAndVerifyProperties(requiredMinusOne);
105                failed = true;
106            } catch (final AssertionError e) {
107                // Expected: Should have been thrown
108            }
109            if (failed) {
110                throw new AssertionError(
111                        "Property is marked as required but the builder accepts if it is missing: "
112                                + property.toString());
113            }
114        }
115    }
116
117    @Override
118    public ParameterizedInstantiator<T> getInstantiator() {
119        return new BuilderParameterizedInstantiator<>(this.builderInstantiator,
120                this.runtimeProperties);
121    }
122
123    /**
124     * Factory method for creating an instance of {@link BuilderContractImpl} depending on the
125     * given parameter
126     *
127     * @param beanType identifying the type to be tested. Must not be null
128     * @param annotated the annotated unit-test-class. It is expected to be annotated with
129     *            {@link VerifyBuilder}, otherwise the method will return
130     *            {@link Optional#empty()}
131     * @param initialPropertyMetadata identifying the complete set of {@link PropertyMetadata},
132     *            where the actual {@link PropertyMetadata} for the bean tests will be filtered by
133     *            using the attributes defined within {@link VerifyBuilder}. Must not be
134     *            null.
135     * @return an instance Of {@link BeanPropertyContractImpl} in case all requirements for the
136     *         parameters are correct, otherwise it will return {@link Optional#empty()}
137     */
138    public static final <T> Optional<BuilderContractImpl<T>> createBuilderTestContract(
139            final Class<T> beanType, final Class<?> annotated,
140            final List<PropertyMetadata> initialPropertyMetadata) {
141
142        requireNonNull(beanType, "beantype must not be null");
143        requireNonNull(annotated, "annotated must not be null");
144        requireNonNull(initialPropertyMetadata, "initialPropertyMetadata must not be null");
145
146        final Optional<VerifyBuilder> config =
147            MoreReflection.extractAnnotation(annotated, VerifyBuilder.class);
148
149        if (!config.isPresent()) {
150            log.debug("No annotation of type BuilderTestContract available on class: "
151                    + annotated);
152            return Optional.empty();
153        }
154        final var metadata =
155            AnnotationHelper.handleMetadataForBuilderTest(annotated, initialPropertyMetadata);
156
157        final var contract = config.get();
158        BuilderInstantiator<T> instantiator;
159        if (VerifyBuilder.class.equals(contract.builderClass())) {
160            if (VerifyBuilder.class.equals(contract.builderFactoryProvidingClass())) {
161                instantiator = new BuilderFactoryBasedInstantiator<>(beanType,
162                        contract.builderFactoryMethodName(),
163                        contract.builderMethodName());
164            } else {
165                instantiator =
166                    new BuilderFactoryBasedInstantiator<>(contract.builderFactoryProvidingClass(),
167                            contract.builderFactoryMethodName(),
168                            contract.builderMethodName());
169            }
170        } else {
171            instantiator = new BuilderConstructorBasedInstantiator<>(contract.builderClass(),
172                    contract.builderMethodName());
173        }
174
175        return Optional
176                .of(new BuilderContractImpl<>(instantiator,
177                        new RuntimeProperties(metadata)));
178    }
179}
