062/**
063 * Factory for creating instances {@link PropertyMetadata} for standard java-types and primitives.
064 * <p>
065 * Usually it is used with the {@link #metadata(String)} method that will create concrete
066 * instances. In case you want to to further configure the result you can access the corresponding
067 * builder with {@link #metadataBuilder(String)}
068 * </p>
069 *
070 * @author Oliver Wolff
071 * @param <T> identifying the type to be generated
072 */
073public final class JavaTypesGenerator<T> {

315    /** The concrete type of the object created by the generator. */
316    private final Class<T> propertyType;
317
318    /** The actual generator. */
319    private final TypedGenerator<T> generator;
320
321    /**
322     * used for primitive types that have always a default, for all
323     * other types this is <code>null</code>
324     */
325    private final Object defaultValue;
326
327    private JavaTypesGenerator(final TypedGenerator<T> typedGenerator,
328            final Object defaultValue) {
329        this.propertyType = typedGenerator.getType();
330        this.generator = typedGenerator;
331        this.defaultValue = defaultValue;
332
333        GENERATORS.add(typedGenerator);
334    }
335
336    /**
337     * @return all {@link TypedGenerator} provided by this class
338     */
339    @SuppressWarnings("squid:S1452") // owolff all generators available, therefore the wildcard is
340                                     // needed
341    public static List<TypedGenerator<?>> allGenerators() {
342        return immutableList(GENERATORS);
343    }
344
345    /**
346     * Creates a configured instance of {@link PropertyMetadata}
347     *
348     * @param name of the property the {@link PropertyMetadata} is related to.
349     * @return {@link PropertyMetadata} with the given name as
350     *         {@link PropertyMetadata#getName()}, the corresponding
351     *         {@link TypedGenerator} and {@link PropertyMetadata#getPropertyClass()}
352     */
353    public PropertyMetadata metadata(final String name) {
354        return metadataBuilder(name).build();
355    }
356
357    /**
358     * Creates a configured instance of {@link PropertyMetadata} with the required attribute set to
359     * <code>true</code> -> Property is required
360     *
361     * @param name of the property the {@link PropertyMetadata} is related to.
362     * @return {@link PropertyMetadata} with the given name as
363     *         {@link PropertyMetadata#getName()}, the corresponding
364     *         {@link TypedGenerator} and {@link PropertyMetadata#getPropertyClass()}
365     */
366    public PropertyMetadata metadataRequired(final String name) {
367        return metadataBuilder(name).required(true).build();
368    }
369
370    /**
371     * Creates a configured instance of {@link PropertyMetadata} with the required attribute set to
372     * <code>true</code> -> Property is required and the variant that the content
373     * is to be be used as collectionGenerator.
374     *
375     * @param name of the property the {@link PropertyMetadata} is related to.
376     * @param collectionType to be set to the
377     *            {@link PropertyMetadataBuilder#collectionType(CollectionType)}
378     * @return {@link PropertyMetadata} with the given name as
379     *         {@link PropertyMetadata#getName()}, the corresponding
380     *         {@link TypedGenerator} and {@link PropertyMetadata#getPropertyClass()}
381     */
382    public PropertyMetadata metadataRequired(final String name,
383            final CollectionType collectionType) {
384        return metadataBuilder(name).required(true).collectionType(collectionType).build();
385    }
386
387    /**
388     * Creates a configured instance of {@link PropertyMetadata} with the variant that the content
389     * is to be be used as collectionGenerator.
390     *
391     * @param name of the property the {@link PropertyMetadata} is related to.
392     * @param collectionType to be set to the
393     *            {@link PropertyMetadataBuilder#collectionType(CollectionType)}
394     * @return {@link PropertyMetadata} with the given name as
395     *         {@link PropertyMetadata#getName()}, the corresponding
396     *         {@link TypedGenerator} and {@link PropertyMetadata#getPropertyClass()}
397     */
398    public PropertyMetadata metadata(final String name, final CollectionType collectionType) {
399        return metadataBuilder(name).collectionType(collectionType).build();
400    }
401
402    /**
403     * Creates pre-configured instances of {@link PropertyMetadataBuilder}
404     *
405     * @param name of the property the {@link PropertyMetadata} is related to.
406     * @return {@link PropertyMetadataBuilder} with the given name as
407     *         {@link PropertyMetadata#getName()}, the corresponding
408     *         {@link TypedGenerator} and {@link PropertyMetadata#getPropertyClass()}
409     */
410    public PropertyMetadataBuilder metadataBuilder(final String name) {
411        checkArgument(!nullToEmpty(name).isEmpty(), "name must not be null nor empty");
412        return PropertyMetadataImpl.builder().name(name)
413                .generator(this.generator).defaultValue(this.defaultValue != null)
414                .propertyClass(this.propertyType);
415    }
