001package de.cuioss.test.valueobjects.objects;
002
003import java.util.List;
004
005import de.cuioss.test.valueobjects.property.PropertyMetadata;
006import de.cuioss.test.valueobjects.property.PropertySupport;
007
008/**
009 * Abstraction of certain ways for creating and populating test-objects .
010 *
011 * @author Oliver Wolff
012 * @param <T> identifying the type of object to be instantiated
013 */
014public interface ParameterizedInstantiator<T> {
015
016    /**
017     * Creates a new Object according to the given parameter, represented as {@link List} of
018     * {@link PropertySupport}
019     *
020     * @param properties to be applied to the newly created object.
021     * @param generatePropertyValues indicating whether this method should implicitly call
022     *            {@link PropertySupport#generateTestValue()} or not. In case being
023     *            <code>false</code> the caller must ensure this.
024     * @return a newly created object with the given properties being applied.
025     * @throws AssertionError in case the object could not be created or the properties could not
026     *             been applied.
027     */
028    T newInstance(List<PropertySupport> properties, boolean generatePropertyValues);
029
030    /**
031     * Similar to {@link #newInstance(List, boolean)} but with the difference that the caller has no
032     * control regarding the set / applied properties, that may be need for some asserts. It is some
033     * kind of any()
034     *
035     * @param properties to be applied to the newly created object.
036     * @return a newly created object with the given properties being applied.
037     * @throws AssertionError in case the object could not be created or the properties could not
038     *             been applied.
039     */
040    T newInstance(List<PropertyMetadata> properties);
041
042    /**
043     * @return the runtime information associated with this {@link ParameterizedInstantiator}
044     */
045    RuntimeProperties getRuntimeProperties();
046
047    /**
048     * @return a new instance of the object with only required attributes set.
049     */
050    T newInstanceMinimal();
051
052    /**
053     * @return a new instance of the object with all attributes set.
054     */
055    T newInstanceFull();
056}
