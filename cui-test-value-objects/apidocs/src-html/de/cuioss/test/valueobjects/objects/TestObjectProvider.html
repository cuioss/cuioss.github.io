001package de.cuioss.test.valueobjects.objects;
002
003import java.io.Serializable;
004
005/**
006 * Helper interface for integrating the test-framework with arbitrary tests. In essence it
007 * provides the method {@link #getUnderTest()} that will return the bean / value-object to be
008 * tested. Usually instantiated by Inject or other ways, but out of control of the actual
009 * test-base-class
010 *
011 * @param <T> identifying the type to be tested is usually but not necessarily at least
012 *            {@link Serializable}.
013 * @author Oliver Wolff
014 */
015public interface TestObjectProvider<T> {
016
017    /**
018     * @return the bean to be tested. Usually instantiated by Inject or other concrete
019     */
020    T getUnderTest();
021}
