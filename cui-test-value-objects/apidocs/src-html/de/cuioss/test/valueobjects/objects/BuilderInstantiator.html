001package de.cuioss.test.valueobjects.objects;
002
003/**
004 * Simple interface used for dynamically creating builder objects.
005 *
006 * @author Oliver Wolff
007 * @param <T> identifying the type of object to be instantiated by the builder
008 */
009public interface BuilderInstantiator<T> {
010
011    /**
012     * @return a newly created builder.
013     * @throws AssertionError in case the object could not be created
014     */
015    Object newBuilderInstance();
016
017    /**
018     * @return the class of the object that will be created by the contained builder.
019     */
020    Class<T> getTargetClass();
021
022    /**
023     * @return the class of the builder.
024     */
025    Class<?> getBuilderClass();
026
027    /**
028     * Actually builds the target object
029     *
030     * @param builder
031     * @return the Object created by the contained builder
032     */
033    T build(Object builder);
034
035}
