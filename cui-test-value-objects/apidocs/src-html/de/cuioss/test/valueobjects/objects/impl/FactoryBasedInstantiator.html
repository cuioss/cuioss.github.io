001package de.cuioss.test.valueobjects.objects.impl;
002
003import static de.cuioss.test.valueobjects.objects.impl.ExceptionHelper.extractCauseMessageFromThrowable;
004import static de.cuioss.tools.base.Preconditions.checkArgument;
005import static de.cuioss.tools.string.MoreStrings.isEmpty;
006import static java.util.Objects.requireNonNull;
007import static org.junit.jupiter.api.Assertions.assertNotNull;
008import static org.junit.jupiter.api.Assertions.assertTrue;
009
010import java.lang.reflect.InvocationTargetException;
011import java.lang.reflect.Method;
012import java.util.ArrayList;
013import java.util.List;
014
015import de.cuioss.test.valueobjects.objects.ParameterizedInstantiator;
016import de.cuioss.test.valueobjects.objects.RuntimeProperties;
017import de.cuioss.tools.logging.CuiLogger;
018
019/**
020 * This {@link ParameterizedInstantiator} uses a factory method derived by the given
021 * {@link RuntimeProperties#getAllProperties()} in order to instantiate {@link Object}s
022 *
023 * @author Oliver Wolff
024 * @param <T> identifying the type of objects to be created
025 */
026public class FactoryBasedInstantiator<T> extends AbstractOrderedArgsInstantiator<T> {
027
028    private static final CuiLogger log = new CuiLogger(FactoryBasedInstantiator.class);
029
030    private final Method factoryMethod;
031
032    /**
033     * Constructor.
034     *
035     * @param type the type of the object to be created
036     * @param runtimeProperties must not be null. defines the attributes in the exact order
037     *            to be used for the constructor:
038     *            {@link RuntimeProperties#getAllProperties()}
039     * @param enclosingType the type providing the factory method, usually the target type to be
040     *            created, must not be null
041     * @param factoryMethodName the name of the factory method, must not be null, nor empty.
042     */
043    public FactoryBasedInstantiator(
044            final Class<T> type,
045            final RuntimeProperties runtimeProperties,
046            final Class<?> enclosingType,
047            final String factoryMethodName) {
048
049        super(runtimeProperties);
050        requireNonNull(type, "type must not be null");
051        requireNonNull(enclosingType, "enclosingType must not be null");
052        requireNonNull(factoryMethodName);
053        checkArgument(!isEmpty(factoryMethodName),
054                "factoryMethodName msut not be null nor empty");
055
056        final List<Class<?>> parameter = new ArrayList<>();
057        runtimeProperties.getAllProperties()
058                .forEach(meta -> parameter.add(meta.resolveActualClass()));
059        try {
060            if (parameter.isEmpty()) {
061                this.factoryMethod = enclosingType.getDeclaredMethod(factoryMethodName);
062            } else {
063                this.factoryMethod =
064                    enclosingType.getDeclaredMethod(factoryMethodName, toClassArray(parameter));
065            }
066            assertNotNull(
067
068                    this.factoryMethod, "Unable to find a factory method with signature " + parameter + " and name "
069                            + factoryMethodName);
070            assertTrue(type.isAssignableFrom(this.factoryMethod.getReturnType()),
071                    "Invalid type found on factory method: " + this.factoryMethod.getReturnType());
072        } catch (NoSuchMethodException | SecurityException e) {
073            final var message = new StringBuilder("Unable to find a constructor with signature ")
074                    .append(parameter).toString();
075            log.error(message, e);
076            throw new AssertionError(message);
077        }
078    }
079
080    @SuppressWarnings("unchecked")
081    @Override
082    protected T doInstantiate(final Object... args) {
083        try {
084            return (T) this.factoryMethod.invoke(null, args);
085        } catch (IllegalAccessException | IllegalArgumentException
086                | InvocationTargetException e) {
087            final var message = new StringBuilder("Unable to invoke constructor ")
088                    .append(", due to ").append(extractCauseMessageFromThrowable(e)).toString();
089            throw new AssertionError(message, e);
090        }
091    }
092
093    @Override
094    public String toString() {
095        final var builder = new StringBuilder(getClass().getName());
096        builder.append("\nFactory Method: ").append(this.factoryMethod);
097        builder.append("\nProperty Configuration: ")
098                .append(getRuntimeProperties().toString());
099        return builder.toString();
100    }
101}
