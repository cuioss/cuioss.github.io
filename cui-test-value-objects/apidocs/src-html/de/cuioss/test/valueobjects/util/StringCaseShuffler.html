014     * Shuffles the case of a given string. Shuffling is done for every Character that is a

017     * @param toShuffle if {@code null} or empty the given String will be returned

020    public static String shuffleCase(String toShuffle) {
021        if (isEmpty(toShuffle)) {
022            return toShuffle;
023        }
024        var result = new StringBuilder();
025        for (char c : toShuffle.toCharArray()) {
026            result.append(handleSingleCharacter(c));
027        }
028        return result.toString();
029    }
030
031    private static char handleSingleCharacter(char c) {
032        if (!Character.isAlphabetic(c)) {
033            return c;
034        }
035        if (Generators.booleans().next().booleanValue()) {
036            return Character.toUpperCase(c);
037        }
038        return Character.toLowerCase(c);
039    }
