053    public static List<PropertyMetadata> constructorConfigToPropertyMetadata(final VerifyConstructor config, final Collection<PropertyMetadata> givenMetadata) {
054        requireNonNull(config);
055        requireNonNull(givenMetadata);
056        final Map<String, PropertyMetadata> map = new HashMap<>();
057        final List<String> ofAsList = Arrays.asList(config.of());
058        givenMetadata.stream().filter(p -> ofAsList.contains(p.getName())).forEach(meta -> map.put(meta.getName(), meta));
059        for (final String name : config.of()) {
060            PropertyHelper.assertPropertyExists(name, map);
061            modifyPropertyMetadata(map, config.defaultValued(), config.readOnly(), config.required(), config.transientProperties(), config.writeOnly(), config.assertUnorderedCollection());
062        }
063        if (config.allRequired()) {
064            map.replaceAll((k, v) -> PropertyMetadataImpl.builder(v).required(true).build());
065        }
066        return orderPropertyMetadata(config.of(), handleWritableAttributes(map));
067    }

069    private static List<PropertyMetadata> handleWritableAttributes(final Map<String, PropertyMetadata> map) {
070        final List<PropertyMetadata> result = new ArrayList<>();
071        for (Entry<String, PropertyMetadata> entry : map.entrySet()) {
072            result.add(PropertyMetadataImpl.builder(entry.getValue()).propertyReadWrite(determinePropertyReadWrite(entry.getValue())).build());
073        }
074        return result;

077    private static PropertyReadWrite determinePropertyReadWrite(final PropertyMetadata propertyMetadata) {
078        if (PropertyReadWrite.WRITE_ONLY.equals(propertyMetadata.getPropertyReadWrite())) {
079            return PropertyReadWrite.WRITE_ONLY;
080        }
081        return PropertyReadWrite.READ_WRITE;

157        final var builder = new CollectionBuilder<PropertyMetadata>();
158        if (0 == of.length) {
159            builder.add(givenMetadata);

161            final Map<String, PropertyMetadata> map = new HashMap<>();
162            for (final PropertyMetadata metadata : givenMetadata) {

179        requireNonNull(annotated);
180        requireNonNull(givenMetadata);
181        if (givenMetadata.isEmpty()) {
182            log.warn(NO_PROPERTIES_GIVEN_IS_THIS_INTENTIONAL);
183            return givenMetadata;
184        }
185        final Optional<VerifyBuilder> contractOption = MoreReflection.extractAnnotation(annotated, VerifyBuilder.class);
186        final var contract = contractOption.orElseThrow(() -> new IllegalArgumentException("Given type does not provide the expected annotation BuilderTestContract, type=" + annotated));
187        var map = PropertyHelper.handleWhiteAndBlacklist(contract.of(), contract.exclude(), givenMetadata);
188        modifyPropertyMetadata(map, contract.defaultValued(), contract.readOnly(), contract.required(), contract.transientProperties(), contract.writeOnly(), contract.assertUnorderedCollection());
189        final Map<String, PropertyMetadata> builderPropertyMap = new HashMap<>();
190        for (final PropertyMetadata metadata : BuilderPropertyHelper.handleBuilderPropertyConfigAnnotations(annotated, mutableList(map.values()))) {
191            checkArgument(map.containsKey(metadata.getName()), "Invalid Configuration found: BuilderPropertyConfig and BuilderTestContract do not agree on configuration. offending property: " + metadata);
192            builderPropertyMap.put(metadata.getName(), metadata);
193        }
194        for (final Entry<String, PropertyMetadata> entry : map.entrySet()) {
195            if (!builderPropertyMap.containsKey(entry.getKey())) {
196                var delegate = entry.getValue();
197                if (PropertyAccessStrategy.BEAN_PROPERTY.equals(delegate.getPropertyAccessStrategy())) {
198                    var propertyReadWrite = PropertyReadWrite.READ_WRITE;
199                    if (PropertyReadWrite.WRITE_ONLY.equals(delegate.getPropertyReadWrite())) {
200                        propertyReadWrite = PropertyReadWrite.WRITE_ONLY;
201                    }
202                    delegate = PropertyMetadataImpl.builder(delegate).propertyAccessStrategy(PropertyAccessStrategy.BUILDER_DIRECT).propertyReadWrite(propertyReadWrite).build();
203                }
204                builderPropertyMap.put(entry.getKey(), BuilderMetadata.builder().delegateMetadata(delegate).builderMethodPrefix(contract.methodPrefix()).build());
205            }
206        }
207        return orderPropertyMetadata(contract.of(), builderPropertyMap.values());
208    }
