031    /**
032     * Convenience method for handling all Generator specific aspects like initially calling
033     * {@link TypedGeneratorRegistry#registerBasicTypes()} and calling of the individual registering
034     * methods like {@link #handleUnitClassImplementation(Object)},
035     * {@link #handlePropertyGenerator(Class)} and {@link #handleGeneratorHints(Class)} in case
036     * there are additionalGenerator given they will be registered as well
037     *
038     * @param testClass must not null
039     * @param additionalGenerator
040     */
041    public static void handleGeneratorsForTestClass(final Object testClass, final List<TypedGenerator<?>> additionalGenerator) {
042        requireNonNull(testClass);
043        // Handle Generator
044        TypedGeneratorRegistry.registerBasicTypes();
045        handleUnitClassImplementation(testClass);
046        handlePropertyGenerator(testClass.getClass());
047        handleGeneratorHints(testClass.getClass());
048        if (null != additionalGenerator) {
049            for (final TypedGenerator<?> additional : additionalGenerator) {
050                TypedGeneratorRegistry.registerGenerator(additional);
051            }
052        }
053    }

055    /**
056     * Checks the given type for the annotation {@link PropertyGeneratorHint} and
057     * {@link PropertyGeneratorHints} and registers all found to the {@link TypedGeneratorRegistry}
058     *
059     * @param annotated the class that may or may not provide the annotations, must not be null
060     */
061    public static void handleGeneratorHints(final Class<?> annotated) {
062        for (final PropertyGeneratorHint hint : extractConfiguredGeneratorHints(annotated)) {
063            final TypedGenerator<?> resolved = GeneratorResolver.resolveGenerator(hint.implementationType());
064            TypedGeneratorRegistry.registerTypedGenerator(hint.declaredType(), new WildcardDecoratorGenerator(hint.declaredType(), resolved));
065        }
066    }

069     * Checks the given type for the annotation {@link PropertyGenerator} and

072     * @param annotated the class that may or may not provide the annotations, must not be null

074    public static void handlePropertyGenerator(final Class<?> annotated) {
075        for (final PropertyGenerator config : extractConfiguredPropertyGenerator(annotated)) {
076            for (final Class<?> typedClass : config.value()) {
077                TypedGeneratorRegistry.registerGenerator((TypedGenerator<?>) new DefaultInstantiator<>(typedClass).newInstance());
078            }
