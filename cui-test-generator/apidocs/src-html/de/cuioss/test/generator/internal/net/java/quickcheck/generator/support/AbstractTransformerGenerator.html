017package de.cuioss.test.generator.internal.net.java.quickcheck.generator.support;
018
019import java.util.Objects;
020
021import de.cuioss.test.generator.internal.net.java.quickcheck.Generator;
022
023/**
024 * {@link Generator} implementation that transforms input generator test cases
025 * of type I to output test cases of type T.
026 *
027 * @param <I>
028 *            input generator test case type
029 * @param <T>
030 *            output generator test case type
031 *
032 */
033public abstract class AbstractTransformerGenerator<I, T> implements
034        Generator<T> {
035
036    private Generator<I> inputGenerator;
037
038    protected AbstractTransformerGenerator(Generator<I> inputGenerator) {
039        setInputGenerator(inputGenerator);
040    }
041
042    protected abstract T transform(Generator<I> inputGenerator);
043
044    @Override
045    public T next() {
046        return transform(inputGenerator);
047    }
048
049    Generator<I> getInputGenerator() {
050        return inputGenerator;
051    }
052
053    void setInputGenerator(Generator<I> inputGenerator) {
054        Objects.requireNonNull(inputGenerator, "inputGenerator");
055        this.inputGenerator = inputGenerator;
056    }
057}
