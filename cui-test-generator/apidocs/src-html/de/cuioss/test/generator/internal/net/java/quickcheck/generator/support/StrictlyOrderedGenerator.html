017package de.cuioss.test.generator.internal.net.java.quickcheck.generator.support;
018
019import java.util.ArrayList;
020import java.util.Comparator;
021import java.util.List;
022import java.util.Objects;
023import java.util.Set;
024
025import de.cuioss.test.generator.internal.net.java.quickcheck.Generator;
026
027public class StrictlyOrderedGenerator<T> implements Generator<List<T>> {
028
029    private final Generator<Set<T>> values;
030    private final Comparator<T> comparator;
031
032    public StrictlyOrderedGenerator(Generator<T> input, Comparator<T> comparator, Generator<Integer> sizes) {
033        Objects.requireNonNull(input, "input");
034        Objects.requireNonNull(comparator, "comparator");
035        Objects.requireNonNull(sizes, "sizes");
036        values = new SetGenerator<>(input, sizes, SetGenerator.MAX_TRIES);
037        this.comparator = comparator;
038    }
039
040    @Override
041    public List<T> next() {
042        ArrayList<T> v = new ArrayList<>(values.next());
043        v.sort(comparator);
044        return v;
045    }
046}
