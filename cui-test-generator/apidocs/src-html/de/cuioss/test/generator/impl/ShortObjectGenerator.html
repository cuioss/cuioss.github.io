001package de.cuioss.test.generator.impl;
002
003import de.cuioss.test.generator.TypedGenerator;
004import de.cuioss.test.generator.internal.net.java.quickcheck.generator.PrimitiveGenerators;
005
006/**
007 * This Class is used to generate a Short Object value which is generated by next() method.
008 *
009 */
010public class ShortObjectGenerator implements TypedGenerator<Short> {
011
012    @Override
013    public Short next() {
014        return PrimitiveGenerators.integers(-32768, 32767).next().shortValue();
015    }
016
017    @Override
018    public Class<Short> getType() {
019        return Short.class;
020    }
021
022}
