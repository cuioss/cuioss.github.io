001package io.cui.test.generator.junit;
002
003import java.lang.reflect.AnnotatedElement;
004import java.lang.reflect.Method;
005
006import org.junit.jupiter.api.extension.BeforeEachCallback;
007import org.junit.jupiter.api.extension.ExtendWith;
008import org.junit.jupiter.api.extension.Extension;
009import org.junit.jupiter.api.extension.ExtensionContext;
010import org.junit.jupiter.api.extension.TestExecutionExceptionHandler;
011import org.opentest4j.AssertionFailedError;
012import org.opentest4j.TestAbortedException;
013
014import io.cui.test.generator.internal.net.java.quickcheck.generator.distribution.RandomConfiguration;
015
016/**
017 * If enabled, either by using {@link ExtendWith} or {@link EnableGeneratorController} this
018 * {@link Extension} controls the seed initialization, by checking for {@link GeneratorSeed} and
019 * intercepts Test-failures by printing information providing the seed to reproduce.
020 *
021 * @author Oliver Wolff
022 *
023 */
024public class GeneratorControllerExtension implements BeforeEachCallback, TestExecutionExceptionHandler {
025
026    private static final String MSG_TEMPLATE = "%s\nGeneratorController seed was %sL. "
027            + "\nUse a fixed seed by applying @GeneratorSeed(%sL) for the method/class, "
028            + "\nor by using the system property '-D" + RandomConfiguration.SEED_SYSTEM_PROPERTY
029            + "=%s'\n";
030
031    @Override
032    public void handleTestExecutionException(ExtensionContext context, Throwable throwable) throws Throwable {
033
034        if (throwable instanceof TestAbortedException) {
035            throw throwable;
036        } else if (throwable instanceof AssertionFailedError) {
037            AssertionFailedError failure = new AssertionFailedError(createErrorMessage(throwable, RandomConfiguration.getLastSeed()));
038            failure.setStackTrace(throwable.getStackTrace());
039            throw failure;
040        } else {
041            AssertionFailedError failure =
042                new AssertionFailedError(throwable.getClass() + ": " + createErrorMessage(throwable, RandomConfiguration.getLastSeed()));
043            failure.setStackTrace(throwable.getStackTrace());
044            throw failure;
045        }
046
047    }
048
049    @Override
050    @SuppressWarnings("java:S3655") // owolff: false positive: isPresent is checked
051    public void beforeEach(ExtensionContext context) throws Exception {
052        boolean seedSetByAnnotation = false;
053        long initialSeed = -1;
054        if (context.getElement().isPresent()) {
055            AnnotatedElement annotatedElement = context.getElement().get();
056            GeneratorSeed seedAnnotation = annotatedElement.getAnnotation(GeneratorSeed.class);
057            if (null == seedAnnotation && (annotatedElement instanceof Method)) {
058                seedAnnotation = ((Method) annotatedElement).getDeclaringClass()
059                        .getAnnotation(GeneratorSeed.class);
060            }
061            if (null != seedAnnotation) {
062                initialSeed = seedAnnotation.value();
063                seedSetByAnnotation = true;
064            }
065        }
066        if (seedSetByAnnotation) {
067            RandomConfiguration.setSeed(initialSeed);
068        } else {
069            RandomConfiguration.initSeed();
070        }
071    }
072
073
074    private String createErrorMessage(Throwable e, Long seed) {
075        String causeMsg = e.getMessage() == null ? "" : e.getMessage();
076        return String.format(MSG_TEMPLATE, causeMsg, seed, seed, seed);
077    }
078
079}
