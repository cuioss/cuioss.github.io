017package io.cui.test.generator.internal.net.java.quickcheck.generator.support;
018
019import static io.cui.tools.collect.MoreCollections.requireNotEmpty;
020
021import java.util.List;
022
023import io.cui.test.generator.internal.net.java.quickcheck.Generator;
024
025public class RoundRobinGenerator<T> implements Generator<T> {
026
027    private final List<Generator<T>> generators;
028    private int index;
029
030    public RoundRobinGenerator(List<Generator<T>> generators) {
031        requireNotEmpty(generators, "generators");
032        this.generators = generators;
033        index = 0;
034    }
035
036    @Override
037    public T next() {
038        Generator<T> generator = generators.get(index++ % generators.size());
039        return generator.next();
040    }
041}
