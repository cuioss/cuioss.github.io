115    /**
116     * Convert a generator into an {@link Iterable iterable}.
117     *
118     * @param numberOfRuns to execute the runner
119     */
120    public static <T> Iterable<T> toIterable(final Generator<T> generator, final int numberOfRuns) {
121        Objects.requireNonNull(generator, "generator");
122        checkArgument(0.0<= numberOfRuns, "number of runs");
123        return () -> new Iterator<>() {
124
125            private int runs;
126
127            @Override
128            public boolean hasNext() {
129                return runs < numberOfRuns;
130            }
131
132            @Override
133            public T next() {
134                if (!hasNext())
135                    throw new NoSuchElementException();
136                runs++;
137                return generator.next();
138            }
139
140            @Override
141            public void remove() {
142                throw new UnsupportedOperationException();
143            }
144        };
145    }
