017package io.cui.test.generator.internal.net.java.quickcheck.generator.support;
018
019import java.util.Objects;
020
021import io.cui.test.generator.internal.net.java.quickcheck.ExtendibleGenerator;
022import io.cui.test.generator.internal.net.java.quickcheck.FrequencyGenerator;
023import io.cui.test.generator.internal.net.java.quickcheck.Generator;
024
025public class StringGenerator implements ExtendibleGenerator<Character, String> {
026
027    public static final int MIN_LENGTH = 0;
028
029    public static final int MAX_LENGTH = 30;
030
031    private final FrequencyGenerator<Character> characters;
032    private final Generator<Integer> length;
033
034    public StringGenerator() {
035        this(new CharacterGenerator());
036    }
037
038    public StringGenerator(char first, char last) {
039        this(new CharacterGenerator(first, last));
040    }
041
042    public StringGenerator(Generator<Character> characters) {
043        this(new IntegerGenerator(MIN_LENGTH, MAX_LENGTH), characters);
044    }
045
046    public StringGenerator(Generator<Integer> length,
047            Generator<Character> characters) {
048        Objects.requireNonNull(length, "length");
049
050        this.length = length;
051        this.characters = new DefaultFrequencyGenerator<>(characters);
052    }
053
054    @Override
055    public String next() {
056        int size = length.next();
057        StringBuilder builder = new StringBuilder();
058        for (int count = 0; count < size; count++) {
059            builder.append(characters.next());
060        }
061        return builder.toString();
062    }
063
064    @Override
065    public ExtendibleGenerator<Character, String> add(
066            Generator<Character> characterGenerator) {
067        characters.add(characterGenerator);
068        return this;
069    }
070}
