001package de.cuioss.test.jsf.junit5;
002
003import static de.cuioss.test.jsf.util.ConfigurationHelper.configureApplication;
004import static de.cuioss.test.jsf.util.ConfigurationHelper.configureComponents;
005import static de.cuioss.test.jsf.util.ConfigurationHelper.configureManagedBeans;
006import static de.cuioss.test.jsf.util.ConfigurationHelper.configureRequestConfig;
007
008import java.util.ArrayList;
009import java.util.LinkedHashSet;
010import java.util.List;
011import java.util.Objects;
012import java.util.Optional;
013import java.util.Set;
014import java.util.stream.Collectors;
015
016import org.apache.myfaces.test.mock.MockFacesContext;
017import org.junit.jupiter.api.extension.AfterEachCallback;
018import org.junit.jupiter.api.extension.ExtensionContext;
019import org.junit.jupiter.api.extension.ExtensionContext.Namespace;
020import org.junit.jupiter.api.extension.TestInstancePostProcessor;
021import org.junit.platform.commons.support.AnnotationSupport;
022
023import de.cuioss.test.jsf.config.JsfTestConfiguration;
024import de.cuioss.test.jsf.util.ConfigurableApplication;
025import de.cuioss.test.jsf.util.JsfEnvironmentConsumer;
026import de.cuioss.test.jsf.util.JsfEnvironmentHolder;
027import de.cuioss.test.jsf.util.JsfRuntimeSetup;
028
029/**
030 * Starts and Configures the {@link JsfRuntimeSetup}, wraps it into an {@link JsfEnvironmentHolder}
031 * and inject it into an {@link JsfEnvironmentConsumer} if available
032 *
033 * @author Oliver Wolff
034 *
035 */
036public class JsfSetupExtension implements TestInstancePostProcessor, AfterEachCallback {
037
038    /**
039     * Identifies the {@link Namespace} under which the concrete instance of {@link JsfRuntimeSetup}
040     * is stored .
041     */
042    public static final Namespace NAMESPACE = Namespace.create("test", "jsf", "JsfRuntimeSetup");
043
044    @Override
045    @SuppressWarnings("squid:S3655")
046    public void postProcessTestInstance(Object testInstance, ExtensionContext context) throws Exception {
047        var setup = new JsfRuntimeSetup();
048        setup.setUp();
049        put(setup, context);
050
051        var useIdentityResourceBundle = false;
052        List<EnableJsfEnvironment> environments = new ArrayList<>();
053        retrieveEnableJSFAnnotations(testInstance.getClass(), environments);
054        if (!environments.isEmpty()) {
055            // Use the outermost annotation
056            useIdentityResourceBundle = environments.get(0).useIdentityResourceBundle();
057        }
058
059        var environment = new JsfEnvironmentHolder(setup);
060        ConfigurableApplication.createWrapAndRegister((MockFacesContext) environment.getFacesContext())
061                .setUseIdentityResouceBundle(useIdentityResourceBundle);
062
063        // Ensure that the ConfigurableApplication is set for JsfRuntimeSetup#getApplication
064        setup.setApplication(environment.getFacesContext().getApplication());
065
066        Set<JsfTestConfiguration> decoratorAnnotations = new LinkedHashSet<>(16);
067
068        retrieveDecoratorAnnotations(testInstance.getClass(), decoratorAnnotations);
069        decoratorAnnotations = decoratorAnnotations.stream().filter(Objects::nonNull).collect(Collectors.toSet());
070
071        configureApplication(testInstance, environment.getApplicationConfigDecorator(), decoratorAnnotations);
072        configureComponents(testInstance, environment.getComponentConfigDecorator(), decoratorAnnotations);
073        configureManagedBeans(testInstance, environment.getBeanConfigDecorator(), decoratorAnnotations);
074        configureRequestConfig(testInstance, environment.getRequestConfigDecorator(), decoratorAnnotations);
075
076        if (testInstance instanceof JsfEnvironmentConsumer) {
077            ((JsfEnvironmentConsumer) testInstance).setEnvironmentHolder(environment);
078        }
079    }
080
081    private void retrieveDecoratorAnnotations(Class<?> type, Set<JsfTestConfiguration> result) {
082        if (null == type || Object.class.equals(type)) {
083            return;
084        }
085        result.addAll(AnnotationSupport.findRepeatableAnnotations(type, JsfTestConfiguration.class));
086        retrieveDecoratorAnnotations(type.getSuperclass(), result);
087    }
088
089    private void retrieveEnableJSFAnnotations(Class<?> type, List<EnableJsfEnvironment> result) {
090        if (null == type || Object.class.equals(type)) {
091            return;
092        }
093        result.addAll(AnnotationSupport.findRepeatableAnnotations(type, EnableJsfEnvironment.class));
094        retrieveEnableJSFAnnotations(type.getSuperclass(), result);
095    }
096
097    @Override
098    public void afterEach(ExtensionContext context) {
099        get(context).ifPresent(JsfRuntimeSetup::tearDown);
100    }
101
102    private static void put(JsfRuntimeSetup runtimeSetup, ExtensionContext context) {
103        context.getStore(NAMESPACE).put(JsfRuntimeSetup.class.getName(), runtimeSetup);
104    }
105
106    private Optional<JsfRuntimeSetup> get(ExtensionContext context) {
107        return Optional
108                .ofNullable((JsfRuntimeSetup) context.getStore(NAMESPACE).get(JsfRuntimeSetup.class.getName()));
109    }
110}
