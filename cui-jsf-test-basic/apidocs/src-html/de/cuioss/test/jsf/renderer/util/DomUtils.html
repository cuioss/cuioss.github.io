030    /**
031     * Creates an instance of {@link Document} for the given htmlString. It always uses
032     * {@link #ROOT_TEMPLATE} as the root element.
033     *
034     * @param htmlString must not be null
035     * @return the created {@link Document} with {@link #ROOT_TEMPLATE} as the root element.
036     */
037    public static Document htmlStringToDocument(final String htmlString) {
038        requireNonNull(htmlString);
039        final var wrappedInput = String.format(ROOT_TEMPLATE, htmlString);
040        try (var input = IOStreams.toInputStream(wrappedInput)) {
041            var saxBuilder = new SAXBuilder();
042            saxBuilder.setProperty(XMLConstants.ACCESS_EXTERNAL_DTD, "");
043            saxBuilder.setProperty(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
044            return saxBuilder.build(input);
045        } catch (JDOMException | IOException e) {
046            throw new IllegalArgumentException("Unable to parse given String, due to ", e);
047        }
048    }

050    /**
051     * Extracts all attributes with the given name. The method will recursively check all children
052     * as well.
053     *
054     * @param element to be checked, must not be null.
055     * @param attributeName to be looked for, must not be null nor empty.
056     * @return a {@link List} with the found attributes, never null but may be empty.
057     */
058    public static List<Attribute> filterForAttribute(final Element element, final String attributeName) {
059        requireNonNull(element);
060        requireNonNull(emptyToNull(attributeName));
061        List<Attribute> found = new ArrayList<>();
062        var current = element.getAttribute(attributeName);
063        if (null != current) {
064            found.add(current);
065        }
066        for (Element child : element.getChildren()) {
067            found.addAll(filterForAttribute(child, attributeName));
068        }
069        return found;
070    }

072    /**
073     * Extracts all attributes with the given name and the attribute-value containing the given
074     * String. The method will recursively check all children as well.
075     *
076     * @param element to be checked, must not be null.
077     * @param attributeName to be looked for, must not be null nor empty.
078     * @param attributeValuePart the string of the attribute value to be filtered for.
079     * @return a {@link List} with the found attributes, never null but may be empty.
080     */
081    public static List<Attribute> filterForAttributeContainingValue(final Element element, final String attributeName, final String attributeValuePart) {
082        requireNonNull(emptyToNull(attributeValuePart));
083        return filterForAttribute(element, attributeName).stream().filter(a -> a.getValue().contains(attributeValuePart)).collect(Collectors.toList());
084    }
