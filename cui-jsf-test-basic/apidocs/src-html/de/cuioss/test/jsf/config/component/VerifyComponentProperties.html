001package de.cuioss.test.jsf.config.component;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Repeatable;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008import java.util.Collection;
009
010import javax.el.ValueExpression;
011
012import de.cuioss.test.jsf.component.AbstractComponentTest;
013import de.cuioss.test.valueobjects.api.property.PropertyConfig;
014import de.cuioss.test.valueobjects.property.PropertyMetadata;
015import de.cuioss.test.valueobjects.property.util.AssertionStrategy;
016
017/**
018 * If used on {@link AbstractComponentTest} type configures the property-tests.
019 * <p>
020 * In essence it checks the getters and setters. As default it assumes the individual properties to
021 * not provide a default value. This can be controlled using {@link #defaultValued()}
022 * <p>
023 *
024 * @author Oliver Wolff
025 */
026@Repeatable(VerifyComponentPropertiesRepeat.class)
027@Retention(RetentionPolicy.RUNTIME)
028@Target({ ElementType.TYPE })
029public @interface VerifyComponentProperties {
030
031    /**
032     * @return a number of properties, identified by their names that are to be considered for
033     *         this test: white-list
034     */
035    String[] of() default {};
036
037    /**
038     * @return a number of properties, identified by their names that are to be treated as having a
039     *         default values, see {@link PropertyMetadata#isDefaultValue()}
040     */
041    String[] defaultValued() default {};
042
043    /**
044     * @return a number of properties, identified by their names that are to be ignore on the tests
045     *         with {@link ValueExpression}s. <em>Caution</em> the need of the configuration hints
046     *         on an invalid implementation, because all properties should consider
047     *         {@link ValueExpression}s.
048     */
049    String[] noValueExpression() default {};
050
051    /**
052     * @return a number of properties, identified by their names representing at least a
053     *         {@link Collection} that are to be asserted ignoring the concrete order, see
054     *         {@link PropertyConfig#assertionStrategy()} and
055     *         {@link AssertionStrategy#COLLECTION_IGNORE_ORDER}. The default implementation will
056     *         always respect / assert the same order of elements.
057     */
058    String[] assertUnorderedCollection() default {};
059}
