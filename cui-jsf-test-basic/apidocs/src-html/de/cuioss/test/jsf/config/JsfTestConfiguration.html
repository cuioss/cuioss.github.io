001package de.cuioss.test.jsf.config;
002
003import static java.lang.annotation.ElementType.TYPE;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Repeatable;
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009
010/**
011 * This annotations can be used to configure certain aspects of the configuration of the jsf-runtime
012 * for unit-testing. For simple cases the test-class can implement the corresponding interfaces
013 * itself: {@link ApplicationConfigurator}, {@link ComponentConfigurator},
014 * {@link RequestConfigurator} or {@link BeanConfigurator}. The base-class will ensure that the
015 * corresponding methods will be called at setup-time. This is useful for cases, where there is no
016 * need to reuse a certain configuration.
017 *
018 * @author Oliver Wolff
019 */
020@Retention(RUNTIME)
021@Target(TYPE)
022@Repeatable(JsfTestConfigurations.class)
023public @interface JsfTestConfiguration {
024
025    /**
026     * @return one or more concrete instances of {@link JsfTestContextConfigurator}
027     */
028    Class<? extends JsfTestContextConfigurator>[] value();
029
030}
