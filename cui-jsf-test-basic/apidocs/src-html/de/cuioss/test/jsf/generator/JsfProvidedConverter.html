001package de.cuioss.test.jsf.generator;
002
003import static de.cuioss.tools.collect.CollectionLiterals.immutableList;
004
005import java.math.BigDecimal;
006import java.math.BigInteger;
007import java.util.Date;
008import java.util.List;
009import java.util.stream.Collectors;
010
011import javax.faces.convert.BigDecimalConverter;
012import javax.faces.convert.BigIntegerConverter;
013import javax.faces.convert.BooleanConverter;
014import javax.faces.convert.ByteConverter;
015import javax.faces.convert.CharacterConverter;
016import javax.faces.convert.Converter;
017import javax.faces.convert.DateTimeConverter;
018import javax.faces.convert.DoubleConverter;
019import javax.faces.convert.EnumConverter;
020import javax.faces.convert.FloatConverter;
021import javax.faces.convert.IntegerConverter;
022import javax.faces.convert.LongConverter;
023import javax.faces.convert.NumberConverter;
024import javax.faces.convert.ShortConverter;
025
026import de.cuioss.test.generator.Generators;
027import de.cuioss.test.generator.TypedGenerator;
028import de.cuioss.test.jsf.config.ComponentConfigurator;
029
030/**
031 * Hybrid class that acts as an {@link TypedGenerator} and {@link ComponentConfigurator}.
032 *
033 * @author Oliver Wolff
034 */
035@SuppressWarnings("rawtypes")
036public class JsfProvidedConverter implements TypedGenerator<Class> {
037
038    /** Defines all converter types provided by JSF. */
039    @SuppressWarnings("java:S2386") // owolff: false positive -> immutableList
040    public static final List<ConverterDescriptor> JSF_CONVERTER =
041    immutableList(new ConverterDescriptor(BigDecimalConverter.class, BigDecimal.class,
042            BigDecimalConverter.CONVERTER_ID),
043            new ConverterDescriptor(BigIntegerConverter.class, BigInteger.class,
044                    BigIntegerConverter.CONVERTER_ID),
045            new ConverterDescriptor(BooleanConverter.class, Boolean.class,
046                    BooleanConverter.CONVERTER_ID),
047            new ConverterDescriptor(ByteConverter.class, Byte.class,
048                    ByteConverter.CONVERTER_ID),
049            new ConverterDescriptor(CharacterConverter.class, Character.class,
050                    CharacterConverter.CONVERTER_ID),
051            new ConverterDescriptor(DateTimeConverter.class, Date.class,
052                    DateTimeConverter.CONVERTER_ID),
053            new ConverterDescriptor(DoubleConverter.class, Double.class,
054                    DoubleConverter.CONVERTER_ID),
055            new ConverterDescriptor(EnumConverter.class, Enum.class,
056                    EnumConverter.CONVERTER_ID),
057            new ConverterDescriptor(FloatConverter.class, Float.class,
058                    FloatConverter.CONVERTER_ID),
059            new ConverterDescriptor(IntegerConverter.class, Integer.class,
060                    IntegerConverter.CONVERTER_ID),
061            new ConverterDescriptor(LongConverter.class, Long.class,
062                    LongConverter.CONVERTER_ID),
063            new ConverterDescriptor(NumberConverter.class, Number.class,
064                    NumberConverter.CONVERTER_ID),
065            new ConverterDescriptor(ShortConverter.class, Short.class,
066                    ShortConverter.CONVERTER_ID));
067
068    /** Defines a generator for all converter types provided by JSF. */
069    public static final TypedGenerator<ConverterDescriptor> CONVERTER_CLASS_GERNERATOR =
070            Generators.fixedValues(JSF_CONVERTER);
071
072    /** A generator for every id of registered generator. */
073    public static final TypedGenerator<String> CONVERTER_ID_GENERATOR =
074            Generators.fixedValues(
075                    JSF_CONVERTER.stream().map(ConverterDescriptor::getConverterId).collect(Collectors.toList()));
076
077    /** A generator for a subset of types a generator is registered as default. */
078    public static final TypedGenerator<Class<?>> TARGET_TYPE_GENERATOR =
079            Generators.fixedValues(
080                    immutableList(Integer.class, Double.class, Float.class, Boolean.class,
081                            BigInteger.class));
082
083    @SuppressWarnings("unchecked")
084    @Override
085    public Class<Converter> next() {
086        return (Class<Converter>) CONVERTER_CLASS_GERNERATOR.next().getConverterClass();
087    }
088
089    @Override
090    public Class<Class> getType() {
091        return Class.class;
092    }
093
094}
