001package de.cuioss.test.jsf.producer;
002
003import javax.enterprise.context.RequestScoped;
004import javax.enterprise.inject.Produces;
005import javax.enterprise.inject.Typed;
006import javax.faces.context.FacesContext;
007import javax.servlet.http.HttpServletResponse;
008
009/**
010 * Produces a {@link RequestScoped} {@link HttpServletResponse}.
011 * This class has no bean annotation. It is designed as 'opt-in'. Use with {@code @AddBeanClasses}.
012 */
013public class TestHttpServletResponseProducer {
014
015    @Produces
016    @Typed({HttpServletResponse.class})
017    @RequestScoped
018    HttpServletResponse getServletResponse() {
019        return (HttpServletResponse) FacesContext.getCurrentInstance().getExternalContext().getResponse();
020    }
021}
