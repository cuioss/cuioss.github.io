395     * <p>
396     * Left pad a String with a specified String.
397     * </p>
398     *
399     * <p>
400     * Pad to a size of {@code size}.
401     * </p>
402     *
403     * <pre>
404     * MoreStrings.leftPad(null, *, *)      = null
405     * MoreStrings.leftPad("", 3, "z")      = "zzz"
406     * MoreStrings.leftPad("bat", 3, "yz")  = "bat"
407     * MoreStrings.leftPad("bat", 5, "yz")  = "yzbat"
408     * MoreStrings.leftPad("bat", 8, "yz")  = "yzyzybat"
409     * MoreStrings.leftPad("bat", 1, "yz")  = "bat"
410     * MoreStrings.leftPad("bat", -1, "yz") = "bat"
411     * MoreStrings.leftPad("bat", 5, null)  = "  bat"
412     * MoreStrings.leftPad("bat", 5, "")    = "  bat"
413     * </pre>
414     *
415     * @param str the String to pad out, may be null
416     * @param size the size to pad to
417     * @param padStr the String to pad with, null or empty treated as single space
418     * @return left padded String or original String if no padding is necessary,
419     *         {@code null} if null String input
420     * @author https://github.com/apache/commons-lang/blob/master/src/main/java/org/apache/commons/lang3/StringUtils.java
421     */
422    public static String leftPad(final String str, final int size, String padStr) {
423        if (str == null) {
424            return null;
425        }
426        if (isEmpty(padStr)) {
427            padStr = SPACE;
428        }
429        final int padLen = padStr.length();
430        final int strLen = str.length();
431        final int pads = size - strLen;
432        if (pads <= 0) {
433            return str; // returns original String when possible
434        }
435        if (padLen == 1 && pads <= PAD_LIMIT) {
436            return leftPad(str, size, padStr.charAt(0));
437        }
438        if (pads == padLen) {
439            return padStr.concat(str);
440        } else if (pads < padLen) {
441            return padStr.substring(0, pads).concat(str);
442        } else {
443            final char[] padding = new char[pads];
444            final char[] padChars = padStr.toCharArray();
445            for (int i = 0; i < pads; i++) {
446                padding[i] = padChars[i % padLen];
447            }
448            return new String(padding).concat(str);
449        }
450    }

522    public static String stripEnd(final String str, final String stripChars) {
523        int end;
524        if (str == null || (end = str.length()) == 0) {
525            return str;
526        }
527        if (stripChars == null) {
528            while (end != 0 && Character.isWhitespace(str.charAt(end - 1))) {
529                end--;
530            }
531        } else if (stripChars.isEmpty()) {
532            return str;
533        } else {
534            while (end != 0 && stripChars.indexOf(str.charAt(end - 1)) != INDEX_NOT_FOUND) {
535                end--;
536            }
537        }
538        return str.substring(0, end);

634     * (Unicode code units).
635     *
636     * @param cs the {@code CharSequence} to be processed, not null
637     * @param searchChar the char to be searched for
638     * @param start the start index, negative starts at the string start
639     * @return the index where the search char was found, -1 if not found
640     * @author https://github.com/apache/commons-lang/blob/master/src/main/java/org/apache/commons/lang3/StringUtils.java
641     */
642    @SuppressWarnings("squid:S3776")
643    public static int indexOf(final CharSequence cs, final int searchChar, int start) {
644        if (isEmpty(cs)) {
645            return INDEX_NOT_FOUND;
646        }
647        if (cs instanceof String) {
648            return ((String) cs).indexOf(searchChar, start);
649        }
650        final int sz = cs.length();
651        if (start < 0) {
652            start = 0;
653        }
654        if (searchChar < Character.MIN_SUPPLEMENTARY_CODE_POINT) {
655            for (int i = start; i < sz; i++) {
656                if (cs.charAt(i) == searchChar) {
657                    return i;
658                }
659            }
660        }
661        // supplementary characters (LANG1300)
662        if (searchChar <= Character.MAX_CODE_POINT) {
663            final char[] chars = Character.toChars(searchChar);
664            for (int i = start; i < sz - 1; i++) {
665                final char high = cs.charAt(i);
666                final char low = cs.charAt(i + 1);
667                if (high == chars[0] && low == chars[1]) {
668                    return i;
669                }
670            }
671        }
672        return INDEX_NOT_FOUND;
673    }

695     * More specifically, this method returns {@code true} if the

807    /**
808     * Returns the given {@code template} string with each occurrence of {@code "%s"} replaced with
809     * the corresponding argument value from {@code args}; or, if the placeholder and argument
810     * counts
811     * do not match, returns a best-effort form of that string. Will not throw an exception under
812     * normal conditions.
813     *
814     * <p>
815     * <b>Note:</b> For most string-formatting needs, use {@link String#format String.format},
816     * {@link java.io.PrintWriter#format PrintWriter.format}, and related methods. These support the
817     * full range of <a
818     * href="https://docs.oracle.com/javase/9/docs/api/java/util/Formatter.html#syntax">format
819     * specifiers</a>, and alert you to usage errors by throwing {@link
820     * java.util.IllegalFormatException}.
821     *
822     * <p>
823     * In certain cases, such as outputting debugging information or constructing a message to be
824     * used for another unchecked exception, an exception during string formatting would serve
825     * little purpose except to supplant the real information you were trying to provide. These are
826     * the cases this method is made for; it instead generates a best-effort string with all
827     * supplied argument values present. This method is also useful in environments such as GWT
828     * where {@code String.format} is not available.
829     *
830     * <p>
831     * <b>Warning:</b> Only the exact two-character placeholder sequence {@code "%s"} is
832     * recognized.
833     *
834     * @author com.google.common.base.Strings
835     * @param template a string containing zero or more {@code "%s"} placeholder sequences. {@code
836     *     null} is treated as the four-character string {@code "null"}.
837     * @param args the arguments to be substituted into the message template. The first argument
838     *            specified is substituted for the first occurrence of {@code "%s"} in the template,
839     *            and so forth. A {@code null} argument is converted to the four-character string
840     *            {@code "null"}; non-null values are converted to strings using {@link Object#toString()}.
841     * @return the resulting formatting String
842     */
843    public static String lenientFormat(String template, Object... args) {
844        template = String.valueOf(template); // null -> "null"
845        if (args == null) {
846            args = new Object[] {"(Object[])null"};
847        } else {
848            for (int i = 0; i < args.length; i++) {
849                args[i] = lenientToString(args[i]);
850            }
851        }
852        // start substituting the arguments into the '%s' placeholders
853        StringBuilder builder = new StringBuilder(template.length() + 16 * args.length);
854        int templateStart = 0;
855        int i = 0;
856        while (i < args.length) {
857            int placeholderStart = template.indexOf("%s", templateStart);
858            if (placeholderStart == -1) {
859                break;
860            }
861            builder.append(template, templateStart, placeholderStart);
862            builder.append(args[i++]);
863            templateStart = placeholderStart + 2;
864        }
865        builder.append(template, templateStart, template.length());
866        // if we run out of placeholders, append the extra args in square braces
867        if (i < args.length) {
868            builder.append(" [");
869            builder.append(args[i++]);
870            while (i < args.length) {
871                builder.append(", ");
872                builder.append(args[i++]);
873            }
874            builder.append(']');
875        }
876        return builder.toString();
877    }

884    public static String ensureEndsWith(@NonNull final String value, @NonNull final String suffix) {
885        if (value == null) {
886            throw new java.lang.NullPointerException("value is marked non-null but is null");
887        }
888        if (suffix == null) {
889            throw new java.lang.NullPointerException("suffix is marked non-null but is null");

892            return value + suffix;

898     * @author com.google.common.base.Strings
899     */
900    static String lenientToString(Object o) {
901        try {
902            return String.valueOf(o);
903        } catch (Exception e) {
904            // Default toString() behavior - see Object.toString()
905            String objectToString = o.getClass().getName() + '@' + Integer.toHexString(System.identityHashCode(o));
906            log.warn(e, "Exception during lenientFormat for {}", objectToString);
907            return "<" + objectToString + " threw " + e.getClass().getName() + ">";
908        }

918    public static Optional<String> coalesce(Predicate<String> checker, String... values) {
919        if (null != values) {
