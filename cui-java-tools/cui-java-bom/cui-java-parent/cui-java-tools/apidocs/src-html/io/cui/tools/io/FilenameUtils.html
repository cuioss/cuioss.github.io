010/**
011 * Copied from commons.io:org.apache.commons.io.FilenameUtils
012 * <p>
013 * General filename and filepath manipulation utilities.
014 * <p>
015 * When dealing with filenames you can hit problems when moving from a Windows
016 * based development machine to a Unix based production machine.
017 * This class aims to help avoid those problems.
018 * <p>
019 * <b>NOTE</b>: You may be able to avoid using this class entirely simply by
020 * using JDK {@link java.io.File File} objects and the two argument constructor
021 * {@link java.io.File#File(java.io.File, java.lang.String) File(File,String)}.
022 * <p>
023 * Most methods on this class are designed to work the same on both Unix and Windows.
024 * Those that don't include 'System', 'Unix' or 'Windows' in their name.
025 * <p>
026 * Most methods recognise both separators (forward and back), and both
027 * sets of prefixes. See the javadoc of each method for details.
028 * <p>
029 * This class defines six components within a filename
030 * (example C:\dev\project\file.txt):
031 * <ul>
032 * <li>the prefix - C:\</li>
033 * <li>the path - dev\project\</li>
034 * <li>the full path - C:\dev\project\</li>
035 * <li>the name - file.txt</li>
036 * <li>the base name - file</li>
037 * <li>the extension - txt</li>
038 * </ul>
039 * Note that this class works best if directory filenames end with a separator.
040 * If you omit the last separator, it is impossible to determine if the filename
041 * corresponds to a file or a directory. As a result, we have chosen to say
042 * it corresponds to a file.
043 * <p>
044 * This class only supports Unix and Windows style names.
045 * Prefixes are matched as follows:
046 *
047 * <pre>
048 * Windows:
049 * a\b\c.txt           --&gt; ""          --&gt; relative
050 * \a\b\c.txt          --&gt; "\"         --&gt; current drive absolute
051 * C:a\b\c.txt         --&gt; "C:"        --&gt; drive relative
052 * C:\a\b\c.txt        --&gt; "C:\"       --&gt; absolute
053 * \\server\a\b\c.txt  --&gt; "\\server\" --&gt; UNC
054 *
055 * Unix:
056 * a/b/c.txt           --&gt; ""          --&gt; relative
057 * /a/b/c.txt          --&gt; "/"         --&gt; absolute
058 * ~/a/b/c.txt         --&gt; "~/"        --&gt; current user
059 * ~                   --&gt; "~/"        --&gt; current user (slash added)
060 * ~user/a/b/c.txt     --&gt; "~user/"    --&gt; named user
061 * ~user               --&gt; "~user/"    --&gt; named user (slash added)
062 * </pre>
063 *
064 * Both prefix styles are matched always, irrespective of the machine that you are
065 * currently running on.
066 * <p>
067 * Origin of code: Excalibur, Alexandria, Tomcat, Commons-Utils.
068 * </p>
069 */

322    /**
323     * Internal method to perform the normalization.
324     *
325     * @param filename the filename
326     * @param separator The separator character to use
327     * @param keepSeparator true to keep the final separator
328     * @return the normalized filename. Null bytes inside string will be removed.
329     */
330    @SuppressWarnings({"squid:S3776", "squid:LabelsShouldNotBeUsedCheck", "squid:ForLoopCounterChangedCheck"})
331    private static String doNormalize(final String filename, final char separator, final boolean keepSeparator) {
332        if (filename == null) {
333            return null;
334        }
335        failIfNullBytePresent(filename);
336        int size = filename.length();
337        if (size == 0) {
338            return filename;
339        }
340        final int prefix = getPrefixLength(filename);
341        if (prefix < 0) {
342            return null;
343        }
344        final char[] array = new char[size + 2]; // +1 for possible extra slash, +2 for arraycopy
345        filename.getChars(0, filename.length(), array, 0);
346        // fix separators throughout
347        final char otherSeparator = separator == SYSTEM_SEPARATOR ? OTHER_SEPARATOR : SYSTEM_SEPARATOR;
348        for (int i = 0; i < array.length; i++) {
349            if (array[i] == otherSeparator) {
350                array[i] = separator;
351            }
352        }
353        // add extra separator on the end to simplify code below
354        boolean lastIsDirectory = true;
355        if (array[size - 1] != separator) {
356            array[size++] = separator;
357            lastIsDirectory = false;
358        }
359        // adjoining slashes
360        for (int i = prefix + 1; i < size; i++) {
361            if (array[i] == separator && array[i - 1] == separator) {
362                System.arraycopy(array, i, array, i - 1, size - i);
363                size--;
364                i--;
365            }
366        }
367        // dot slash
368        for (int i = prefix + 1; i < size; i++) {
369            if (array[i] == separator && array[i - 1] == '.' && (i == prefix + 1 || array[i - 2] == separator)) {
370                if (i == size - 1) {
371                    lastIsDirectory = true;
372                }
373                System.arraycopy(array, i + 1, array, i - 1, size - i);
374                size -= 2;
375                i--;
376            }
377        }
378        // double dot slash
379        outer:
380        for (int i = prefix + 2; i < size; i++) {
381            if (array[i] == separator && array[i - 1] == '.' && array[i - 2] == '.' && (i == prefix + 2 || array[i - 3] == separator)) {
382                if (i == prefix + 2) {
383                    return null;
384                }
385                if (i == size - 1) {
386                    lastIsDirectory = true;
387                }
388                int j;
389                for (j = i - 4; j >= prefix; j--) {
390                    if (array[j] == separator) {
391                        // remove b/../ from a/b/../c
392                        System.arraycopy(array, i + 1, array, j + 1, size - i);
393                        size -= i - j;
394                        i = j + 1;
395                        continue outer;
396                    }
397                }
398                // remove a/../ from a/../c
399                System.arraycopy(array, i + 1, array, prefix, size - i);
400                size -= i + 1 - prefix;
401                i = prefix + 1;
402            }
403        }
404        if (size <= 0) {
405            // should never be less than 0
406            return "";
407        }
408        if ((size <= prefix) || (lastIsDirectory && keepSeparator)) {
409            return new String(array, 0, size); // keep trailing separator
410        }
411        return new String(array, 0, size - 1); // lose trailing separator
412    }

415    /**
416     * Concatenates a filename to a base path using normal command line style rules.
417     * <p>
418     * The effect is equivalent to resultant directory after changing
419     * directory to the first argument, followed by changing directory to
420     * the second argument.
421     * <p>
422     * The first argument is the base path, the second is the path to concatenate.
423     * The returned path is always normalized via {@link #normalize(String)},
424     * thus <code>..</code> is handled.
425     * <p>
426     * If <code>pathToAdd</code> is absolute (has an absolute prefix), then
427     * it will be normalized and returned.
428     * Otherwise, the paths will be joined, normalized and returned.
429     * <p>
430     * The output will be the same on both Unix and Windows except
431     * for the separator character.
432     *
433     * <pre>
434     * /foo/ + bar          --&gt;   /foo/bar
435     * /foo + bar           --&gt;   /foo/bar
436     * /foo + /bar          --&gt;   /bar
437     * /foo + C:/bar        --&gt;   C:/bar
438     * /foo + C:bar         --&gt;   C:bar (*)
439     * /foo/a/ + ../bar     --&gt;   foo/bar
440     * /foo/ + ../../bar    --&gt;   null
441     * /foo/ + /bar         --&gt;   /bar
442     * /foo/.. + /bar       --&gt;   /bar
443     * /foo + bar/c.txt     --&gt;   /foo/bar/c.txt
444     * /foo/c.txt + bar     --&gt;   /foo/c.txt/bar (!)
445     * </pre>
446     *
447     * (*) Note that the Windows relative drive prefix is unreliable when
448     * used with this method.
449     * (!) Note that the first parameter must be a path. If it ends with a name, then
450     * the name will be built into the concatenated path. If this might be a problem,
451     * use {@link #getFullPath(String)} on the base path argument.
452     *
453     * @param basePath the base path to attach to, always treated as a path
454     * @param fullFilenameToAdd the filename (or path) to attach to the base
455     * @return the concatenated path, or null if invalid. Null bytes inside string will be removed
456     */
457    public static String concat(final String basePath, final String fullFilenameToAdd) {
458        final int prefix = getPrefixLength(fullFilenameToAdd);
459        if (prefix < 0) {
460            return null;
461        }
462        if (prefix > 0) {
463            return normalize(fullFilenameToAdd);
464        }
465        if (basePath == null) {
466            return null;
467        }
468        final int len = basePath.length();
469        if (len == 0) {
470            return normalize(fullFilenameToAdd);
471        }
472        final char ch = basePath.charAt(len - 1);
473        if (isSeparator(ch)) {
474            return normalize(basePath + fullFilenameToAdd);
475        } else {
476            return normalize(basePath + '/' + fullFilenameToAdd);
477        }
478    }

480    /**
481     * Determines whether the {@code parent} directory contains the {@code child} element (a file or
482     * directory).
483     * <p>
484     * The files names are expected to be normalized.
485     * </p>
486     *
487     * Edge cases:
488     * <ul>
489     * <li>A {@code directory} must not be null: if null, throw IllegalArgumentException</li>
490     * <li>A directory does not contain itself: return false</li>
491     * <li>A null child file is not contained in any parent: return false</li>
492     * </ul>
493     *
494     * @param canonicalParent
495     *            the file to consider as the parent.
496     * @param canonicalChild
497     *            the file to consider as the child.
498     * @return true is the candidate leaf is under by the specified composite. False otherwise.
499     */
500    public static boolean directoryContains(final String canonicalParent, final String canonicalChild) {
501        // Fail fast against NullPointerException
502        if (canonicalParent == null) {
503            throw new IllegalArgumentException("Directory must not be null");
504        }
505        if ((canonicalChild == null) || IOCase.SYSTEM.checkEquals(canonicalParent, canonicalChild)) {
506            return false;
507        }
508        return IOCase.SYSTEM.checkStartsWith(canonicalChild, canonicalParent);
509    }

557    /**
558     * Returns the length of the filename prefix, such as <code>C:/</code> or <code>~/</code>.
559     * <p>
560     * This method will handle a file in either Unix or Windows format.
561     * <p>
562     * The prefix length includes the first slash in the full filename
563     * if applicable. Thus, it is possible that the length returned is greater
564     * than the length of the input string.
565     *
566     * <pre>
567     * Windows:
568     * a\b\c.txt           --&gt; ""          --&gt; relative
569     * \a\b\c.txt          --&gt; "\"         --&gt; current drive absolute
570     * C:a\b\c.txt         --&gt; "C:"        --&gt; drive relative
571     * C:\a\b\c.txt        --&gt; "C:\"       --&gt; absolute
572     * \\server\a\b\c.txt  --&gt; "\\server\" --&gt; UNC
573     * \\\a\b\c.txt        --&gt;  error, length = -1
574     *
575     * Unix:
576     * a/b/c.txt           --&gt; ""          --&gt; relative
577     * /a/b/c.txt          --&gt; "/"         --&gt; absolute
578     * ~/a/b/c.txt         --&gt; "~/"        --&gt; current user
579     * ~                   --&gt; "~/"        --&gt; current user (slash added)
580     * ~user/a/b/c.txt     --&gt; "~user/"    --&gt; named user
581     * ~user               --&gt; "~user/"    --&gt; named user (slash added)
582     * //server/a/b/c.txt  --&gt; "//server/"
583     * ///a/b/c.txt        --&gt; error, length = -1
584     * </pre>
585     * <p>
586     * The output will be the same irrespective of the machine that the code is running on.
587     * i.e. both Unix and Windows prefixes are matched regardless.
588     * <p>
589     * Note that a leading // (or \\) is used to indicate a UNC name on Windows.
590     * These must be followed by a server name, so double-slashes are not collapsed
591     * to a single slash at the start of the filename.
592     *
593     * @param filename the filename to find the prefix in, null returns -1
594     * @return the length of the prefix, -1 if invalid or null
595     */
596    @SuppressWarnings({"squid:S3776"})
597    public static int getPrefixLength(final String filename) {
598        if (filename == null) {
599            return NOT_FOUND;
600        }
601        final int len = filename.length();
602        if (len == 0) {
603            return 0;
604        }
605        char ch0 = filename.charAt(0);
606        if (ch0 == ':') {
607            return NOT_FOUND;
608        }
609        if (len == 1) {
610            if (ch0 == '~') {
611                return 2; // return a length greater than the input
612            }
613            return isSeparator(ch0) ? 1 : 0;
614        } else {
615            if (ch0 == '~') {
616                int posUnix = filename.indexOf(UNIX_SEPARATOR, 1);
617                int posWin = filename.indexOf(WINDOWS_SEPARATOR, 1);
618                if (posUnix == NOT_FOUND && posWin == NOT_FOUND) {
619                    return len + 1; // return a length greater than the input
620                }
621                posUnix = posUnix == NOT_FOUND ? posWin : posUnix;
622                posWin = posWin == NOT_FOUND ? posUnix : posWin;
623                return Math.min(posUnix, posWin) + 1;
624            }
625            final char ch1 = filename.charAt(1);
626            if (ch1 == ':') {
627                ch0 = Character.toUpperCase(ch0);
628                if (ch0 >= 'A' && ch0 <= 'Z') {
629                    if (len == 2 || !isSeparator(filename.charAt(2))) {
630                        return 2;
631                    }
632                    return 3;
633                } else if (ch0 == UNIX_SEPARATOR) {
634                    return 1;
635                }
636                return NOT_FOUND;
637            } else if (isSeparator(ch0) && isSeparator(ch1)) {
638                int posUnix = filename.indexOf(UNIX_SEPARATOR, 2);
639                int posWin = filename.indexOf(WINDOWS_SEPARATOR, 2);
640                if (posUnix == NOT_FOUND && posWin == NOT_FOUND || posUnix == 2 || posWin == 2) {
641                    return NOT_FOUND;
642                }
643                posUnix = posUnix == NOT_FOUND ? posWin : posUnix;
644                posWin = posWin == NOT_FOUND ? posUnix : posWin;
645                return Math.min(posUnix, posWin) + 1;
646            } else {
647                return isSeparator(ch0) ? 1 : 0;
648            }
649        }
650    }

892     *
893     * @param filename the filename
894     * @param includeSeparator true to include the end separator
895     * @return the path
896     */
897    private static String doGetFullPath(final String filename, final boolean includeSeparator) {
898        if (filename == null) {
899            return null;
900        }
901        final int prefix = getPrefixLength(filename);
902        if (prefix < 0) {
903            return null;
904        }
905        if (prefix >= filename.length()) {
906            if (includeSeparator) {
907                return getPrefix(filename); // add end slash if necessary

909                return filename;

914            return filename.substring(0, prefix);
915        }
916        int end = index + (includeSeparator ? 1 : 0);
917        if (end == 0) {
918            end++;
919        }
920        return filename.substring(0, end);
921    }

1281    /**
1282     * Checks a filename to see if it matches the specified wildcard matcher
1283     * allowing control over case-sensitivity.
1284     * <p>
1285     * The wildcard matcher uses the characters '?' and '*' to represent a
1286     * single or multiple (zero or more) wildcard characters.
1287     * N.B. the sequence "*?" does not work properly at present in match strings.
1288     *
1289     * @param filename the filename to match on
1290     * @param wildcardMatcher the wildcard string to match against
1291     * @param caseSensitivity what case sensitivity rule to use, null means case-sensitive
1292     * @return true if the filename matches the wildcard string
1293     */
1294    @SuppressWarnings({"squid:S3776", "squid:S135"})
1295    public static boolean wildcardMatch(final String filename, final String wildcardMatcher, IOCase caseSensitivity) {
1296        if (filename == null && wildcardMatcher == null) {
1297            return true;
1298        }
1299        if (filename == null || wildcardMatcher == null) {
1300            return false;
1301        }
1302        if (caseSensitivity == null) {
1303            caseSensitivity = IOCase.SENSITIVE;
1304        }
1305        final String[] wcs = splitOnTokens(wildcardMatcher);
1306        boolean anyChars = false;
1307        int textIdx = 0;
1308        int wcsIdx = 0;
1309        final Deque<int[]> backtrack = new ArrayDeque<>();
1310        // loop around a backtrack stack, to handle complex * matching
1311        do {
1312            if (!backtrack.isEmpty()) {
1313                final int[] array = backtrack.pop();
1314                wcsIdx = array[0];
1315                textIdx = array[1];
1316                anyChars = true;
1317            }
1318            // loop whilst tokens and text left to process
1319            while (wcsIdx < wcs.length) {
1320                if (wcs[wcsIdx].equals("?")) {
1321                    // ? so move to next text char
1322                    textIdx++;
1323                    if (textIdx > filename.length()) {
1324                        break;
1325                    }
1326                    anyChars = false;
1327                } else if (wcs[wcsIdx].equals("*")) {
1328                    // set any chars status
1329                    anyChars = true;
1330                    if (wcsIdx == wcs.length - 1) {
1331                        textIdx = filename.length();
1332                    }
1333                } else {
1334                    // matching text token
1335                    if (anyChars) {
1336                        // any chars then try to locate text token
1337                        textIdx = caseSensitivity.checkIndexOf(filename, textIdx, wcs[wcsIdx]);
1338                        if (textIdx == NOT_FOUND) {
1339                            // token not found
1340                            break;
1341                        }
1342                        final int repeat = caseSensitivity.checkIndexOf(filename, textIdx + 1, wcs[wcsIdx]);
1343                        if (repeat >= 0) {
1344                            backtrack.push(new int[] {wcsIdx, repeat});
1345                        }
1346                    } else {
1347                        // matching from current position
1348                        if (!caseSensitivity.checkRegionMatches(filename, textIdx, wcs[wcsIdx])) {
1349                            // couldn't match token
1350                            break;
1351                        }
1352                    }
1353                    // matched text token, move text index to end of matched token
1354                    textIdx += wcs[wcsIdx].length();
1355                    anyChars = false;
1356                }
1357                wcsIdx++;
1358            }
1359            // full match
1360            if (wcsIdx == wcs.length && textIdx == filename.length()) {
1361                return true;
1362            }
1363        } while (!backtrack.isEmpty());
1364        return false;
1365    }

1367    /**
1368     * Splits a string into a number of tokens.
1369     * The text is split by '?' and '*'.
1370     * Where multiple '*' occur consecutively they are collapsed into a single '*'.
1371     *
1372     * @param text the text to split
1373     * @return the array of tokens, never null
1374     */
1375    static String[] splitOnTokens(final String text) {
1376        // used by wildcardMatch
1377        // package level so a unit test may run on this
1378        if (text.indexOf('?') == NOT_FOUND && text.indexOf('*') == NOT_FOUND) {
1379            return new String[] {text};
1380        }
1381        final char[] array = text.toCharArray();
1382        final ArrayList<String> list = new ArrayList<>();
1383        final StringBuilder buffer = new StringBuilder();
1384        char prevChar = 0;
1385        for (final char ch : array) {
1386            if (ch == '?' || ch == '*') {
1387                if (buffer.length() != 0) {
1388                    list.add(buffer.toString());
1389                    buffer.setLength(0);
1390                }
1391                if (ch == '?') {
1392                    list.add("?");
1393                } else if (prevChar != '*') {
1394                    // ch == '*' here; check if previous char was '*'
1395                    list.add("*");
1396                }
1397            } else {
1398                buffer.append(ch);
1399            }
1400            prevChar = ch;
1401        }
1402        if (buffer.length() != 0) {
1403            list.add(buffer.toString());
1404        }
1405        return list.toArray(new String[list.size()]);
1406    }
