033            long end = System.nanoTime() + remainingNanos;
034            while (true) {
035                try {
036                    // TimeUnit.sleep() treats negative timeouts just like zero.
037                    NANOSECONDS.sleep(remainingNanos);
038                    return;
039                } catch (InterruptedException e) {
040                    interrupted = true;

043            }
044        } finally {
