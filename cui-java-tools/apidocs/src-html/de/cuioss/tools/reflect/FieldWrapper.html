035    /**
036     * Reads from the field determined by {@link #getField()}. It implicitly sets and resets the
037     * {@link Field#isAccessible()} flag.
038     *
039     * @param object to be read from
040     * @return an {@link Optional} on the given field value if applicable. May return
041     *         {@link Optional#empty()} for cases where:
042     *         <ul>
043     *         <li>Field value is {@code null}</li>
044     *         <li>Given Object is {@code null}</li>
045     *         <li>Given Object is improper type</li>
046     *         <li>an {@link IllegalAccessException} occurred while accessing</li>
047     *         </ul>
048     */
049    public Optional<Object> readValue(Object object) {
050        if (null == object) {
051            log.trace("No Object given, returning Optional#empty()");
052            return Optional.empty();
053        }
054        if (!declaringClass.isAssignableFrom(object.getClass())) {
055            log.trace("Given Object is improper type, returning Optional#empty()");
056            return Optional.empty();
057        }
058        var initialAccessible = field.canAccess(object);
059        log.trace("Reading from field \'{}\' with accessibleFlag=\'{}\' ", field, initialAccessible);
060        synchronized (field) {
061            if (!initialAccessible) {
062                log.trace("Explicitly setting accessible flag");
063                SecuritySupport.setAccessible(field, true);
064            }
065            try {
066                return Optional.ofNullable(field.get(object));
067            } catch (IllegalArgumentException | IllegalAccessException e) {
068                log.warn(e, "Reading from field \'{}\' with accessible=\'{}\' and parameter =\'{}\' could not complete", field, initialAccessible, object);
069                return Optional.empty();
070            } finally {
071                if (!initialAccessible) {
072                    log.trace("Resetting accessible flag");
073                    SecuritySupport.setAccessible(field, false);
074                }
075            }
076        }
077    }

099     * Writes to the field determined by {@link #getField()}. It implicitly sets and resets the
100     * {@link Field#isAccessible()} flag.
101     *
102     * @param object to be written to, must not be null
103     * @param value to be written, may be null
104     *
105     * @throws NullPointerException in case object is {@code null}
106     * @throws IllegalArgumentException in case the value is not applicable to the field
107     * @throws IllegalStateException wrapping an {@link IllegalAccessException}
108     */
109    public void writeValue(@NonNull Object object, Object value) {
110        if (object == null) {
111            throw new java.lang.NullPointerException("object is marked non-null but is null");
112        }
113        var initialAccessible = field.canAccess(object);
114        log.trace("Writing to field \'{}\' with accessibleFlag=\'{}\' ", field, initialAccessible);
115        synchronized (field) {
116            if (!initialAccessible) {
117                log.trace("Explicitly setting accessible flag");
118                SecuritySupport.setAccessible(field, true);
119            }
120            try {
121                field.set(object, value);
122            } catch (IllegalAccessException e) {
123                var message = MoreStrings.lenientFormat("Writing to field \'{}\' with accessible=\'{}\' and parameter =\'{}\' could not complete", field, initialAccessible, object);
124                throw new IllegalStateException(message, e);
125            } finally {
126                if (!initialAccessible) {
127                    log.trace("Resetting accessible flag");
128                    SecuritySupport.setAccessible(field, false);
129                }
130            }
131        }
