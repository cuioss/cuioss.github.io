054     * Tries to access a field on a given type. If none can be found it recursively calls itself
055     * with the corresponding parent until {@link Object}.
056     * <em>Caution:</em>
057     * <p>
058     * The field elements are shared between requests (cached), therefore you must
059     * ensure that changes to the instance, like {@link Field#setAccessible(boolean)} are reseted
060     * by the client. This can be simplified by using {@link FieldWrapper}
061     * </p>
062     *
063     * @param type to be checked, must not be null
064     * @param fieldName to be checked, must not be null
065     * @return an {@link Optional} {@link Field} if it can be found
066     */
067    @SuppressWarnings("java:S3824")
068    public static Optional<Field> accessField(final Class<?> type, final String fieldName) {
069        synchronized (MoreReflection.$LOCK) {
070            requireNonNull(type);
071            requireNonNull(fieldName);
072            if (!fieldCache.containsKey(type)) {
073                fieldCache.put(type, new HashMap<>());
074            }
075            final Map<String, Field> typeMap = fieldCache.get(type);
076            if (!typeMap.containsKey(fieldName)) {
077                typeMap.put(fieldName, resolveField(type, fieldName).orElse(null));
078            }
079            return Optional.ofNullable(typeMap.get(fieldName));
080        }
081    }

083    private static Optional<Field> resolveField(final Class<?> type, final String fieldName) {
084        try {
085            return Optional.of(type.getDeclaredField(fieldName));
086        } catch (final NoSuchFieldException | SecurityException e) {
087            log.trace("Error while trying to read field {} on type {}", type, fieldName, e);
088            if (Object.class.equals(type.getClass()) || null == type.getSuperclass()) {
089                return Optional.empty();
090            }
091            return resolveField(type.getSuperclass(), fieldName);
092        }
093    }

098     *
099     * @param clazz to be checked

102    public static List<Method> retrievePublicObjectMethods(final Class<?> clazz) {
103        synchronized (MoreReflection.$LOCK) {
104            requireNonNull(clazz);
105            if (!publicObjectMethodCache.containsKey(clazz)) {
106                final List<Method> found = new ArrayList<>();
107                for (final Method method : clazz.getMethods()) {
108                    final int modifiers = method.getModifiers();
109                    if (Modifier.isPublic(modifiers) && !Modifier.isStatic(modifiers) && !"getClass".equals(method.getName())) {
110                        found.add(method);
111                    }
112                }
113                publicObjectMethodCache.put(clazz, found);
114                return found;
115            }
116            return publicObjectMethodCache.get(clazz);
117        }
118    }

166     * public not static single-argument method that is prefixed with either "set" or consists of
167     * the propertyName only.

187     * @param assignableSource the type to be checked
188     * @param queryType to be checked for
189     * @return boolean indicating whether the given parameter, identified by their class attributes
190     *         match
191     */
192    public static boolean checkWhetherParameterIsAssignable(final Class<?> assignableSource, final Class<?> queryType) {
193        requireNonNull(assignableSource);
194        requireNonNull(queryType);
195        if (assignableSource.equals(queryType)) {
196            log.trace("Parameter-type matches exactly \'%s\'", assignableSource);
197            return true;
198        }
199        if (assignableSource.isAssignableFrom(queryType)) {
200            log.trace("Parameter \'%s\' is assignable from \'%s\'", assignableSource, queryType);
201            return true;
202        }
203        final Class<?> boxedSource = resolveWrapperTypeForPrimitive(assignableSource);
204        final Class<?> boxedQuery = resolveWrapperTypeForPrimitive(queryType);
205        if (boxedSource.equals(boxedQuery)) {
206            log.trace("Parameter-type matches exactly after autoboxing \'%s\'", assignableSource);
207            return true;
208        }
209        return boxedSource.isAssignableFrom(boxedQuery);
210    }

301     * Helper method that extract the property-name from a given accessor-method name.
302     *
303     * @param methodName must not be null nor empty
304     * @return the possible attribute name of a given method-name, e.g. it return 'name' for
305     *         getName/setName/isName. If
306     *         none of the prefixes 'get', 'set', 'is' is found it returns the passed String.
307     */
308    public static String computePropertyNameFromMethodName(final String methodName) {
309        requireNotEmpty(methodName);
310        if (methodName.startsWith("get") || methodName.startsWith("set")) {
311            if (methodName.length() > 3) {
312                return methodName.substring(3, 4).toLowerCase() + methodName.substring(4);
313            }
314            log.debug("Name to short for extracting attributeName \'{}\'", methodName);
315        }
316        if (methodName.startsWith("is")) {
317            if (methodName.length() > 2) {
318                return methodName.substring(2, 3).toLowerCase() + methodName.substring(3);
319            }
320            log.debug("Name to short for extracting attributeName \'{}\'", methodName);
321        }
322        return methodName;
323    }

338    public static <A extends Annotation> List<A> extractAllAnnotations(final Class<?> annotatedType, final Class<A> annotation) {

340            return Collections.emptyList();

342        final var builder = new CollectionBuilder<A>();
343        builder.add(annotatedType.getAnnotationsByType(annotation));
344        builder.add(extractAllAnnotations(annotatedType.getSuperclass(), annotation));

391     * @return if applicable the actual type argument for the given type. If the type represents

397    public static Optional<Class<?>> extractGenericTypeCovariantly(final Type type) {
398        if (null == type) {
399            log.trace("No KeyStoreType given, returning empty");
400            return Optional.empty();
401        }
402        if (type instanceof Class) {
403            log.debug("Found actual class returning as result {}", type);
404            return Optional.of((Class) type);
405        }
406        if (type instanceof ParameterizedType) {
407            log.debug("found Parameterized type, for {}, calling recursively", type);
408            return extractGenericTypeCovariantly(((ParameterizedType) type).getRawType());
409        }
410        log.warn("Unable to determines generic-type for {}", type);
411        return Optional.empty();

416     *
417     * @param typeToBeExtractedFrom must not be null
418     * @return an {@link Optional} of the {@link ParameterizedType} view of the given class.
419     */
420    public static Optional<ParameterizedType> extractParameterizedType(final Class<?> typeToBeExtractedFrom) {
421        log.debug("Extracting ParameterizedType from {}", typeToBeExtractedFrom);
422        if (null == typeToBeExtractedFrom) {
423            return Optional.empty();
424        }
425        if (Object.class.equals(typeToBeExtractedFrom)) {
426            log.debug("java.lang.Object is not a ParameterizedType");
427            return Optional.empty();
428        }
429        final var genericSuperclass = typeToBeExtractedFrom.getGenericSuperclass();
430        if (genericSuperclass instanceof ParameterizedType) {
431            return Optional.of((ParameterizedType) genericSuperclass);
432        }
433        // Check the tree
434        return extractParameterizedType(typeToBeExtractedFrom.getSuperclass());
435    }

473     * Tries to detect class from call stack which was the previous, before the marker class name
474     *
475     * @param throwable is an optional parameter, will be used to access the stack
476     * @param markerClasses class names which could be used as marker before the real caller name.
477     *            Collection must not be {@code null}.
478     * @return option of detected caller class name
479     */
480    public static Optional<StackTraceElement> findCallerElement(final Throwable throwable, final Collection<String> markerClasses) {
481        Objects.requireNonNull(markerClasses, "Marker class names are missing");
482        final StackTraceElement[] stackTraceElements;
483        if (null == throwable) {
484            stackTraceElements = Thread.currentThread().getStackTrace();
485        } else {
486            stackTraceElements = throwable.getStackTrace();
487        }
488        if (null == stackTraceElements || stackTraceElements.length < 5) {
489            return Optional.empty();
490        }
491        for (var index = 2; index < stackTraceElements.length; index++) {
492            final var element = stackTraceElements[index];
493            if (markerClasses.contains(element.getClassName())) {
494                if (stackTraceElements.length > index + 1) {
495                    return Optional.of(stackTraceElements[index + 1]);
496                }
497                return Optional.empty();
498            }
499        }
500        return Optional.empty();
501    }
