064    private List<String> cleanUp(final String... values) {
065        final List<String> result = new ArrayList<>(0);
066        if (null != values) {
067            for (final String item : values) {
068                final var value = emptyToNull(item);
069                if (null == value) {
070                    if (ValueHandling.FORMAT_IF_ALL_AVAILABLE.equals(handling)) {
071                        result.clear();
072                        break;
073                    }
074                } else {

078        }

082    private String getJoined(final List<String> values) {
083        if (isEmpty(values)) {
084            return null;
085        }
086        List<String> filtered = values.stream().filter(element -> !isEmpty(element)).collect(Collectors.toList());
087        if (isEmpty(filtered)) {
088            return null;
089        }
090        return String.join(separator, filtered);
091    }

094        final var joined = getJoined(values);
