091    private void doLog(final Logger logger, final String message, final Throwable throwable) {
092        // We go up the stack-trace until we found the call from CuiLogger.
093        final var caller = MoreReflection.findCallerElement(throwable, MARKER_CLASS_NAMES);
094        if (caller.isPresent()) {
095            // This is needed because otherwise LogRecord will assume this class and this method as
096            // the source of the log-statement

098        } else {
