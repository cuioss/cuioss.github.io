010/**
011 * Copied from commons.io:org.apache.commons.io.FilenameUtils
012 * <p>
013 * General filename and filepath manipulation utilities.
014 * <p>
015 * When dealing with filenames you can hit problems when moving from a Windows
016 * based development machine to a Unix based production machine.
017 * This class aims to help avoid those problems.
018 * <p>
019 * <b>NOTE</b>: You may be able to avoid using this class entirely simply by
020 * using JDK {@link java.io.File File} objects and the two argument constructor
021 * {@link java.io.File#File(java.io.File, java.lang.String) File(File,String)}.
022 * <p>
023 * Most methods on this class are designed to work the same on both Unix and Windows.
024 * Those that don't include 'System', 'Unix' or 'Windows' in their name.
025 * <p>
026 * Most methods recognise both separators (forward and back), and both
027 * sets of prefixes. See the javadoc of each method for details.
028 * <p>
029 * This class defines six components within a filename
030 * (example C:\dev\project\file.txt):
031 * <ul>
032 * <li>the prefix - C:\</li>
033 * <li>the path - dev\project\</li>
034 * <li>the full path - C:\dev\project\</li>
035 * <li>the name - file.txt</li>
036 * <li>the base name - file</li>
037 * <li>the extension - txt</li>
038 * </ul>
039 * Note that this class works best if directory filenames end with a separator.
040 * If you omit the last separator, it is impossible to determine if the filename
041 * corresponds to a file or a directory. As a result, we have chosen to say
042 * it corresponds to a file.
043 * <p>
044 * This class only supports Unix and Windows style names.
045 * Prefixes are matched as follows:
046 *
047 * <pre>
048 * Windows:
049 * a\b\c.txt           --&gt; ""          --&gt; relative
050 * \a\b\c.txt          --&gt; "\"         --&gt; current drive absolute
051 * C:a\b\c.txt         --&gt; "C:"        --&gt; drive relative
052 * C:\a\b\c.txt        --&gt; "C:\"       --&gt; absolute
053 * \\server\a\b\c.txt  --&gt; "\\server\" --&gt; UNC
054 *
055 * Unix:
056 * a/b/c.txt           --&gt; ""          --&gt; relative
057 * /a/b/c.txt          --&gt; "/"         --&gt; absolute
058 * ~/a/b/c.txt         --&gt; "~/"        --&gt; current user
059 * ~                   --&gt; "~/"        --&gt; current user (slash added)
060 * ~user/a/b/c.txt     --&gt; "~user/"    --&gt; named user
061 * ~user               --&gt; "~user/"    --&gt; named user (slash added)
062 * </pre>
063 *
064 * Both prefix styles are matched always, irrespective of the machine that you are
065 * currently running on.
066 * <p>
067 * Origin of code: Excalibur, Alexandria, Tomcat, Commons-Utils.
068 * </p>

322    /**
323     * Internal method to perform the normalization.
324     *
325     * @param filename the filename
326     * @param separator The separator character to use
327     * @param keepSeparator true to keep the final separator
328     * @return the normalized filename. Null bytes inside string will be removed.
329     */
330    @SuppressWarnings({"squid:S3776", "squid:LabelsShouldNotBeUsedCheck", "squid:ForLoopCounterChangedCheck"})
331    private static String doNormalize(final String filename, final char separator, final boolean keepSeparator) {
332        if (filename == null) {
333            return null;
334        }
335        failIfNullBytePresent(filename);
336        var size = filename.length();
337        if (size == 0) {
338            return filename;
339        }
340        final var prefix = getPrefixLength(filename);
341        if (prefix < 0) {
342            return null;
343        }
344        final var array = new char[size + 2]; // +1 for possible extra slash, +2 for arraycopy
345        filename.getChars(0, filename.length(), array, 0);
346        // fix separators throughout
347        final var otherSeparator = separator == SYSTEM_SEPARATOR ? OTHER_SEPARATOR : SYSTEM_SEPARATOR;
348        for (var i = 0; i < array.length; i++) {
349            if (array[i] == otherSeparator) {
350                array[i] = separator;
351            }
352        }
353        // add extra separator on the end to simplify code below
354        var lastIsDirectory = true;
355        if (array[size - 1] != separator) {
356            array[size] = separator;
357            size++;
358            lastIsDirectory = false;
359        }
360        // adjoining slashes
361        for (var i = prefix + 1; i < size; i++) {
362            if (array[i] == separator && array[i - 1] == separator) {
363                System.arraycopy(array, i, array, i - 1, size - i);
364                size--;
365                i--;
366            }
367        }
368        // dot slash
369        for (var i = prefix + 1; i < size; i++) {
370            if (array[i] == separator && array[i - 1] == '.' && (i == prefix + 1 || array[i - 2] == separator)) {
371                if (i == size - 1) {
372                    lastIsDirectory = true;
373                }
374                System.arraycopy(array, i + 1, array, i - 1, size - i);
375                size -= 2;
376                i--;
377            }
378        }
379        // double dot slash
380        outer:
381        for (var i = prefix + 2; i < size; i++) {
382            if (array[i] == separator && array[i - 1] == '.' && array[i - 2] == '.' && (i == prefix + 2 || array[i - 3] == separator)) {
383                if (i == prefix + 2) {
384                    return null;
385                }
386                if (i == size - 1) {
387                    lastIsDirectory = true;
388                }
389                int j;
390                for (j = i - 4; j >= prefix; j--) {
391                    if (array[j] == separator) {
392                        // remove b/../ from a/b/../c
393                        System.arraycopy(array, i + 1, array, j + 1, size - i);
394                        size -= i - j;
395                        i = j + 1;
396                        continue outer;
397                    }
398                }
399                // remove a/../ from a/../c
400                System.arraycopy(array, i + 1, array, prefix, size - i);
401                size -= i + 1 - prefix;
402                i = prefix + 1;
403            }
404        }
405        if (size <= 0) {
406            // should never be less than 0
407            return "";
408        }
409        if (size <= prefix || lastIsDirectory && keepSeparator) {
410            return new String(array, 0, size); // keep trailing separator
411        }
412        return new String(array, 0, size - 1); // lose trailing separator
413    }

417     * Concatenates a filename to a base path using normal command line style rules.
418     * <p>
419     * The effect is equivalent to resultant directory after changing
420     * directory to the first argument, followed by changing directory to
421     * the second argument.
422     * <p>
423     * The first argument is the base path, the second is the path to concatenate.
424     * The returned path is always normalized via {@link #normalize(String)},
425     * thus <code>..</code> is handled.
426     * <p>
427     * If <code>pathToAdd</code> is absolute (has an absolute prefix), then
428     * it will be normalized and returned.
429     * Otherwise, the paths will be joined, normalized and returned.
430     * <p>
431     * The output will be the same on both Unix and Windows except
432     * for the separator character.
433     *
434     * <pre>
435     * /foo/ + bar          --&gt;   /foo/bar
436     * /foo + bar           --&gt;   /foo/bar
437     * /foo + /bar          --&gt;   /bar
438     * /foo + C:/bar        --&gt;   C:/bar
439     * /foo + C:bar         --&gt;   C:bar (*)
440     * /foo/a/ + ../bar     --&gt;   foo/bar
441     * /foo/ + ../../bar    --&gt;   null
442     * /foo/ + /bar         --&gt;   /bar
443     * /foo/.. + /bar       --&gt;   /bar
444     * /foo + bar/c.txt     --&gt;   /foo/bar/c.txt
445     * /foo/c.txt + bar     --&gt;   /foo/c.txt/bar (!)
446     * </pre>
447     *
448     * (*) Note that the Windows relative drive prefix is unreliable when
449     * used with this method.
450     * (!) Note that the first parameter must be a path. If it ends with a name, then
451     * the name will be built into the concatenated path. If this might be a problem,
452     * use {@link #getFullPath(String)} on the base path argument.
453     *
454     * @param basePath the base path to attach to, always treated as a path
455     * @param fullFilenameToAdd the filename (or path) to attach to the base
456     * @return the concatenated path, or null if invalid. Null bytes inside string will be removed
457     */
458    public static String concat(final String basePath, final String fullFilenameToAdd) {
459        final var prefix = getPrefixLength(fullFilenameToAdd);
460        if (prefix < 0) {
461            return null;
462        }
463        if (prefix > 0) {
464            return normalize(fullFilenameToAdd);
465        }
466        if (basePath == null) {
467            return null;
468        }
469        final var len = basePath.length();
470        if (len == 0) {
471            return normalize(fullFilenameToAdd);
472        }
473        final var ch = basePath.charAt(len - 1);
474        if (isSeparator(ch)) {
475            return normalize(basePath + fullFilenameToAdd);
476        }
477        return normalize(basePath + '/' + fullFilenameToAdd);
478    }

481     * Determines whether the {@code parent} directory contains the {@code child} element (a file or
482     * directory).
483     * <p>
484     * The files names are expected to be normalized.
485     * </p>
486     *
487     * Edge cases:
488     * <ul>
489     * <li>A {@code directory} must not be null: if null, throw IllegalArgumentException</li>
490     * <li>A directory does not contain itself: return false</li>
491     * <li>A null child file is not contained in any parent: return false</li>
492     * </ul>
493     *
494     * @param canonicalParent
495     *            the file to consider as the parent.
496     * @param canonicalChild
497     *            the file to consider as the child.
498     * @return true is the candidate leaf is under by the specified composite. False otherwise.
499     */
500    public static boolean directoryContains(final String canonicalParent, final String canonicalChild) {
501        // Fail fast against NullPointerException
502        if (canonicalParent == null) {
503            throw new IllegalArgumentException("Directory must not be null");
504        }
505        if (canonicalChild == null || IOCase.SYSTEM.checkEquals(canonicalParent, canonicalChild)) {
506            return false;
507        }
508        return IOCase.SYSTEM.checkStartsWith(canonicalChild, canonicalParent);
509    }

556    /**
557     * Returns the length of the filename prefix, such as <code>C:/</code> or <code>~/</code>.
558     * <p>
559     * This method will handle a file in either Unix or Windows format.
560     * <p>
561     * The prefix length includes the first slash in the full filename
562     * if applicable. Thus, it is possible that the length returned is greater
563     * than the length of the input string.
564     *
565     * <pre>
566     * Windows:
567     * a\b\c.txt           --&gt; ""          --&gt; relative
568     * \a\b\c.txt          --&gt; "\"         --&gt; current drive absolute
569     * C:a\b\c.txt         --&gt; "C:"        --&gt; drive relative
570     * C:\a\b\c.txt        --&gt; "C:\"       --&gt; absolute
571     * \\server\a\b\c.txt  --&gt; "\\server\" --&gt; UNC
572     * \\\a\b\c.txt        --&gt;  error, length = -1
573     *
574     * Unix:
575     * a/b/c.txt           --&gt; ""          --&gt; relative
576     * /a/b/c.txt          --&gt; "/"         --&gt; absolute
577     * ~/a/b/c.txt         --&gt; "~/"        --&gt; current user
578     * ~                   --&gt; "~/"        --&gt; current user (slash added)
579     * ~user/a/b/c.txt     --&gt; "~user/"    --&gt; named user
580     * ~user               --&gt; "~user/"    --&gt; named user (slash added)
581     * //server/a/b/c.txt  --&gt; "//server/"
582     * ///a/b/c.txt        --&gt; error, length = -1
583     * </pre>
584     * <p>
585     * The output will be the same irrespective of the machine that the code is running on.
586     * i.e. both Unix and Windows prefixes are matched regardless.
587     * <p>
588     * Note that a leading // (or \\) is used to indicate a UNC name on Windows.
589     * These must be followed by a server name, so double-slashes are not collapsed
590     * to a single slash at the start of the filename.
591     *
592     * @param filename the filename to find the prefix in, null returns -1
593     * @return the length of the prefix, -1 if invalid or null
594     */
595    @SuppressWarnings({"squid:S3776"})
596    public static int getPrefixLength(final String filename) {
597        if (filename == null) {
598            return NOT_FOUND;
599        }
600        final var len = filename.length();
601        if (len == 0) {
602            return 0;
603        }
604        var ch0 = filename.charAt(0);
605        if (ch0 == ':') {
606            return NOT_FOUND;
607        }
608        if (len == 1) {
609            if (ch0 == '~') {
610                return 2; // return a length greater than the input
611            }
612            return isSeparator(ch0) ? 1 : 0;
613        }
614        if (ch0 == '~') {
615            var posUnix = filename.indexOf(UNIX_SEPARATOR, 1);
616            var posWin = filename.indexOf(WINDOWS_SEPARATOR, 1);
617            if (posUnix == NOT_FOUND && posWin == NOT_FOUND) {
618                return len + 1; // return a length greater than the input
619            }
620            posUnix = posUnix == NOT_FOUND ? posWin : posUnix;
621            posWin = posWin == NOT_FOUND ? posUnix : posWin;
622            return Math.min(posUnix, posWin) + 1;
623        }
624        final var ch1 = filename.charAt(1);
625        if (ch1 == ':') {
626            ch0 = Character.toUpperCase(ch0);
627            if (ch0 >= 'A' && ch0 <= 'Z') {
628                if (len == 2 || !isSeparator(filename.charAt(2))) {
629                    return 2;
630                }
631                return 3;
632            }
633            if (ch0 == UNIX_SEPARATOR) {
634                return 1;
635            }
636            return NOT_FOUND;
637        }
638        if (!isSeparator(ch0) || !isSeparator(ch1)) {
639            return isSeparator(ch0) ? 1 : 0;
640        }
641        var posUnix = filename.indexOf(UNIX_SEPARATOR, 2);
642        var posWin = filename.indexOf(WINDOWS_SEPARATOR, 2);
643        if (posUnix == NOT_FOUND && posWin == NOT_FOUND || posUnix == 2 || posWin == 2) {
644            return NOT_FOUND;
645        }
646        posUnix = posUnix == NOT_FOUND ? posWin : posUnix;
647        posWin = posWin == NOT_FOUND ? posUnix : posWin;
648        return Math.min(posUnix, posWin) + 1;
649    }

915        if (end == 0) {
916            end++;
917        }
918        return filename.substring(0, end);

1277    /**
1278     * Checks a filename to see if it matches the specified wildcard matcher
1279     * allowing control over case-sensitivity.
1280     * <p>
1281     * The wildcard matcher uses the characters '?' and '*' to represent a
1282     * single or multiple (zero or more) wildcard characters.
1283     * N.B. the sequence "*?" does not work properly at present in match strings.
1284     *
1285     * @param filename the filename to match on
1286     * @param wildcardMatcher the wildcard string to match against
1287     * @param caseSensitivity what case sensitivity rule to use, null means case-sensitive
1288     * @return true if the filename matches the wildcard string
1289     */
1290    @SuppressWarnings({"squid:S3776", "squid:S135"})
1291    public static boolean wildcardMatch(final String filename, final String wildcardMatcher, IOCase caseSensitivity) {
1292        if (filename == null && wildcardMatcher == null) {
1293            return true;
1294        }
1295        if (filename == null || wildcardMatcher == null) {
1296            return false;
1297        }
1298        if (caseSensitivity == null) {
1299            caseSensitivity = IOCase.SENSITIVE;
1300        }
1301        final var wcs = splitOnTokens(wildcardMatcher);
1302        var anyChars = false;
1303        var textIdx = 0;
1304        var wcsIdx = 0;
1305        final Deque<int[]> backtrack = new ArrayDeque<>();
1306        // loop around a backtrack stack, to handle complex * matching
1307        do {
1308            if (!backtrack.isEmpty()) {
1309                final var array = backtrack.pop();
1310                wcsIdx = array[0];
1311                textIdx = array[1];
1312                anyChars = true;
1313            }
1314            // loop whilst tokens and text left to process
1315            while (wcsIdx < wcs.length) {
1316                if (wcs[wcsIdx].equals("?")) {
1317                    // ? so move to next text char
1318                    textIdx++;
1319                    if (textIdx > filename.length()) {
1320                        break;
1321                    }
1322                    anyChars = false;
1323                } else if (wcs[wcsIdx].equals("*")) {
1324                    // set any chars status
1325                    anyChars = true;
1326                    if (wcsIdx == wcs.length - 1) {
1327                        textIdx = filename.length();
1328                    }
1329                } else {
1330                    // matching text token
1331                    if (anyChars) {
1332                        // any chars then try to locate text token
1333                        textIdx = caseSensitivity.checkIndexOf(filename, textIdx, wcs[wcsIdx]);
1334                        if (textIdx == NOT_FOUND) {
1335                            // token not found
1336                            break;
1337                        }
1338                        final var repeat = caseSensitivity.checkIndexOf(filename, textIdx + 1, wcs[wcsIdx]);
1339                        if (repeat >= 0) {
1340                            backtrack.push(new int[] {wcsIdx, repeat});
1341                        }
1342                    } else  // matching from current position
1343                    if (!caseSensitivity.checkRegionMatches(filename, textIdx, wcs[wcsIdx])) {
1344                        // couldn't match token
1345                        break;
1346                    }
1347                    // matched text token, move text index to end of matched token
1348                    textIdx += wcs[wcsIdx].length();
1349                    anyChars = false;
1350                }
1351                wcsIdx++;
1352            }
1353            // full match
1354            if (wcsIdx == wcs.length && textIdx == filename.length()) {
1355                return true;
1356            }
1357        } while (!backtrack.isEmpty());
1358        return false;
1359    }

1361    /**
1362     * Splits a string into a number of tokens.
1363     * The text is split by '?' and '*'.
1364     * Where multiple '*' occur consecutively they are collapsed into a single '*'.
1365     *
1366     * @param text the text to split
1367     * @return the array of tokens, never null
1368     */
1369    static String[] splitOnTokens(final String text) {
1370        // used by wildcardMatch
1371        // package level so a unit test may run on this
1372        if (text.indexOf('?') == NOT_FOUND && text.indexOf('*') == NOT_FOUND) {
1373            return new String[] {text};
1374        }
1375        final var array = text.toCharArray();
1376        final var list = new ArrayList<String>();
1377        final var buffer = new StringBuilder();
1378        char prevChar = 0;
1379        for (final char ch : array) {
1380            if (ch == '?' || ch == '*') {
1381                if (buffer.length() != 0) {
1382                    list.add(buffer.toString());
1383                    buffer.setLength(0);
1384                }
1385                if (ch == '?') {
1386                    list.add("?");
1387                } else if (prevChar != '*') {
1388                    // ch == '*' here; check if previous char was '*'
1389                    list.add("*");
1390                }
1391            } else {
1392                buffer.append(ch);
1393            }
1394            prevChar = ch;
1395        }
1396        if (buffer.length() != 0) {
1397            list.add(buffer.toString());
1398        }
1399        return list.toArray(new String[list.size()]);
1400    }
