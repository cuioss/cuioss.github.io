023     * Returns an implementation of {@link FileLoader} matching to the given path.
024     *
025     * @param pathName must not be null or empty.
026     * @return a configured implementation of {@link FileLoader}. In case the pathName is prefixed
027     *         with {@link FileTypePrefix#CLASSPATH} it returns a {@link ClassPathLoader}. If
028     *         prefixed with
029     *         {@link FileTypePrefix#URL} it returns a {@link UrlLoader}. Otherwise, it returns a
030     *         {@link FileSystemLoader}.
031     */
032    public static FileLoader getLoaderForPath(final String pathName) {
033        if (isEmpty(pathName)) {
034            throw new IllegalArgumentException("pathName must not be null nor empty");
035        }
036        if (pathName.startsWith(FileTypePrefix.CLASSPATH.getPrefix())) {
037            return new ClassPathLoader(pathName);
038        }
039        if (pathName.startsWith(FileTypePrefix.URL.getPrefix())) {
040            return new UrlLoader(pathName);
041        }
042        return new FileSystemLoader(pathName);
043    }

059    @SuppressWarnings("java:S5443")
060    public static Path copyFileToTemp(final FileLoader source, final boolean markDeleteOnExit) throws IOException {
061        checkArgument(null != source, "Attribute with name source must not be null");
062        checkArgument(source.isReadable(), "Source must be readable");
063        final var target = Files.createTempFile(source.getFileName().getNamePart(), source.getFileName().getSuffix());
064        try (var inputStream = source.inputStream()) {
065            Files.copy(new BufferedInputStream(inputStream), target, StandardCopyOption.REPLACE_EXISTING);
066        }
067        if (markDeleteOnExit) {
068            target.toFile().deleteOnExit();
069        }
070        return target;
