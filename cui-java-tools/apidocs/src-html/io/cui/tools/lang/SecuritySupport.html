034    public static Optional<ClassLoader> getContextClassLoader() {
035        if (null == System.getSecurityManager()) {
036            LOGGER.trace("No SecurityManager configured, accessing context-class-loader");
037            return Optional.ofNullable(Thread.currentThread().getContextClassLoader());
038        } else {
039            LOGGER.trace(SECURITY_MANAGER_CONFIGURED);
040            return AccessController.doPrivileged((PrivilegedAction<Optional<ClassLoader>>) () -> {
041                try {
042                    return Optional.ofNullable(Thread.currentThread().getContextClassLoader());
043                } catch (SecurityException e) {

045                    return Optional.empty();
