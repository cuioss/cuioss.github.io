057     * represent any given

059     *
060     * @param data An array of characters containing hexadecimal digits

064    @SuppressWarnings("squid:ForLoopCounterChangedCheck")
065    public static byte[] decodeHex(final char[] data) throws DecoderException {
066        final int len = data.length;
067        if ((len & 1) != 0) {
068            throw new DecoderException("Odd number of characters.");
069        }
070        final byte[] out = new byte[len >> 1];
071        // two characters form the hex value.
072        for (int i = 0, j = 0; j < len; i++) {
073            int f = toDigit(data[j], j) << 4;
074            j++;
075            f = f | toDigit(data[j], j);
076            j++;
077            out[i] = (byte) (f & 255);
078        }
079        return out;
080    }

269    private static byte[] toByteArray(final ByteBuffer byteBuffer) {
270        final int remaining = byteBuffer.remaining();
271        // Use the underlying buffer if possible
272        if (byteBuffer.hasArray()) {
273            final byte[] byteArray = byteBuffer.array();
274            if (remaining == byteArray.length) {
275                byteBuffer.position(remaining);
276                return byteArray;
277            }
278        }
279        // Copy the bytes
280        final byte[] byteArray = new byte[remaining];
281        byteBuffer.get(byteArray);
282        return byteArray;
283    }
