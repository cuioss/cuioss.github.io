092    private void doLog(final Logger logger, final String message, final Throwable throwable) {
093        // We go up the stack-trace until we found the call from CuiLogger.
094        final Optional<StackTraceElement> caller = MoreReflection.findCallerElement(throwable, MARKER_CLASS_NAMES);
095        if (caller.isPresent()) {
096            // This is needed because otherwise LogRecord will assume this class and this method as
097            // the source of the log-statement
098            logger.logp(getJuliLevel(), caller.get().getClassName(), caller.get().getMethodName(), message, throwable);
099        } else {

101        }
