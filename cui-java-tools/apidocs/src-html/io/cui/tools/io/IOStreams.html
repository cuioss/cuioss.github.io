057    /**
058     * Compares the contents of two Streams to determine if they are equal or
059     * not.
060     * <p>
061     * This method buffers the input internally using
062     * <code>BufferedInputStream</code> if they are not already buffered.
063     *
064     * @param input1 the first stream
065     * @param input2 the second stream
066     * @return true if the content of the streams are equal or they both don't
067     *         exist, false otherwise
068     * @throws NullPointerException if either input is null
069     * @throws IOException if an I/O error occurs
070     */
071    public static boolean contentEquals(InputStream input1, InputStream input2) throws IOException {
072        if (input1 == input2) {
073            return true;
074        }
075        if (!(input1 instanceof BufferedInputStream)) {
076            input1 = new BufferedInputStream(input1);
077        }
078        if (!(input2 instanceof BufferedInputStream)) {
079            input2 = new BufferedInputStream(input2);
080        }
081        int ch = input1.read();
082        while (EOF != ch) {
083            final int ch2 = input2.read();
084            if (ch != ch2) {
085                return false;
086            }
087            ch = input1.read();
088        }
089        final int ch2 = input2.read();
090        return ch2 == EOF;
091    }

320     *
321     * @param input the <code>InputStream</code> to read from
322     * @param output the <code>OutputStream</code> to write to
323     * @param inputOffset number of bytes to skip from input before copying
324     *            -ve values are ignored
325     * @param length number of bytes to copy. -ve means all
326     * @param buffer the buffer to use for the copy
327     * @return the number of bytes copied
328     * @throws NullPointerException if the input or output is null
329     * @throws IOException if an I/O error occurs

331    public static long copyLarge(final InputStream input, final OutputStream output, final long inputOffset, final long length, final byte[] buffer) throws IOException {
332        if (inputOffset > 0) {
333            skipFully(input, inputOffset);
334        }
335        if (length == 0) {
336            return 0;
337        }
338        final int bufferLength = buffer.length;
339        int bytesToRead = bufferLength;
340        if (length > 0 && length < bufferLength) {
341            bytesToRead = (int) length;
342        }
343        int read;
344        long totalRead = 0;
345        while (bytesToRead > 0 && EOF != (read = input.read(buffer, 0, bytesToRead))) {
346            output.write(buffer, 0, read);
347            totalRead += read;
348            if (length > 0) {
349                // only adjust length if not reading to the end
350                // Note the cast must work because buffer.length is an integer
351                bytesToRead = (int) Math.min(length - totalRead, bufferLength);
352            }
353        }
354        return totalRead;
355    }

632     * @param input byte stream to skip
633     * @param toSkip number of bytes to skip.
634     * @return number of bytes actually skipped.
635     * @throws IOException if there is a problem reading the file
636     * @throws IllegalArgumentException if toSkip is negative
637     * @see InputStream#skip(long)
638     * @see <a href="https://issues.apache.org/jira/browse/IO-203">IO-203 - Add skipFully() method
639     *      for InputStreams</a>
640     */
641    public static long skip(final InputStream input, final long toSkip) throws IOException {
642        if (toSkip < 0) {
643            throw new IllegalArgumentException(SKIP_COUNT_MUST_BE_NON_NEGATIVE_ACTUAL + toSkip);
644        }
645        /*
646         * N.B. no need to synchronize this because: - we don't care if the buffer is created
647         * multiple times (the data
648         * is ignored) - we always use the same size buffer, so if it is recreated it will still
649         * be OK (if the buffer
650         * size were variable, we would need to synch. to ensure some other thread did not create a
651         * smaller one)
652         */
653        if (skipByteBuffer == null) {
654            skipByteBuffer = new byte[SKIP_BUFFER_SIZE];
655        }
656        long remain = toSkip;
657        while (remain > 0) {
658            // See https://issues.apache.org/jira/browse/IO-203 for why we use read() rather than
659            // delegating to skip()
660            final long n = input.read(skipByteBuffer, 0, (int) Math.min(remain, SKIP_BUFFER_SIZE));
661            if (n < 0) {
662                // EOF
663                break;
664            }
665            remain -= n;
666        }
667        return toSkip - remain;
668    }
