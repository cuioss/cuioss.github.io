094    public static boolean checkAccessiblePath(@NonNull final Path path, final boolean checkForDirectory, final boolean verbose) {
095        if (path == null) {
096            throw new java.lang.NullPointerException("path is marked non-null but is null");
097        }
098        if (!checkReadablePath(path, checkForDirectory, verbose)) {
099            return false;

102        final String absolutePath = pathFile.getAbsolutePath();
103        if (!pathFile.canWrite()) {
104            if (verbose) {
105                log.warn(MSG_DIRECTORY_NOT_ACCESSIBLE, absolutePath, "Not Writable");
106            }
107            return false;
108        }
109        log.debug("{} denotes an existing file / directory with read and write permissions", absolutePath);
110        return true;
111    }

203    public static Path getBackupDirectoryForPath(final Path directory) {
204        if (!checkAccessiblePath(directory, true, true)) {
205            throw new IllegalArgumentException(String.format("Given path \'%s\' does not denote an existing writable directory", directory.toFile().getAbsolutePath()));
206        }
207        final Path backup = directory.resolve(BACKUP_DIR_NAME);
208        final File backupAsFile = backup.toFile();
209        if (!backupAsFile.exists() && !backupAsFile.mkdir()) {
210            throw new IllegalStateException(String.format("Unable to create directory \'%s\'", backup.toFile().getAbsolutePath()));
211        }
212        return backup;
213    }

216     * Backups the file, identified by the given path into the backup directory, derived with
217     * {@link #getBackupDirectoryForPath(Path)}. The original file attributes will be applied to the
218     * copied filed, See {@link StandardCopyOption#COPY_ATTRIBUTES}.
219     *
220     * @param path must not be null and denote an existing read and writable file
221     * @return Path on the newly created file
222     * @throws IOException if an I/O error occurs
223     */
224    public static Path backupFile(final Path path) throws IOException {
225        assertAccessibleFile(path);
226        Path backupDir = getBackupDirectoryForPath(path.getParent());
227        Path backupFile = createNonExistingPath(backupDir, path.getFileName() + BACKUP_FILE_SUFFIX + new SimpleDateFormat("yyyyMMddHHmmss").format(new Date()));
228        Files.copy(path, backupFile, StandardCopyOption.REPLACE_EXISTING, StandardCopyOption.COPY_ATTRIBUTES);
229        log.debug("Created backup from \'{}\' at \'{}\'", path.toFile().getAbsolutePath(), backupFile.toFile().getAbsolutePath());
230        return backupFile;
231    }

269    static Path createNonExistingPath(final Path parentDir, final String fileName) {
270        Path backupFile = parentDir.resolve(fileName);
271        if (!backupFile.toFile().exists()) {
272            return backupFile;
273        }
274        for (int counter = 1; counter < 20; counter++) {
275            String newName = fileName + "_" + counter;
276            Path newBackupFile = parentDir.resolve(newName);
277            if (!newBackupFile.toFile().exists()) {
278                return newBackupFile;
279            }
280        }
281        throw new IllegalStateException(String.format("Unable to determine a non-existing file within \'%s\' for file-name \'%s\'", parentDir.toFile().getAbsolutePath(), fileName));
282    }

285     * Deletes a file, never throwing an exception. If file is a directory, delete it and all
286     * subdirectories.
287     * Inspired by org.apache.commons.io.FileUtils#deleteQuietly
288     * <p>
289     * The difference between File.delete() and this method are:
290     * <ul>
291     * <li>A directory to be deleted does not have to be empty.</li>
292     * <li>No exceptions are thrown when a file or directory cannot be deleted.</li>
293     * </ul>
294     *
295     * @param path file or directory to delete, can be {@code null}
296     * @return {@code true} if the file or directory was deleted, otherwise
297     *         {@code false}
298     */
299    public static boolean deleteQuietly(final Path path) {
300        log.trace("Deleting file {}", path);
301        if (path == null) {
302            return false;
303        }
304        File file = path.toFile();
305        final String absolutePath = file.getAbsolutePath();
306        if (!file.exists()) {
307            log.trace("Path {} does not exist", absolutePath);
308            return false;
309        }
310        boolean recursiveSucceful = true;
311        try {
312            if (file.isDirectory()) {
313                log.trace("Path {} is directory, checking children", absolutePath);
314                for (String child : file.list()) {
315                    if (!deleteQuietly(path.resolve(child))) {
316                        recursiveSucceful = false;
317                    }
318                }
319            }
320        } catch (final Exception e) {
321            log.trace(e, "Unable to check Path {} whether it is a directory", absolutePath);
322        }
323        try {
324            if (Files.deleteIfExists(path)) {
325                log.trace("Successully deleted path {}", absolutePath);
326            } else {
327                recursiveSucceful = false;
328            }
329        } catch (final Exception e) {
330            log.trace(e, "Unable to delete Path {}", absolutePath);
331            return false;
332        }
333        return recursiveSucceful;
334    }

336    /**
337     * Compares the contents of two files to determine if they are equal or not.
338     * <p>
339     * This method checks to see if the two files are different lengths
340     * or if they point to the same file, before resorting to byte-by-byte
341     * comparison of the contents.
342     * <p>
343     * Taken from org.apache.commons.io.FileUtils.contentEquals(File, File)
344     * Code origin: Avalon
345     *
346     * @param path1 the first file
347     * @param path2 the second file
348     * @return true if the content of the files are equal or they both don't
349     *         exist, false otherwise
350     * @throws IOException in case of an I/O error
351     */
352    public static boolean contentEquals(final Path path1, final Path path2) throws IOException {
353        requireNonNull(path1);
354        requireNonNull(path2);
355        File file1 = path1.toFile();
356        File file2 = path2.toFile();
357        final boolean file1Exists = file1.exists();
358        if (file1Exists != file2.exists()) {
359            return false;
360        }
361        if (!file1Exists) {
362            // two not existing files are equal
363            return true;
364        }
365        if (file1.isDirectory() || file2.isDirectory()) {
366            // don't want to compare directory contents
367            throw new IOException("Can\'t compare directories, only files");
368        }
369        if (file1.length() != file2.length()) {
370            // lengths differ, cannot be equal
371            return false;
372        }
373        if (file1.getCanonicalFile().equals(file2.getCanonicalFile())) {
374            // same file
375            return true;
376        }
377        try (
378            InputStream input1 = new FileInputStream(file1);
379            InputStream input2 = new FileInputStream(file2)) {
380            return IOStreams.contentEquals(input1, input2);
381        }
382    }

425     * Checks, if the two given paths are pointing to the same location.
426     * <p>
427     * If both paths are not {@code null} and do {@link File#exists()}, the
428     * {@link Files#isSameFile(Path, Path)}
429     * method is used to check if both paths are pointing to the same location.
430     * Otherwise, if one of the paths does not exist, the {@link Paths#equals(Object)} method is
431     * used.
432     *
433     * @param path to be compared with path2
434     * @param path2 to be compared with path
435     *
436     * @return {@code true}, if both paths are {@code null}.
437     *         {@code true}, if both paths not {@code null}, do exist, and
438     *         {@link Files#isSameFile(Path, Path)}.
439     *         {@code true}, if both paths not {@code null} and {@link Paths#equals(Object)}
440     *         {@code false} otherwise.
441     */
442    public static boolean isSameFile(Path path, Path path2) {
443        if (null == path && null == path2) {
444            return true;
445        }
446        if (null != path && null != path2) {
447            if (path.toFile().exists() && path2.toFile().exists()) {
448                try {
449                    return Files.isSameFile(path, path2);
450                } catch (final IOException e) {
451                    log.error(e, "Portal-123: Unable to compare path_a={} and path_b={}", path, path2);
452                }
453            } else {
454                log.debug("Comparing paths with #equals, as at least one path does not exist. " + "path_a={}, path_b={}", path, path2);
455                return path.equals(path2);
456            }
457        } else {
458            log.trace("at least one path is null: path_a={}, path_b={}", path, path2);
459        }
460        return false;
461    }
