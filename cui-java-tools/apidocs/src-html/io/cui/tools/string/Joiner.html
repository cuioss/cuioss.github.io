142    private String doJoin(Iterable<?> parts) {
143        log.trace("Joining elements with configuration {}", joinerConfig);
144        if (isEmpty(parts)) {
145            return "";
146        }
147        ArrayList<CharSequence> builder = new ArrayList<>();
148        for (Object element : parts) {
149            if (null == element) {
150                if (!joinerConfig.isSkipNulls()) {
151                    builder.add(joinerConfig.getUseForNull());
152                }
153            } else {
154                if (element instanceof CharSequence) {
155                    builder.add((CharSequence) element);
156                } else {
157                    builder.add(MoreStrings.lenientToString(element));
158                }
159            }
160        }
161        if (joinerConfig.isSkipEmpty()) {
162            builder = builder.stream().filter(element -> !MoreStrings.isEmpty(element)).collect(Collectors.toCollection(ArrayList::new));
163        }
164        if (joinerConfig.isSkipBlank()) {
165            builder = builder.stream().filter(element -> !isBlank(element)).collect(Collectors.toCollection(ArrayList::new));
166        }
167        return String.join(joinerConfig.getSeparator(), builder);
168    }
