230        log.trace("Parsing Query String %s", queryString);
231        if (MoreStrings.isEmpty(queryString)) {
232            return Collections.emptyList();
233        }
234        String cleaned = queryString.trim();
235        if (cleaned.startsWith("?")) {
236            cleaned = cleaned.substring(1);
237        }
238        if (MoreStrings.isEmpty(cleaned)) {
239            log.debug("Given String solely consists of \'?\' symbol, ignoring");
240            return Collections.emptyList();
241        }
242        List<String> elements = Splitter.on("&").omitEmptyStrings().splitToList(cleaned);
243        CollectionBuilder<UrlParameter> builder = new CollectionBuilder<>();
244        for (String element : elements) {
245            if (element.contains("=")) {
246                List<String> splitted = Splitter.on("=").omitEmptyStrings().splitToList(element);
247                switch (splitted.size()) {
248                case 0: 
249                    log.debug("Unable to parse queryString \'%s\' correctly, unable to extract key-value-pair for element \'%s\'", queryString, element);
250                    break;
251                case 1: 
252                    builder.add(createDecoded(splitted.get(0), null));
253                    break;
254                case 2: 
255                    builder.add(createDecoded(splitted.get(0), splitted.get(1)));
256                    break;
257                default: 
258                    log.debug("Unable to parse queryString \'%s\' correctly, multiple \'=\' symbols found at unexpected locations", queryString);

260                }
261            } else {

264        }
